036/**
037 * This program is the CLI utility for the CredentialProvider facilities in
038 * Hadoop.
039 */
040public class CredentialShell extends CommandShell {
041  final static private String USAGE_PREFIX = "Usage: hadoop credential " +
042      "[generic options]\n";
043  final static private String COMMANDS =
044      "   [-help]\n" +
045      "   [" + CreateCommand.USAGE + "]\n" +
046      "   [" + DeleteCommand.USAGE + "]\n" +
047      "   [" + ListCommand.USAGE + "]\n";
048  @VisibleForTesting
049  public static final String NO_VALID_PROVIDERS =
050      "There are no valid (non-transient) providers configured.\n" +
051      "No action has been taken. Use the -provider option to specify\n" +
052      "a provider. If you want to use a transient provider then you\n" +
053      "MUST use the -provider argument.";
054
055  private boolean interactive = true;
056
057  /** If true, fail if the provider requires a password and none is given. */
058  private boolean strict = false;
059
060  private boolean userSuppliedProvider = false;
061  private String value = null;
062  private PasswordReader passwordReader;
063
064  /**
065   * Parse the command line arguments and initialize the data.
066   * <pre>
067   * % hadoop credential create alias [-provider providerPath]
068   * % hadoop credential list [-provider providerPath]
069   * % hadoop credential delete alias [-provider providerPath] [-f]
070   * </pre>
071   * @param args
072   * @return 0 if the argument(s) were recognized, 1 otherwise
073   * @throws IOException
074   */
075  @Override
076  protected int init(String[] args) throws IOException {
077    // no args should print the help message
078    if (0 == args.length) {
079      ToolRunner.printGenericCommandUsage(getErr());
080      return 1;
081    }
082
083    for (int i = 0; i < args.length; i++) { // parse command line
084      if (args[i].equals("create")) {
085        if (i == args.length - 1) {
086          return 1;
087        }
088        setSubCommand(new CreateCommand(args[++i]));
089      } else if (args[i].equals("delete")) {
090        if (i == args.length - 1) {
091          return 1;
092        }
093        setSubCommand(new DeleteCommand(args[++i]));
094      } else if (args[i].equals("list")) {
095        setSubCommand(new ListCommand());
096      } else if (args[i].equals("-provider")) {
097        if (i == args.length - 1) {
098          return 1;
099        }
100        userSuppliedProvider = true;
101        getConf().set(CredentialProviderFactory.CREDENTIAL_PROVIDER_PATH,
102            args[++i]);
103      } else if (args[i].equals("-f") || (args[i].equals("-force"))) {
104        interactive = false;
105      } else if (args[i].equals("-strict")) {
106        strict = true;
107      } else if (args[i].equals("-v") || (args[i].equals("-value"))) {
108        value = args[++i];
109      } else if (args[i].equals("-help")) {
110        printShellUsage();
111        return 0;
112      } else {
113        ToolRunner.printGenericCommandUsage(getErr());
114        return 1;
115      }
116    }
117    return 0;
118  }
119
120  @Override
121  public String getCommandUsage() {
122    StringBuffer sbuf = new StringBuffer(USAGE_PREFIX + COMMANDS);
123    String banner = StringUtils.repeat("=", 66);
124    sbuf.append(banner + "\n");
125    sbuf.append(CreateCommand.USAGE + ":\n\n" + CreateCommand.DESC + "\n");
126    sbuf.append(banner + "\n");
127    sbuf.append(DeleteCommand.USAGE + ":\n\n" + DeleteCommand.DESC + "\n");
128    sbuf.append(banner + "\n");
129    sbuf.append(ListCommand.USAGE + ":\n\n" + ListCommand.DESC + "\n");
130    return sbuf.toString();
131  }
132
133  private abstract class Command extends SubCommand {
134    protected CredentialProvider provider = null;
135
136    protected CredentialProvider getCredentialProvider() {
137      CredentialProvider prov = null;
138      List<CredentialProvider> providers;
139      try {
140        providers = CredentialProviderFactory.getProviders(getConf());
141        if (userSuppliedProvider) {
142          prov = providers.get(0);
143        } else {
144          for (CredentialProvider p : providers) {
145            if (!p.isTransient()) {
146              prov = p;
147              break;
148            }
149          }
150        }
151      } catch (IOException e) {
152        e.printStackTrace(getErr());
153      }
154      if (prov == null) {
155        getOut().println(NO_VALID_PROVIDERS);
156      }
157      return prov;
158    }
159
160    protected void printProviderWritten() {
161      getOut().println("Provider " + provider.toString() + " was updated.");
162    }
163
164    protected void warnIfTransientProvider() {
165      if (provider.isTransient()) {
166        getOut().println("WARNING: you are modifying a transient provider.");
167      }
168    }
169
170    protected void doHelp() {
171      getOut().println(USAGE_PREFIX + COMMANDS);
172      printShellUsage();
173    }
174
175    public abstract void execute() throws Exception;
176
177    public abstract String getUsage();
178  }
179
180  private class ListCommand extends Command {
181    public static final String USAGE =
182        "list [-provider provider-path] [-strict]";
183    public static final String DESC =
184        "The list subcommand displays the aliases contained within \n" +
185        "a particular provider - as configured in core-site.xml or\n" +
186        "indicated through the -provider argument. If -strict is supplied,\n" +
187        "fail immediately if the provider requires a password and none is\n" +
188        "provided.";
189
190    public boolean validate() {
191      provider = getCredentialProvider();
192      return (provider != null);
193    }
194
195    public void execute() throws IOException {
196      List<String> aliases;
197      try {
198        aliases = provider.getAliases();
199        getOut().println("Listing aliases for CredentialProvider: " +
200            provider.toString());
201        for (String alias : aliases) {
202          getOut().println(alias);
203        }
204      } catch (IOException e) {
205        getOut().println("Cannot list aliases for CredentialProvider: " +
206            provider.toString()
207            + ": " + e.getMessage());
208        throw e;
209      }
210    }
211
212    @Override
213    public String getUsage() {
214      return USAGE + ":\n\n" + DESC;
215    }
216  }

218  private class DeleteCommand extends Command {
219    public static final String USAGE =
220        "delete <alias> [-f] [-provider provider-path] [-strict]";
221    public static final String DESC =
222        "The delete subcommand deletes the credential\n" +
223        "specified as the <alias> argument from within the provider\n" +
224        "indicated through the -provider argument. The command asks for\n" +
225        "confirmation unless the -f option is specified. If -strict is\n" +
226        "supplied, fail immediately if the provider requires a password\n" +
227        "and none is given.";
228
229    private String alias = null;
230    private boolean cont = true;
231
232    public DeleteCommand(String alias) {
233      this.alias = alias;
234    }
235
236    @Override
237    public boolean validate() {
238      if (alias == null) {
239        getOut().println("There is no alias specified. Please provide the" +
240            "mandatory <alias>. See the usage description with -help.");
241        return false;
242      }
243      if (alias.equals("-help")) {
244        return true;
245      }
246      provider = getCredentialProvider();
247      if (provider == null) {
248        return false;
249      }
250      if (interactive) {
251        try {
252          cont = ToolRunner
253              .confirmPrompt("You are about to DELETE the credential " +
254                  alias + " from CredentialProvider " + provider.toString() +
255                  ". Continue? ");
256          if (!cont) {
257            getOut().println("Nothing has been deleted.");
258          }
259          return cont;
260        } catch (IOException e) {
261          getOut().println(alias + " will not be deleted.");
262          e.printStackTrace(getErr());
263        }
264      }
265      return true;
266    }
267
268    public void execute() throws IOException {
269      if (alias.equals("-help")) {
270        doHelp();
271        return;
272      }
273      warnIfTransientProvider();
274      getOut().println("Deleting credential: " + alias +
275          " from CredentialProvider: " + provider.toString());
276      if (cont) {
277        try {
278          provider.deleteCredentialEntry(alias);
279          getOut().println("Credential " + alias +
280              " has been successfully deleted.");
281          provider.flush();
282          printProviderWritten();
283        } catch (IOException e) {
284          getOut().println("Credential " + alias + " has NOT been deleted.");
285          throw e;
286        }
287      }
288    }
289
290    @Override
291    public String getUsage() {
292      return USAGE + ":\n\n" + DESC;
293    }
294  }

296  private class CreateCommand extends Command {
297    public static final String USAGE = "create <alias> [-value alias-value] " +
298        "[-provider provider-path] [-strict]";
299    public static final String DESC =
300        "The create subcommand creates a new credential for the name\n" +
301        "specified as the <alias> argument within the provider indicated\n" +
302        "through the -provider argument. If -strict is supplied, fail\n" +
303        "immediately if the provider requires a password and none is given.\n" +
304        "If -value is provided, use that for the value of the credential\n" +
305        "instead of prompting the user.";
306
307    private String alias = null;
308
309    public CreateCommand(String alias) {
310      this.alias = alias;
311    }
312
313    public boolean validate() {
314      if (alias == null) {
315        getOut().println("There is no alias specified. Please provide the" +
316            "mandatory <alias>. See the usage description with -help.");
317        return false;
318      }
319      if (alias.equals("-help")) {
320        return true;
321      }
322      try {
323        provider = getCredentialProvider();
324        if (provider == null) {
325          return false;
326        } else if (provider.needsPassword()) {
327          if (strict) {
328            getOut().println(provider.noPasswordError());
329            return false;
330          } else {
331            getOut().println(provider.noPasswordWarning());
332          }
333        }
334      } catch (IOException e) {
335        e.printStackTrace(getErr());
336      }
337      return true;
338    }
339
340    public void execute() throws IOException, NoSuchAlgorithmException {
341      if (alias.equals("-help")) {
342        doHelp();
343        return;
344      }
345      warnIfTransientProvider();
346      try {
347        char[] credential = null;
348        if (value != null) {
349          // testing only
350          credential = value.toCharArray();
351        } else {
352          credential = promptForCredential();
353        }
354        provider.createCredentialEntry(alias, credential);
355        provider.flush();
356        getOut().println(alias + " has been successfully created.");
357        printProviderWritten();
358      } catch (InvalidParameterException e) {
359        getOut().println("Credential " + alias + " has NOT been created. " +
360            e.getMessage());
361        throw e;
362      } catch (IOException e) {
363        getOut().println("Credential " + alias + " has NOT been created. " +
364            e.getMessage());
365        throw e;
366      }
367    }
368
369    @Override
370    public String getUsage() {
371      return USAGE + ":\n\n" + DESC;
372    }
373  }
374
375  protected char[] promptForCredential() throws IOException {
376    PasswordReader c = getPasswordReader();
377    if (c == null) {
378      throw new IOException("No console available for prompting user.");
379    }
380
381    char[] cred = null;
382
383    boolean noMatch;
384    do {
385      char[] newPassword1 = c.readPassword("Enter alias password: ");
386      char[] newPassword2 = c.readPassword("Enter alias password again: ");
387      noMatch = !Arrays.equals(newPassword1, newPassword2);
388      if (noMatch) {
389        if (newPassword1 != null) {
390          Arrays.fill(newPassword1, ' ');
391        }
392        c.format("Passwords don't match. Try again.%n");
393      } else {
394        cred = newPassword1;
395      }
396      if (newPassword2 != null) {
397        Arrays.fill(newPassword2, ' ');
398      }
399    } while (noMatch);
400    return cred;
401  }
402
403  public PasswordReader getPasswordReader() {
404    if (passwordReader == null) {
405      passwordReader = new PasswordReader();
406    }
407    return passwordReader;
408  }
409
410  public void setPasswordReader(PasswordReader reader) {
411    passwordReader = reader;
412  }

414  /** To facilitate testing since Console is a final class. */
415  public static class PasswordReader {
416    public char[] readPassword(String prompt) {
417      Console console = System.console();
418      char[] pass = console.readPassword(prompt);
419      return pass;
420    }
421
422    public void format(String message) {
423      Console console = System.console();
424      console.format(message);
425    }
426  }
