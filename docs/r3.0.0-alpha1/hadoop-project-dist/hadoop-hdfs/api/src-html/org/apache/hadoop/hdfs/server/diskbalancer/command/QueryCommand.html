020package org.apache.hadoop.hdfs.server.diskbalancer.command;
021
022import com.google.common.base.Preconditions;
023import org.apache.commons.cli.CommandLine;
024import org.apache.commons.cli.HelpFormatter;
025import org.apache.hadoop.conf.Configuration;
026import org.apache.hadoop.hdfs.DFSConfigKeys;
027import org.apache.hadoop.hdfs.protocol.ClientDatanodeProtocol;
028import org.apache.hadoop.hdfs.server.datanode.DiskBalancerWorkStatus;
029import org.apache.hadoop.hdfs.server.diskbalancer.DiskBalancerException;
030import org.apache.hadoop.hdfs.tools.DiskBalancer;
031import org.apache.hadoop.net.NetUtils;
032
033/**
034 * Gets the current status of disk balancer command.
035 */
036public class QueryCommand extends Command {
037
038  /**
039   * Constructs QueryCommand.
040   *
041   * @param conf - Configuration.
042   */
043  public QueryCommand(Configuration conf) {
044    super(conf);
045    addValidCommandParameters(DiskBalancer.QUERY, "Queries the status of disk" +
046        " plan running on a given datanode.");
047    addValidCommandParameters(DiskBalancer.VERBOSE, "Prints verbose results.");
048  }
049
050  /**
051   * Executes the Client Calls.
052   *
053   * @param cmd - CommandLine
054   */
055  @Override
056  public void execute(CommandLine cmd) throws Exception {
057    LOG.info("Executing \"query plan\" command.");
058    Preconditions.checkState(cmd.hasOption(DiskBalancer.QUERY));
059    verifyCommandOptions(DiskBalancer.QUERY, cmd);
060    String nodeName = cmd.getOptionValue(DiskBalancer.QUERY);
061    Preconditions.checkNotNull(nodeName);
062    nodeName = nodeName.trim();
063    String nodeAddress = nodeName;
064
065    // if the string is not name:port format use the default port.
066    if (!nodeName.matches("[^\\:]+:[0-9]{2,5}")) {
067      int defaultIPC = NetUtils.createSocketAddr(
068          getConf().getTrimmed(DFSConfigKeys.DFS_DATANODE_IPC_ADDRESS_KEY,
069              DFSConfigKeys.DFS_DATANODE_IPC_ADDRESS_DEFAULT)).getPort();
070      nodeAddress = nodeName + ":" + defaultIPC;
071      LOG.debug("Using default data node port :  {}", nodeAddress);
072    }
073
074    ClientDatanodeProtocol dataNode = getDataNodeProxy(nodeAddress);
075    try {
076      DiskBalancerWorkStatus workStatus = dataNode.queryDiskBalancerPlan();
077      System.out.printf("Plan File: %s%nPlan ID: %s%nResult: %s%n",
078              workStatus.getPlanFile(),
079              workStatus.getPlanID(),
080              workStatus.getResult().toString());
081
082      if (cmd.hasOption(DiskBalancer.VERBOSE)) {
083        System.out.printf("%s", workStatus.currentStateString());
084      }
085    } catch (DiskBalancerException ex) {
086      LOG.error("Query plan failed. ex: {}", ex);
087      throw ex;
088    }
089  }
090
091  /**
092   * Gets extended help for this command.
093   */
094  @Override
095  public void printHelp() {
096    String header = "Query Plan queries a given data node about the " +
097        "current state of disk balancer execution.\n\n";
098
099    String footer = "\nQuery command retrievs the plan ID and the current " +
100        "running state. ";
101
102    HelpFormatter helpFormatter = new HelpFormatter();
103    helpFormatter.printHelp("hdfs diskbalancer -query <hostname>  [options]",
104        header, DiskBalancer.getQueryOptions(), footer);
105  }
106}
