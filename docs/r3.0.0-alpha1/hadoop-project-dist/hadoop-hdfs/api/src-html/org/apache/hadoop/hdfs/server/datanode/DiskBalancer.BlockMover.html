059/**
060 * Worker class for Disk Balancer.
061 * <p>
062 * Here is the high level logic executed by this class. Users can submit disk
063 * balancing plans using submitPlan calls. After a set of sanity checks the plan
064 * is admitted and put into workMap.
065 * <p>
066 * The executePlan launches a thread that picks up work from workMap and hands
067 * it over to the BlockMover#copyBlocks function.
068 * <p>
069 * Constraints :
070 * <p>
071 * Only one plan can be executing in a datanode at any given time. This is
072 * ensured by checking the future handle of the worker thread in submitPlan.
073 */
074@InterfaceAudience.Private
075public class DiskBalancer {
076
077  private static final Logger LOG = LoggerFactory.getLogger(DiskBalancer
078      .class);
079  private final FsDatasetSpi<?> dataset;
080  private final String dataNodeUUID;
081  private final BlockMover blockMover;
082  private final ReentrantLock lock;
083  private final ConcurrentHashMap<VolumePair, DiskBalancerWorkItem> workMap;
084  private boolean isDiskBalancerEnabled = false;
085  private ExecutorService scheduler;
086  private Future future;
087  private String planID;
088  private String planFile;
089  private DiskBalancerWorkStatus.Result currentResult;
090  private long bandwidth;

092  /**
093   * Constructs a Disk Balancer object. This object takes care of reading a
094   * NodePlan and executing it against a set of volumes.
095   *
096   * @param dataNodeUUID - Data node UUID
097   * @param conf         - Hdfs Config
098   * @param blockMover   - Object that supports moving blocks.
099   */
100  public DiskBalancer(String dataNodeUUID,
101                      Configuration conf, BlockMover blockMover) {
102    this.currentResult = Result.NO_PLAN;
103    this.blockMover = blockMover;
104    this.dataset = this.blockMover.getDataset();
105    this.dataNodeUUID = dataNodeUUID;
106    scheduler = Executors.newSingleThreadExecutor();
107    lock = new ReentrantLock();
108    workMap = new ConcurrentHashMap<>();
109    this.planID = "";  // to keep protobuf happy.
110    this.planFile = "";  // to keep protobuf happy.
111    this.isDiskBalancerEnabled = conf.getBoolean(
112        DFSConfigKeys.DFS_DISK_BALANCER_ENABLED,
113        DFSConfigKeys.DFS_DISK_BALANCER_ENABLED_DEFAULT);
114    this.bandwidth = conf.getInt(
115        DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_THROUGHPUT,
116        DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_THROUGHPUT_DEFAULT);
117  }

119  /**
120   * Shutdown  disk balancer services.
121   */
122  public void shutdown() {
123    lock.lock();
124    try {
125      this.isDiskBalancerEnabled = false;
126      this.currentResult = Result.NO_PLAN;
127      if ((this.future != null) && (!this.future.isDone())) {
128        this.currentResult = Result.PLAN_CANCELLED;
129        this.blockMover.setExitFlag();
130        shutdownExecutor();
131      }
132    } finally {
133      lock.unlock();
134    }
135  }
136
137  /**
138   * Shutdown the executor.
139   */
140  private void shutdownExecutor() {
141    final int secondsTowait = 10;
142    scheduler.shutdown();
143    try {
144      if (!scheduler.awaitTermination(secondsTowait, TimeUnit.SECONDS)) {
145        scheduler.shutdownNow();
146        if (!scheduler.awaitTermination(secondsTowait, TimeUnit.SECONDS)) {
147          LOG.error("Disk Balancer : Scheduler did not terminate.");
148        }
149      }
150    } catch (InterruptedException ex) {
151      scheduler.shutdownNow();
152      Thread.currentThread().interrupt();
153    }
154  }

156  /**
157   * Takes a client submitted plan and converts into a set of work items that
158   * can be executed by the blockMover.
159   *
160   * @param planId      - A SHA-1 of the plan string
161   * @param planVersion - version of the plan string - for future use.
162   * @param planFileName    - Plan file name
163   * @param planData    - Plan data in json format
164   * @param force       - Skip some validations and execute the plan file.
165   * @throws DiskBalancerException
166   */
167  public void submitPlan(String planId, long planVersion, String planFileName,
168                         String planData, boolean force)
169          throws DiskBalancerException {
170    lock.lock();
171    try {
172      checkDiskBalancerEnabled();
173      if ((this.future != null) && (!this.future.isDone())) {
174        LOG.error("Disk Balancer - Executing another plan, submitPlan failed.");
175        throw new DiskBalancerException("Executing another plan",
176            DiskBalancerException.Result.PLAN_ALREADY_IN_PROGRESS);
177      }
178      NodePlan nodePlan = verifyPlan(planId, planVersion, planData, force);
179      createWorkPlan(nodePlan);
180      this.planID = planId;
181      this.planFile = planFileName;
182      this.currentResult = Result.PLAN_UNDER_PROGRESS;
183      executePlan();
184    } finally {
185      lock.unlock();
186    }
187  }

189  /**
190   * Returns the Current Work Status of a submitted Plan.
191   *
192   * @return DiskBalancerWorkStatus.
193   * @throws DiskBalancerException
194   */
195  public DiskBalancerWorkStatus queryWorkStatus() throws DiskBalancerException {
196    lock.lock();
197    try {
198      checkDiskBalancerEnabled();
199      // if we had a plan in progress, check if it is finished.
200      if (this.currentResult == Result.PLAN_UNDER_PROGRESS &&
201          this.future != null &&
202          this.future.isDone()) {
203        this.currentResult = Result.PLAN_DONE;
204      }
205
206      DiskBalancerWorkStatus status =
207          new DiskBalancerWorkStatus(this.currentResult, this.planID,
208                  this.planFile);
209      for (Map.Entry<VolumePair, DiskBalancerWorkItem> entry :
210          workMap.entrySet()) {
211        DiskBalancerWorkEntry workEntry = new DiskBalancerWorkEntry(
212            entry.getKey().getSource().getBasePath(),
213            entry.getKey().getDest().getBasePath(),
214            entry.getValue());
215        status.addWorkEntry(workEntry);
216      }
217      return status;
218    } finally {
219      lock.unlock();
220    }
221  }

223  /**
224   * Cancels a running plan.
225   *
226   * @param planID - Hash of the plan to cancel.
227   * @throws DiskBalancerException
228   */
229  public void cancelPlan(String planID) throws DiskBalancerException {
230    lock.lock();
231    try {
232      checkDiskBalancerEnabled();
233      if (this.planID == null ||
234          !this.planID.equals(planID) ||
235          this.planID.isEmpty()) {
236        LOG.error("Disk Balancer - No such plan. Cancel plan failed. PlanID: " +
237            planID);
238        throw new DiskBalancerException("No such plan.",
239            DiskBalancerException.Result.NO_SUCH_PLAN);
240      }
241      if (!this.future.isDone()) {
242        this.blockMover.setExitFlag();
243        shutdownExecutor();
244        this.currentResult = Result.PLAN_CANCELLED;
245      }
246    } finally {
247      lock.unlock();
248    }
249  }

272          DiskBalancerException.Result.INTERNAL_ERROR);
273    } finally {
274      lock.unlock();
275    }

307  /**
308   * Verifies that user provided plan is valid.
309   *
310   * @param planID      - SHA-1 of the plan.
311   * @param planVersion - Version of the plan, for future use.
312   * @param plan        - Plan String in Json.
313   * @param force       - Skip verifying when the plan was generated.
314   * @return a NodePlan Object.
315   * @throws DiskBalancerException
316   */
317  private NodePlan verifyPlan(String planID, long planVersion, String plan,
318                              boolean force) throws DiskBalancerException {
319
320    Preconditions.checkState(lock.isHeldByCurrentThread());
321    verifyPlanVersion(planVersion);
322    NodePlan nodePlan = verifyPlanHash(planID, plan);
323    if (!force) {
324      verifyTimeStamp(nodePlan);
325    }
326    verifyNodeUUID(nodePlan);
327    return nodePlan;
328  }
329
330  /**
331   * Verifies the plan version is something that we support.
332   *
333   * @param planVersion - Long version.
334   * @throws DiskBalancerException
335   */
336  private void verifyPlanVersion(long planVersion)
337      throws DiskBalancerException {
338    if ((planVersion < DiskBalancerConstants.DISKBALANCER_MIN_VERSION) ||
339        (planVersion > DiskBalancerConstants.DISKBALANCER_MAX_VERSION)) {
340      LOG.error("Disk Balancer - Invalid plan version.");
341      throw new DiskBalancerException("Invalid plan version.",
342          DiskBalancerException.Result.INVALID_PLAN_VERSION);
343    }
344  }
345
346  /**
347   * Verifies that plan matches the SHA-1 provided by the client.
348   *
349   * @param planID - SHA-1 Hex Bytes
350   * @param plan   - Plan String
351   * @throws DiskBalancerException
352   */
353  private NodePlan verifyPlanHash(String planID, String plan)
354      throws DiskBalancerException {
355    final long sha1Length = 40;
356    if (plan == null || plan.length() == 0) {
357      LOG.error("Disk Balancer -  Invalid plan.");
358      throw new DiskBalancerException("Invalid plan.",
359          DiskBalancerException.Result.INVALID_PLAN);
360    }
361
362    if ((planID == null) ||
363        (planID.length() != sha1Length) ||
364        !DigestUtils.shaHex(plan.getBytes(Charset.forName("UTF-8")))
365            .equalsIgnoreCase(planID)) {
366      LOG.error("Disk Balancer - Invalid plan hash.");
367      throw new DiskBalancerException("Invalid or mis-matched hash.",
368          DiskBalancerException.Result.INVALID_PLAN_HASH);
369    }
370
371    try {
372      return NodePlan.parseJson(plan);
373    } catch (IOException ex) {
374      throw new DiskBalancerException("Parsing plan failed.", ex,
375          DiskBalancerException.Result.MALFORMED_PLAN);
376    }
377  }

379  /**
380   * Verifies that this plan is not older than 24 hours.
381   *
382   * @param plan - Node Plan
383   */
384  private void verifyTimeStamp(NodePlan plan) throws DiskBalancerException {
385    long now = Time.now();
386    long planTime = plan.getTimeStamp();
387
388    // TODO : Support Valid Plan hours as a user configurable option.
389    if ((planTime +
390        (TimeUnit.HOURS.toMillis(
391            DiskBalancerConstants.DISKBALANCER_VALID_PLAN_HOURS))) < now) {
392      String hourString = "Plan was generated more than " +
393          Integer.toString(DiskBalancerConstants.DISKBALANCER_VALID_PLAN_HOURS)
394          + " hours ago.";
395      LOG.error("Disk Balancer - " + hourString);
396      throw new DiskBalancerException(hourString,
397          DiskBalancerException.Result.OLD_PLAN_SUBMITTED);
398    }
399  }

402   * Verify Node UUID.
403   *

416  /**
417   * Convert a node plan to DiskBalancerWorkItem that Datanode can execute.
418   *
419   * @param plan - Node Plan
420   */
421  private void createWorkPlan(NodePlan plan) throws DiskBalancerException {
422    Preconditions.checkState(lock.isHeldByCurrentThread());
423
424    // Cleanup any residual work in the map.
425    workMap.clear();
426    Map<String, FsVolumeSpi> pathMap = getStorageIDToVolumeMap();
427
428    for (Step step : plan.getVolumeSetPlans()) {
429      String sourceuuid = step.getSourceVolume().getUuid();
430      String destinationuuid = step.getDestinationVolume().getUuid();
431
432      FsVolumeSpi sourceVol = pathMap.get(sourceuuid);
433      if (sourceVol == null) {
434        LOG.error("Disk Balancer - Unable to find source volume. submitPlan " +
435            "failed.");
436        throw new DiskBalancerException("Unable to find source volume.",
437            DiskBalancerException.Result.INVALID_VOLUME);
438      }
439
440      FsVolumeSpi destVol = pathMap.get(destinationuuid);
441      if (destVol == null) {
442        LOG.error("Disk Balancer - Unable to find destination volume. " +
443            "submitPlan failed.");
444        throw new DiskBalancerException("Unable to find destination volume.",
445            DiskBalancerException.Result.INVALID_VOLUME);
446      }
447      createWorkPlan(sourceVol, destVol, step);
448    }
449  }

451  /**
452   * Returns a path to Volume Map.
453   *
454   * @return Map
455   * @throws DiskBalancerException
456   */
457  private Map<String, FsVolumeSpi> getStorageIDToVolumeMap()
458      throws DiskBalancerException {
459    Map<String, FsVolumeSpi> pathMap = new HashMap<>();
460    FsDatasetSpi.FsVolumeReferences references;
461    try {
462      try(AutoCloseableLock lock = this.dataset.acquireDatasetLock()) {
463        references = this.dataset.getFsVolumeReferences();
464        for (int ndx = 0; ndx < references.size(); ndx++) {
465          FsVolumeSpi vol = references.get(ndx);
466          pathMap.put(vol.getStorageID(), vol);
467        }
468        references.close();
469      }
470    } catch (IOException ex) {
471      LOG.error("Disk Balancer - Internal Error.", ex);
472      throw new DiskBalancerException("Internal error", ex,
473          DiskBalancerException.Result.INTERNAL_ERROR);
474    }
475    return pathMap;
476  }

478  /**
479   * Starts Executing the plan, exits when the plan is done executing.
480   */
481  private void executePlan() {
482    Preconditions.checkState(lock.isHeldByCurrentThread());
483    this.blockMover.setRunnable();
484    if (this.scheduler.isShutdown()) {
485      this.scheduler = Executors.newSingleThreadExecutor();
486    }
487
488    this.future = scheduler.submit(new Runnable() {
489      @Override
490      public void run() {
491        Thread.currentThread().setName("DiskBalancerThread");
492        LOG.info("Executing Disk balancer plan. Plan File: {}, Plan ID: {}",
493                planFile, planID);
494        try {
495          for (Map.Entry<VolumePair, DiskBalancerWorkItem> entry :
496              workMap.entrySet()) {
497            blockMover.copyBlocks(entry.getKey(), entry.getValue());
498          }
499        } finally {
500          blockMover.setExitFlag();
501        }
502      }
503    });
504  }

506  /**
507   * Insert work items to work map.
508   *
509   * @param source - Source vol
510   * @param dest   - destination volume
511   * @param step   - Move Step
512   */
513  private void createWorkPlan(FsVolumeSpi source, FsVolumeSpi dest,
514                              Step step) throws DiskBalancerException {
515
516    if (source.getStorageID().equals(dest.getStorageID())) {
517      LOG.info("Disk Balancer - source & destination volumes are same.");
518      throw new DiskBalancerException("source and destination volumes are " +
519          "same.", DiskBalancerException.Result.INVALID_MOVE);
520    }
521    VolumePair pair = new VolumePair(source, dest);
522    long bytesToMove = step.getBytesToMove();
523    // In case we have a plan with more than
524    // one line of same <source, dest>
525    // we compress that into one work order.
526    if (workMap.containsKey(pair)) {
527      bytesToMove += workMap.get(pair).getBytesToCopy();
528    }
529
530    DiskBalancerWorkItem work = new DiskBalancerWorkItem(bytesToMove, 0);
531
532    // all these values can be zero, if so we will use
533    // values from configuration.
534    work.setBandwidth(step.getBandwidth());
535    work.setTolerancePercent(step.getTolerancePercent());
536    work.setMaxDiskErrors(step.getMaxDiskErrors());
537    workMap.put(pair, work);
538  }

585  /**
586   * Holds references to actual volumes that we will be operating against.
587   */
588  public static class VolumePair {
589    private final FsVolumeSpi source;
590    private final FsVolumeSpi dest;
591
592    /**
593     * Constructs a volume pair.
594     *
595     * @param source - Source Volume
596     * @param dest   - Destination Volume
597     */
598    public VolumePair(FsVolumeSpi source, FsVolumeSpi dest) {
599      this.source = source;
600      this.dest = dest;
601    }
602
603    /**
604     * gets source volume.
605     *
606     * @return volume
607     */
608    public FsVolumeSpi getSource() {
609      return source;
610    }
611
612    /**
613     * Gets Destination volume.
614     *
615     * @return volume.
616     */
617    public FsVolumeSpi getDest() {
618      return dest;
619    }
620
621    @Override
622    public boolean equals(Object o) {
623      if (this == o) {
624        return true;
625      }
626      if (o == null || getClass() != o.getClass()) {
627        return false;
628      }
629
630      VolumePair that = (VolumePair) o;
631      return source.equals(that.source) && dest.equals(that.dest);
632    }
633
634    @Override
635    public int hashCode() {
636      int result = source.getBasePath().hashCode();
637      result = 31 * result + dest.getBasePath().hashCode();
638      return result;
639    }
640  }

642  /**
643   * Actual DataMover class for DiskBalancer.
644   * <p>
645   */
646  public static class DiskBalancerMover implements BlockMover {
647    private final FsDatasetSpi dataset;
648    private long diskBandwidth;
649    private long blockTolerance;
650    private long maxDiskErrors;
651    private int poolIndex;
652    private AtomicBoolean shouldRun;
653    private long startTime;
654    private long secondsElapsed;
655
656    /**
657     * Constructs diskBalancerMover.
658     *
659     * @param dataset Dataset
660     * @param conf    Configuration
661     */
662    public DiskBalancerMover(FsDatasetSpi dataset, Configuration conf) {
663      this.dataset = dataset;
664      shouldRun = new AtomicBoolean(false);
665
666      this.diskBandwidth = conf.getLong(
667          DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_THROUGHPUT,
668          DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_THROUGHPUT_DEFAULT);
669
670      this.blockTolerance = conf.getLong(
671          DFSConfigKeys.DFS_DISK_BALANCER_BLOCK_TOLERANCE,
672          DFSConfigKeys.DFS_DISK_BALANCER_BLOCK_TOLERANCE_DEFAULT);
673
674      this.maxDiskErrors = conf.getLong(
675          DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_ERRORS,
676          DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_ERRORS_DEFAULT);
677
678      // Since these are user provided values make sure it is sane
679      // or ignore faulty values.
680      if (this.diskBandwidth <= 0) {
681        LOG.debug("Found 0 or less as max disk throughput, ignoring config " +
682            "value. value : " + diskBandwidth);
683        diskBandwidth =
684            DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_THROUGHPUT_DEFAULT;
685      }
686
687      if (this.blockTolerance <= 0) {
688        LOG.debug("Found 0 or less for block tolerance value, ignoring config" +
689            "value. value : " + blockTolerance);
690        blockTolerance =
691            DFSConfigKeys.DFS_DISK_BALANCER_BLOCK_TOLERANCE_DEFAULT;
692
693      }
694
695      if (this.maxDiskErrors < 0) {
696        LOG.debug("Found  less than 0 for maxDiskErrors value, ignoring " +
697            "config value. value : " + maxDiskErrors);
698        maxDiskErrors =
699            DFSConfigKeys.DFS_DISK_BALANCER_MAX_DISK_ERRORS_DEFAULT;
700      }
701    }
702
703    /**
704     * Sets Diskmover copyblocks into runnable state.
705     */
706    @Override
707    public void setRunnable() {
708      this.shouldRun.set(true);
709    }
710
711    /**
712     * Signals copy block to exit.
713     */
714    @Override
715    public void setExitFlag() {

726    /**
727     * Checks if a given block is less than needed size to meet our goal.
728     *
729     * @param blockSize - block len
730     * @param item      - Work item
731     * @return true if this block meets our criteria, false otherwise.
732     */
733    private boolean isLessThanNeeded(long blockSize,
734                                     DiskBalancerWorkItem item) {
735      long bytesToCopy = item.getBytesToCopy() - item.getBytesCopied();
736      bytesToCopy = bytesToCopy +
737          ((bytesToCopy * getBlockTolerancePercentage(item)) / 100);
738      return (blockSize <= bytesToCopy) ? true : false;
739    }

742     * Returns the default block tolerance if the plan does not have value of
743     * tolerance specified.
744     *
745     * @param item - DiskBalancerWorkItem
746     * @return long
747     */
748    private long getBlockTolerancePercentage(DiskBalancerWorkItem item) {
749      return item.getTolerancePercent() <= 0 ? this.blockTolerance :
750          item.getTolerancePercent();

753    /**
754     * Inflates bytesCopied and returns true or false. This allows us to stop
755     * copying if we have reached close enough.
756     *
757     * @param item DiskBalancerWorkItem
758     * @return -- false if we need to copy more, true if we are done
759     */
760    private boolean isCloseEnough(DiskBalancerWorkItem item) {
761      long temp = item.getBytesCopied() +
762          ((item.getBytesCopied() * getBlockTolerancePercentage(item)) / 100);
763      return (item.getBytesToCopy() >= temp) ? false : true;
764    }

768     * returns the global default.

770     * @param item DiskBalancerWorkItem.
771     * @return MB/s - long

775          .getBandwidth();

778    /**
779     * Computes sleep delay needed based on the block that just got copied. we
780     * copy using a burst mode, that is we let the copy proceed in full
781     * throttle. Once a copy is done, we compute how many bytes have been
782     * transferred and try to average it over the user specified bandwidth. In
783     * other words, This code implements a poor man's token bucket algorithm for
784     * traffic shaping.
785     *
786     * @param bytesCopied - byteCopied.
787     * @param timeUsed    in milliseconds
788     * @param item        DiskBalancerWorkItem
789     * @return sleep delay in Milliseconds.
790     */
791    private long computeDelay(long bytesCopied, long timeUsed,
792                              DiskBalancerWorkItem item) {
793
794      // we had an overflow, ignore this reading and continue.
795      if (timeUsed == 0) {
796        return 0;
797      }
798      final int megaByte = 1024 * 1024;
799      long bytesInMB = bytesCopied / megaByte;
800      long lastThroughput = bytesInMB / SECONDS.convert(timeUsed,
801          TimeUnit.MILLISECONDS);
802      long delay = (bytesInMB / getDiskBandwidth(item)) - lastThroughput;
803      return (delay <= 0) ? 0 : MILLISECONDS.convert(delay, TimeUnit.SECONDS);
804    }

806    /**
807     * Returns maximum errors to tolerate for the specific plan or the default.
808     *
809     * @param item - DiskBalancerWorkItem
810     * @return maximum error counts to tolerate.
811     */
812    private long getMaxError(DiskBalancerWorkItem item) {
813      return item.getMaxDiskErrors() <= 0 ? this.maxDiskErrors :
814          item.getMaxDiskErrors();
815    }

817    /**
818     * Gets the next block that we can copy, returns null if we cannot find a
819     * block that fits our parameters or if have run out of blocks.
820     *
821     * @param iter Block Iter
822     * @param item - Work item
823     * @return Extended block or null if no copyable block is found.
824     */
825    private ExtendedBlock getBlockToCopy(FsVolumeSpi.BlockIterator iter,
826                                         DiskBalancerWorkItem item) {
827      while (!iter.atEnd() && item.getErrorCount() < getMaxError(item)) {
828        try {
829          ExtendedBlock block = iter.nextBlock();
830
831          // A valid block is a finalized block, we iterate until we get
832          // finalized blocks
833          if (!this.dataset.isValidBlock(block)) {
834            continue;
835          }
836
837          // We don't look for the best, we just do first fit
838          if (isLessThanNeeded(block.getNumBytes(), item)) {
839            return block;
840          }
841
842        } catch (IOException e) {
843          item.incErrorCount();
844        }
845      }
846
847      if (item.getErrorCount() >= getMaxError(item)) {
848        item.setErrMsg("Error count exceeded.");
849        LOG.info("Maximum error count exceeded. Error count: {} Max error:{} "
850            , item.getErrorCount(), item.getMaxDiskErrors());
851      }
852
853      return null;
854    }

856    /**
857     * Opens all Block pools on a given volume.
858     *
859     * @param source    Source
860     * @param poolIters List of PoolIters to maintain.
861     */
862    private void openPoolIters(FsVolumeSpi source, List<FsVolumeSpi
863        .BlockIterator> poolIters) {
864      Preconditions.checkNotNull(source);
865      Preconditions.checkNotNull(poolIters);
866
867      for (String blockPoolID : source.getBlockPoolList()) {
868        poolIters.add(source.newBlockIterator(blockPoolID,
869            "DiskBalancerSource"));
870      }
871    }
872
873    /**
874     * Returns the next block that we copy from all the block pools. This
875     * function looks across all block pools to find the next block to copy.
876     *
877     * @param poolIters - List of BlockIterators
878     * @return ExtendedBlock.
879     */
880    ExtendedBlock getNextBlock(List<FsVolumeSpi.BlockIterator> poolIters,
881                               DiskBalancerWorkItem item) {
882      Preconditions.checkNotNull(poolIters);
883      int currentCount = 0;
884      ExtendedBlock block = null;
885      while (block == null && currentCount < poolIters.size()) {
886        currentCount++;
887        poolIndex = poolIndex++ % poolIters.size();
888        FsVolumeSpi.BlockIterator currentPoolIter = poolIters.get(poolIndex);
889        block = getBlockToCopy(currentPoolIter, item);
890      }
891
892      if (block == null) {
893        try {
894          item.setErrMsg("No source blocks found to move.");
895          LOG.error("No movable source blocks found. {}", item.toJson());
896        } catch (IOException e) {
897          LOG.error("Unable to get json from Item.");
898        }
899      }
900      return block;
901    }

903    /**
904     * Close all Pool Iters.
905     *
906     * @param poolIters List of BlockIters
907     */
908    private void closePoolIters(List<FsVolumeSpi.BlockIterator> poolIters) {
909      Preconditions.checkNotNull(poolIters);
910      for (FsVolumeSpi.BlockIterator iter : poolIters) {
911        try {
912          iter.close();
913        } catch (IOException ex) {
914          LOG.error("Error closing a block pool iter. ex: {}", ex);
915        }
916      }
917    }

919    /**
920     * Copies blocks from a set of volumes.
921     *
922     * @param pair - Source and Destination Volumes.
923     * @param item - Number of bytes to move from volumes.
924     */
925    @Override
926    public void copyBlocks(VolumePair pair, DiskBalancerWorkItem item) {
927      FsVolumeSpi source = pair.getSource();
928      FsVolumeSpi dest = pair.getDest();
929      List<FsVolumeSpi.BlockIterator> poolIters = new LinkedList<>();
930      startTime = Time.now();
931      item.setStartTime(startTime);
932      secondsElapsed = 0;
933
934      if (source.isTransientStorage() || dest.isTransientStorage()) {
935        return;
936      }
937
938      try {
939        openPoolIters(source, poolIters);
940        if (poolIters.size() == 0) {
941          LOG.error("No block pools found on volume. volume : {}. Exiting.",
942              source.getBasePath());
943          return;
944        }
945
946        while (shouldRun()) {
947          try {
948
949            // Check for the max error count constraint.
950            if (item.getErrorCount() > getMaxError(item)) {
951              LOG.error("Exceeded the max error count. source {}, dest: {} " +
952                      "error count: {}", source.getBasePath(),
953                  dest.getBasePath(), item.getErrorCount());
954              break;
955            }
956
957            // Check for the block tolerance constraint.
958            if (isCloseEnough(item)) {
959              LOG.info("Copy from {} to {} done. copied {} bytes and {} " +
960                      "blocks.",
961                  source.getBasePath(), dest.getBasePath(),
962                  item.getBytesCopied(), item.getBlocksCopied());
963              break;
964            }
965
966            ExtendedBlock block = getNextBlock(poolIters, item);
967            // we are not able to find any blocks to copy.
968            if (block == null) {
969              LOG.error("No source blocks, exiting the copy. Source: {}, " +
970                  "dest:{}", source.getBasePath(), dest.getBasePath());
971              break;
972            }
973
974            // check if someone told us exit, treat this as an interruption
975            // point
976            // for the thread, since both getNextBlock and moveBlocAcrossVolume
977            // can take some time.
978            if (!shouldRun()) {
979              break;
980            }
981
982            long timeUsed;
983            // There is a race condition here, but we will get an IOException
984            // if dest has no space, which we handle anyway.
985            if (dest.getAvailable() > item.getBytesToCopy()) {
986              long begin = System.nanoTime();
987              this.dataset.moveBlockAcrossVolumes(block, dest);
988              long now = System.nanoTime();
989              timeUsed = (now - begin) > 0 ? now - begin : 0;
990            } else {
991
992              // Technically it is possible for us to find a smaller block and
993              // make another copy, but opting for the safer choice of just
994              // exiting here.
995              LOG.error("Destination volume: {} does not have enough space to" +
996                  " accommodate a block. Block Size: {} Exiting from" +
997                  " copyBlocks.", dest.getBasePath(), block.getNumBytes());
998              break;
999            }
1000
1001            LOG.debug("Moved block with size {} from  {} to {}",
1002                block.getNumBytes(), source.getBasePath(),
1003                dest.getBasePath());
1004
1005            // Check for the max throughput constraint.
1006            // We sleep here to keep the promise that we will not
1007            // copy more than Max MB/sec. we sleep enough time
1008            // to make sure that our promise is good on average.
1009            // Because we sleep, if a shutdown or cancel call comes in
1010            // we exit via Thread Interrupted exception.
1011            Thread.sleep(computeDelay(block.getNumBytes(), timeUsed, item));
1012
1013            // We delay updating the info to avoid confusing the user.
1014            // This way we report the copy only if it is under the
1015            // throughput threshold.
1016            item.incCopiedSoFar(block.getNumBytes());
1017            item.incBlocksCopied();
1018            secondsElapsed = TimeUnit.MILLISECONDS.toSeconds(Time.now() -
1019                startTime);
1020            item.setSecondsElapsed(secondsElapsed);
1021          } catch (IOException ex) {
1022            LOG.error("Exception while trying to copy blocks. error: {}", ex);
1023            item.incErrorCount();
1024          } catch (InterruptedException e) {
1025            LOG.error("Copy Block Thread interrupted, exiting the copy.");
1026            Thread.currentThread().interrupt();
1027            item.incErrorCount();
1028            this.setExitFlag();
1029          }
1030        }
1031      } finally {
1032        // Close all Iters.
1033        closePoolIters(poolIters);
1034      }
1035    }
