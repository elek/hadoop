019package org.apache.hadoop.hdfs;
020
021import java.io.IOException;
022import java.net.URI;
023
024import org.apache.commons.logging.Log;
025import org.apache.commons.logging.LogFactory;
026
027import org.apache.hadoop.conf.Configuration;
028import org.apache.hadoop.fs.FileSystem;
029import org.apache.hadoop.hdfs.protocol.HdfsConstants;
030import org.apache.hadoop.io.Text;
031import org.apache.hadoop.security.Credentials;
032import org.apache.hadoop.security.UserGroupInformation;
033import org.apache.hadoop.security.token.DtFetcher;
034import org.apache.hadoop.security.token.Token;
035
036
037/**
038 *  DtFetcher is an interface which permits the abstraction and separation of
039 *  delegation token fetch implementaions across different packages and
040 *  compilation units.  Resolution of fetcher impl will be done at runtime.
041 */
042public class HdfsDtFetcher implements DtFetcher {
043  private static final Log LOG = LogFactory.getLog(HdfsDtFetcher.class);
044
045  private static final String SERVICE_NAME = HdfsConstants.HDFS_URI_SCHEME;
046
047  private static final String FETCH_FAILED = "Fetch of delegation token failed";
048
049  /**
050   * Returns the service name for HDFS, which is also a valid URL prefix.
051   */
052  public Text getServiceName() {
053    return new Text(SERVICE_NAME);
054  }
055
056  public boolean isTokenRequired() {
057    return UserGroupInformation.isSecurityEnabled();
058  }
059
060  /**
061   *  Returns Token object via FileSystem, null if bad argument.
062   *  @param conf - a Configuration object used with FileSystem.get()
063   *  @param creds - a Credentials object to which token(s) will be added
064   *  @param renewer  - the renewer to send with the token request
065   *  @param url  - the URL to which the request is sent
066   *  @return a Token, or null if fetch fails.
067   */
068  public Token<?> addDelegationTokens(Configuration conf, Credentials creds,
069                                  String renewer, String url) throws Exception {
070    if (!url.startsWith(getServiceName().toString())) {
071      url = getServiceName().toString() + "://" + url;
072    }
073    FileSystem fs = FileSystem.get(URI.create(url), conf);
074    Token<?> token = fs.getDelegationToken(renewer);
075    if (token == null) {
076      LOG.error(FETCH_FAILED);
077      throw new IOException(FETCH_FAILED);
078    }
079    creds.addToken(token.getService(), token);
080    return token;
081  }
082}
