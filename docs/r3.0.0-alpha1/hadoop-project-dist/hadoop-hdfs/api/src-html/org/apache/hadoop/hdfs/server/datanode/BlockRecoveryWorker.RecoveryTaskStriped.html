102  /** A block recovery task for a contiguous block. */
103  class RecoveryTaskContiguous {
104    private final RecoveringBlock rBlock;
105    private final ExtendedBlock block;
106    private final String bpid;
107    private final DatanodeInfo[] locs;
108    private final long recoveryId;
109
110    RecoveryTaskContiguous(RecoveringBlock rBlock) {
111      this.rBlock = rBlock;
112      block = rBlock.getBlock();
113      bpid = block.getBlockPoolId();
114      locs = rBlock.getLocations();
115      recoveryId = rBlock.getNewGenerationStamp();
116    }
117
118    protected void recover() throws IOException {
119      List<BlockRecord> syncList = new ArrayList<>(locs.length);
120      int errorCount = 0;
121      int candidateReplicaCnt = 0;
122
123      // Check generation stamps, replica size and state. Replica must satisfy
124      // the following criteria to be included in syncList for recovery:
125      // - Valid generation stamp
126      // - Non-zero length
127      // - Original state is RWR or better
128      for(DatanodeID id : locs) {
129        try {
130          DatanodeID bpReg = new DatanodeID(
131              datanode.getBPOfferService(bpid).bpRegistration);
132          InterDatanodeProtocol proxyDN = bpReg.equals(id)?
133              datanode: DataNode.createInterDataNodeProtocolProxy(id, conf,
134              dnConf.socketTimeout, dnConf.connectToDnViaHostname);
135          ReplicaRecoveryInfo info = callInitReplicaRecovery(proxyDN, rBlock);
136          if (info != null &&
137              info.getGenerationStamp() >= block.getGenerationStamp() &&
138              info.getNumBytes() > 0) {
139            // Count the number of candidate replicas received.
140            ++candidateReplicaCnt;
141            if (info.getOriginalReplicaState().getValue() <=
142                ReplicaState.RWR.getValue()) {
143              syncList.add(new BlockRecord(id, proxyDN, info));
144            } else {
145              if (LOG.isDebugEnabled()) {
146                LOG.debug("Block recovery: Ignored replica with invalid " +
147                    "original state: " + info + " from DataNode: " + id);
148              }
149            }
150          } else {
151            if (LOG.isDebugEnabled()) {
152              if (info == null) {
153                LOG.debug("Block recovery: DataNode: " + id + " does not have "
154                    + "replica for block: " + block);
155              } else {
156                LOG.debug("Block recovery: Ignored replica with invalid "
157                    + "generation stamp or length: " + info + " from " +
158                    "DataNode: " + id);
159              }
160            }
161          }
162        } catch (RecoveryInProgressException ripE) {
163          InterDatanodeProtocol.LOG.warn(
164              "Recovery for replica " + block + " on data-node " + id
165                  + " is already in progress. Recovery id = "
166                  + rBlock.getNewGenerationStamp() + " is aborted.", ripE);
167          return;
168        } catch (IOException e) {
169          ++errorCount;
170          InterDatanodeProtocol.LOG.warn(
171              "Failed to obtain replica info for block (=" + block
172                  + ") from datanode (=" + id + ")", e);
173        }
174      }
175
176      if (errorCount == locs.length) {
177        throw new IOException("All datanodes failed: block=" + block
178            + ", datanodeids=" + Arrays.asList(locs));
179      }
180
181      // None of the replicas reported by DataNodes has the required original
182      // state, report the error.
183      if (candidateReplicaCnt > 0 && syncList.isEmpty()) {
184        throw new IOException("Found " + candidateReplicaCnt +
185            " replica(s) for block " + block + " but none is in " +
186            ReplicaState.RWR.name() + " or better state. datanodeids=" +
187            Arrays.asList(locs));
188      }
189
190      syncBlock(syncList);
191    }
192
193    /** Block synchronization. */
194    void syncBlock(List<BlockRecord> syncList) throws IOException {
195      DatanodeProtocolClientSideTranslatorPB nn =
196          getActiveNamenodeForBP(block.getBlockPoolId());
197
198      boolean isTruncateRecovery = rBlock.getNewBlock() != null;
199      long blockId = (isTruncateRecovery) ?
200          rBlock.getNewBlock().getBlockId() : block.getBlockId();
201
202      if (LOG.isDebugEnabled()) {
203        LOG.debug("block=" + block + ", (length=" + block.getNumBytes()
204            + "), syncList=" + syncList);
205      }
206
207      // syncList.isEmpty() means that all data-nodes do not have the block
208      // or their replicas have 0 length.
209      // The block can be deleted.
210      if (syncList.isEmpty()) {
211        if (LOG.isDebugEnabled()) {
212          LOG.debug("syncBlock for block " + block + ", all datanodes don't " +
213              "have the block or their replicas have 0 length. The block can " +
214              "be deleted.");
215        }
216        nn.commitBlockSynchronization(block, recoveryId, 0,
217            true, true, DatanodeID.EMPTY_ARRAY, null);
218        return;
219      }
220
221      // Calculate the best available replica state.
222      ReplicaState bestState = ReplicaState.RWR;
223      long finalizedLength = -1;
224      for (BlockRecord r : syncList) {
225        assert r.rInfo.getNumBytes() > 0 : "zero length replica";
226        ReplicaState rState = r.rInfo.getOriginalReplicaState();
227        if (rState.getValue() < bestState.getValue()) {
228          bestState = rState;
229        }
230        if(rState == ReplicaState.FINALIZED) {
231          if (finalizedLength > 0 && finalizedLength != r.rInfo.getNumBytes()) {
232            throw new IOException("Inconsistent size of finalized replicas. " +
233                "Replica " + r.rInfo + " expected size: " + finalizedLength);
234          }
235          finalizedLength = r.rInfo.getNumBytes();
236        }
237      }
238
239      // Calculate list of nodes that will participate in the recovery
240      // and the new block size
241      List<BlockRecord> participatingList = new ArrayList<>();
242      final ExtendedBlock newBlock = new ExtendedBlock(bpid, blockId,
243          -1, recoveryId);
244      switch(bestState) {
245      case FINALIZED:
246        assert finalizedLength > 0 : "finalizedLength is not positive";
247        for(BlockRecord r : syncList) {
248          ReplicaState rState = r.rInfo.getOriginalReplicaState();
249          if (rState == ReplicaState.FINALIZED ||
250              rState == ReplicaState.RBW &&
251                  r.rInfo.getNumBytes() == finalizedLength) {
252            participatingList.add(r);
253          }
254          if (LOG.isDebugEnabled()) {
255            LOG.debug("syncBlock replicaInfo: block=" + block +
256                ", from datanode " + r.id + ", receivedState=" + rState.name() +
257                ", receivedLength=" + r.rInfo.getNumBytes() +
258                ", bestState=FINALIZED, finalizedLength=" + finalizedLength);
259          }
260        }
261        newBlock.setNumBytes(finalizedLength);
262        break;
263      case RBW:
264      case RWR:
265        long minLength = Long.MAX_VALUE;
266        for(BlockRecord r : syncList) {
267          ReplicaState rState = r.rInfo.getOriginalReplicaState();
268          if(rState == bestState) {
269            minLength = Math.min(minLength, r.rInfo.getNumBytes());
270            participatingList.add(r);
271          }
272          if (LOG.isDebugEnabled()) {
273            LOG.debug("syncBlock replicaInfo: block=" + block +
274                ", from datanode " + r.id + ", receivedState=" + rState.name() +
275                ", receivedLength=" + r.rInfo.getNumBytes() + ", bestState=" +
276                bestState.name());
277          }
278        }
279        // recover() guarantees syncList will have at least one replica with RWR
280        // or better state.
281        assert minLength != Long.MAX_VALUE : "wrong minLength";
282        newBlock.setNumBytes(minLength);
283        break;
284      case RUR:
285      case TEMPORARY:
286        assert false : "bad replica state: " + bestState;
287      default:
288        break; // we have 'case' all enum values
289      }
290      if (isTruncateRecovery) {
291        newBlock.setNumBytes(rBlock.getNewBlock().getNumBytes());
292      }
293
294      List<DatanodeID> failedList = new ArrayList<>();
295      final List<BlockRecord> successList = new ArrayList<>();
296      for (BlockRecord r : participatingList) {
297        try {
298          r.updateReplicaUnderRecovery(bpid, recoveryId, blockId,
299              newBlock.getNumBytes());
300          successList.add(r);
301        } catch (IOException e) {
302          InterDatanodeProtocol.LOG.warn("Failed to updateBlock (newblock="
303              + newBlock + ", datanode=" + r.id + ")", e);
304          failedList.add(r.id);
305        }
306      }
307
308      // If any of the data-nodes failed, the recovery fails, because
309      // we never know the actual state of the replica on failed data-nodes.
310      // The recovery should be started over.
311      if (!failedList.isEmpty()) {
312        throw new IOException("Cannot recover " + block
313            + ", the following datanodes failed: " + failedList);
314      }
315
316      // Notify the name-node about successfully recovered replicas.
317      final DatanodeID[] datanodes = new DatanodeID[successList.size()];
318      final String[] storages = new String[datanodes.length];
319      for (int i = 0; i < datanodes.length; i++) {
320        final BlockRecord r = successList.get(i);
321        datanodes[i] = r.id;
322        storages[i] = r.storageID;
323      }
324
325      if (LOG.isDebugEnabled()) {
326        LOG.debug("Datanode triggering commitBlockSynchronization, block=" +
327            block + ", newGs=" + newBlock.getGenerationStamp() +
328            ", newLength=" + newBlock.getNumBytes());
329      }
330
331      nn.commitBlockSynchronization(block,
332          newBlock.getGenerationStamp(), newBlock.getNumBytes(), true, false,
333          datanodes, storages);
334    }
335  }

337  /**
338   * blk_0  blk_1  blk_2  blk_3  blk_4  blk_5  blk_6  blk_7  blk_8
339   *  64k    64k    64k    64k    64k    64k    64k    64k    64k   <-- stripe_0
340   *  64k    64k    64k    64k    64k    64k    64k    64k    64k
341   *  64k    64k    64k    64k    64k    64k    64k    61k    <-- startStripeIdx
342   *  64k    64k    64k    64k    64k    64k    64k
343   *  64k    64k    64k    64k    64k    64k    59k
344   *  64k    64k    64k    64k    64k    64k
345   *  64k    64k    64k    64k    64k    64k                <-- last full stripe
346   *  64k    64k    13k    64k    55k     3k              <-- target last stripe
347   *  64k    64k           64k     1k
348   *  64k    64k           58k
349   *  64k    64k
350   *  64k    19k
351   *  64k                                               <-- total visible stripe
352   *
353   *  Due to different speed of streamers, the internal blocks in a block group
354   *  could have different lengths when the block group isn't ended normally.
355   *  The purpose of this class is to recover the UnderConstruction block group,
356   *  so all internal blocks end at the same stripe.
357   *
358   * The steps:
359   * 1. get all blocks lengths from DataNodes.
360   * 2. calculate safe length, which is at the target last stripe.
361   * 3. decode and feed blk_6~8, make them end at last full stripe. (the last
362   * full stripe means the last decodable stripe.)
363   * 4. encode the target last stripe, with the remaining sequential data. In
364   * this case, the sequential data is 64k+64k+13k. Feed blk_6~8 the parity cells.
365   * Overwrite the parity cell if have to.
366   * 5. truncate the stripes from visible stripe, to target last stripe.
367   * TODO: implement step 3,4
368   */
369  public class RecoveryTaskStriped {
370    private final RecoveringBlock rBlock;
371    private final ExtendedBlock block;
372    private final String bpid;
373    private final DatanodeInfo[] locs;
374    private final long recoveryId;
375
376    private final byte[] blockIndices;
377    private final ErasureCodingPolicy ecPolicy;
378
379    RecoveryTaskStriped(RecoveringStripedBlock rBlock) {
380      this.rBlock = rBlock;
381      // TODO: support truncate
382      Preconditions.checkArgument(rBlock.getNewBlock() == null);
383
384      block = rBlock.getBlock();
385      bpid = block.getBlockPoolId();
386      locs = rBlock.getLocations();
387      recoveryId = rBlock.getNewGenerationStamp();
388      blockIndices = rBlock.getBlockIndices();
389      ecPolicy = rBlock.getErasureCodingPolicy();
390    }
391
392    protected void recover() throws IOException {
393      checkLocations(locs.length);
394
395      Map<Long, BlockRecord> syncBlocks = new HashMap<>(locs.length);
396      final int dataBlkNum = ecPolicy.getNumDataUnits();
397      final int totalBlkNum = dataBlkNum + ecPolicy.getNumParityUnits();
398      //check generation stamps
399      for (int i = 0; i < locs.length; i++) {
400        DatanodeID id = locs[i];
401        try {
402          DatanodeID bpReg = new DatanodeID(
403              datanode.getBPOfferService(bpid).bpRegistration);
404          InterDatanodeProtocol proxyDN = bpReg.equals(id) ?
405              datanode : DataNode.createInterDataNodeProtocolProxy(id, conf,
406              dnConf.socketTimeout, dnConf.connectToDnViaHostname);
407          ExtendedBlock internalBlk = new ExtendedBlock(block);
408          final long blockId = block.getBlockId() + blockIndices[i];
409          internalBlk.setBlockId(blockId);
410          ReplicaRecoveryInfo info = callInitReplicaRecovery(proxyDN,
411              new RecoveringBlock(internalBlk, null, recoveryId));
412
413          if (info != null &&
414              info.getGenerationStamp() >= block.getGenerationStamp() &&
415              info.getNumBytes() > 0) {
416            final BlockRecord existing = syncBlocks.get(blockId);
417            if (existing == null ||
418                info.getNumBytes() > existing.rInfo.getNumBytes()) {
419              // if we have >1 replicas for the same internal block, we
420              // simply choose the one with larger length.
421              // TODO: better usage of redundant replicas
422              syncBlocks.put(blockId, new BlockRecord(id, proxyDN, info));
423            }
424          }
425        } catch (RecoveryInProgressException ripE) {
426          InterDatanodeProtocol.LOG.warn(
427              "Recovery for replica " + block + " on data-node " + id
428                  + " is already in progress. Recovery id = "
429                  + rBlock.getNewGenerationStamp() + " is aborted.", ripE);
430          return;
431        } catch (IOException e) {
432          InterDatanodeProtocol.LOG.warn(
433              "Failed to obtain replica info for block (=" + block
434                  + ") from datanode (=" + id + ")", e);
435        }
436      }
437      checkLocations(syncBlocks.size());
438
439      final long safeLength = getSafeLength(syncBlocks);
440      if (LOG.isDebugEnabled()) {
441        LOG.debug("Recovering block " + block
442            + ", length=" + block.getNumBytes() + ", safeLength=" + safeLength
443            + ", syncList=" + syncBlocks);
444      }
445
446      // If some internal blocks reach the safe length, convert them to RUR
447      List<BlockRecord> rurList = new ArrayList<>(locs.length);
448      for (BlockRecord r : syncBlocks.values()) {
449        int blockIndex = (int) (r.rInfo.getBlockId() & BLOCK_GROUP_INDEX_MASK);
450        long newSize = getInternalBlockLength(safeLength, ecPolicy.getCellSize(),
451            dataBlkNum, blockIndex);
452        if (r.rInfo.getNumBytes() >= newSize) {
453          rurList.add(r);
454        }
455      }
456      assert rurList.size() >= dataBlkNum : "incorrect safe length";
457
458      // Recovery the striped block by truncating internal blocks to the safe
459      // length. Abort if there is any failure in this step.
460      truncatePartialBlock(rurList, safeLength);
461
462      // notify Namenode the new size and locations
463      final DatanodeID[] newLocs = new DatanodeID[totalBlkNum];
464      final String[] newStorages = new String[totalBlkNum];
465      for (int i = 0; i < totalBlkNum; i++) {
466        newLocs[blockIndices[i]] = DatanodeID.EMPTY_DATANODE_ID;
467        newStorages[blockIndices[i]] = "";
468      }
469      for (BlockRecord r : rurList) {
470        int index = (int) (r.rInfo.getBlockId() &
471            HdfsServerConstants.BLOCK_GROUP_INDEX_MASK);
472        newLocs[index] = r.id;
473        newStorages[index] = r.storageID;
474      }
475      ExtendedBlock newBlock = new ExtendedBlock(bpid, block.getBlockId(),
476          safeLength, recoveryId);
477      DatanodeProtocolClientSideTranslatorPB nn = getActiveNamenodeForBP(bpid);
478      nn.commitBlockSynchronization(block, newBlock.getGenerationStamp(),
479          newBlock.getNumBytes(), true, false, newLocs, newStorages);
480    }
481
482    private void truncatePartialBlock(List<BlockRecord> rurList,
483        long safeLength) throws IOException {
484      int cellSize = ecPolicy.getCellSize();
485      int dataBlkNum = ecPolicy.getNumDataUnits();
486      List<DatanodeID> failedList = new ArrayList<>();
487      for (BlockRecord r : rurList) {
488        int blockIndex = (int) (r.rInfo.getBlockId() & BLOCK_GROUP_INDEX_MASK);
489        long newSize = getInternalBlockLength(safeLength, cellSize, dataBlkNum,
490            blockIndex);
491        try {
492          r.updateReplicaUnderRecovery(bpid, recoveryId, r.rInfo.getBlockId(),
493              newSize);
494        } catch (IOException e) {
495          InterDatanodeProtocol.LOG.warn("Failed to updateBlock (newblock="
496              + ", datanode=" + r.id + ")", e);
497          failedList.add(r.id);
498        }
499      }
500
501      // If any of the data-nodes failed, the recovery fails, because
502      // we never know the actual state of the replica on failed data-nodes.
503      // The recovery should be started over.
504      if (!failedList.isEmpty()) {
505        throw new IOException("Cannot recover " + block
506            + ", the following datanodes failed: " + failedList);
507      }
508    }

510    /**
511     * TODO: the current implementation depends on the assumption that the
512     * parity cells are only generated based on the full stripe. This is not
513     * true after we support hflush.
514     */
515    @VisibleForTesting
516    long getSafeLength(Map<Long, BlockRecord> syncBlocks) {
517      final int dataBlkNum = ecPolicy.getNumDataUnits();
518      Preconditions.checkArgument(syncBlocks.size() >= dataBlkNum);
519      long[] blockLengths = new long[syncBlocks.size()];
520      int i = 0;
521      for (BlockRecord r : syncBlocks.values()) {
522        ReplicaRecoveryInfo rInfo = r.getReplicaRecoveryInfo();
523        blockLengths[i++] = rInfo.getNumBytes();
524      }
525      return StripedBlockUtil.getSafeLength(ecPolicy, blockLengths);
526    }
527
528    private void checkLocations(int locationCount)
529        throws IOException {
530      if (locationCount < ecPolicy.getNumDataUnits()) {
531        throw new IOException(block + " has no enough internal blocks" +
532            ", unable to start recovery. Locations=" + Arrays.asList(locs));
533      }
534    }
535  }

537  private static void logRecoverBlock(String who, RecoveringBlock rb) {
538    ExtendedBlock block = rb.getBlock();
539    DatanodeInfo[] targets = rb.getLocations();
540
541    LOG.info(who + " calls recoverBlock(" + block
542        + ", targets=[" + Joiner.on(", ").join(targets) + "]"
543        + ", newGenerationStamp=" + rb.getNewGenerationStamp()
544        + ", newBlock=" + rb.getNewBlock()
545        + ", isStriped=" + rb.isStriped()
546        + ")");
547  }

549  /**
550   * Convenience method, which unwraps RemoteException.
551   * @throws IOException not a RemoteException.
552   */
553  private static ReplicaRecoveryInfo callInitReplicaRecovery(
554      InterDatanodeProtocol datanode, RecoveringBlock rBlock)
555      throws IOException {
556    try {
557      return datanode.initReplicaRecovery(rBlock);
558    } catch(RemoteException re) {
559      throw re.unwrapRemoteException();
560    }

563  /**
564   * Get the NameNode corresponding to the given block pool.
565   *
566   * @param bpid Block pool Id
567   * @return Namenode corresponding to the bpid
568   * @throws IOException if unable to get the corresponding NameNode
569   */
570  DatanodeProtocolClientSideTranslatorPB getActiveNamenodeForBP(
571      String bpid) throws IOException {
572    BPOfferService bpos = datanode.getBPOfferService(bpid);
573    if (bpos == null) {
574      throw new IOException("No block pool offer service for bpid=" + bpid);
575    }
576
577    DatanodeProtocolClientSideTranslatorPB activeNN = bpos.getActiveNN();
578    if (activeNN == null) {
579      throw new IOException(
580          "Block pool " + bpid + " has not recognized an active NN");
581    }
582    return activeNN;
583  }

585  public Daemon recoverBlocks(final String who,
586      final Collection<RecoveringBlock> blocks) {
587    Daemon d = new Daemon(datanode.threadGroup, new Runnable() {
588      @Override
589      public void run() {
590        for(RecoveringBlock b : blocks) {
591          try {
592            logRecoverBlock(who, b);
593            if (b.isStriped()) {
594              new RecoveryTaskStriped((RecoveringStripedBlock) b).recover();
595            } else {
596              new RecoveryTaskContiguous(b).recover();
597            }
598          } catch (IOException e) {
599            LOG.warn("recoverBlocks FAILED: " + b, e);
600          }
601        }
602      }
603    });
604    d.start();
605    return d;
606  }
