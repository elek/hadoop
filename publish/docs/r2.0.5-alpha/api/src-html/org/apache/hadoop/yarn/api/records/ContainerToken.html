019package org.apache.hadoop.yarn.api.records;
020
021import org.apache.hadoop.classification.InterfaceAudience.Public;
022import org.apache.hadoop.classification.InterfaceStability.Stable;
023import org.apache.hadoop.yarn.api.AMRMProtocol;
024import org.apache.hadoop.yarn.api.ContainerManager;
025
026/**
027 * <p><code>ContainerToken</code> is the security token used by the framework
028 * to verify authenticity of any <code>Container</code>.</p>
029 *
030 * <p>The <code>ResourceManager</code>, on container allocation provides a
031 * secure token which is verified by the <code>NodeManager</code> on 
032 * container launch.</p>
033 * 
034 * <p>Applications do not need to care about <code>ContainerToken</code>, they
035 * are transparently handled by the framework - the allocated 
036 * <code>Container</code> includes the <code>ContainerToken</code>.</p>
037 * 
038 * @see AMRMProtocol#allocate(org.apache.hadoop.yarn.api.protocolrecords.AllocateRequest)
039 * @see ContainerManager#startContainer(org.apache.hadoop.yarn.api.protocolrecords.StartContainerRequest)
040 */
041@Public
042@Stable
043public interface ContainerToken extends Token {}
