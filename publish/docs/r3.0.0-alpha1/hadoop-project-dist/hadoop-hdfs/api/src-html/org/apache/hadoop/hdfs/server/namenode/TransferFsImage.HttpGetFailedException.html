075  public enum TransferResult{
076    SUCCESS(HttpServletResponse.SC_OK, false),
077    AUTHENTICATION_FAILURE(HttpServletResponse.SC_FORBIDDEN, true),
078    NOT_ACTIVE_NAMENODE_FAILURE(HttpServletResponse.SC_EXPECTATION_FAILED, false),
079    OLD_TRANSACTION_ID_FAILURE(HttpServletResponse.SC_CONFLICT, false),
080    UNEXPECTED_FAILURE(-1, true);
081
082    private final int response;
083    private final boolean shouldReThrowException;
084
085    private TransferResult(int response, boolean rethrow) {
086      this.response = response;
087      this.shouldReThrowException = rethrow;
088    }
089
090    public static TransferResult getResultForCode(int code){
091      TransferResult ret = UNEXPECTED_FAILURE;
092      for(TransferResult result:TransferResult.values()){
093        if(result.response == code){
094          return result;
095        }
096      }
097      return ret;
098    }
099  }

131  public static MD5Hash downloadImageToStorage(URL fsName, long imageTxId,
132      Storage dstStorage, boolean needDigest, boolean isBootstrapStandby)
133      throws IOException {
134    String fileid = ImageServlet.getParamStringForImage(null,
135        imageTxId, dstStorage, isBootstrapStandby);
136    String fileName = NNStorage.getCheckpointImageFileName(imageTxId);
137    
138    List<File> dstFiles = dstStorage.getFiles(
139        NameNodeDirType.IMAGE, fileName);
140    if (dstFiles.isEmpty()) {
141      throw new IOException("No targets in destination storage!");
142    }
143    
144    MD5Hash hash = getFileClient(fsName, fileid, dstFiles, dstStorage, needDigest);
145    LOG.info("Downloaded file " + dstFiles.get(0).getName() + " size " +
146        dstFiles.get(0).length() + " bytes.");
147    return hash;
148  }

150  static MD5Hash handleUploadImageRequest(HttpServletRequest request,
151      long imageTxId, Storage dstStorage, InputStream stream,
152      long advertisedSize, DataTransferThrottler throttler) throws IOException {
153
154    String fileName = NNStorage.getCheckpointImageFileName(imageTxId);
155
156    List<File> dstFiles = dstStorage.getFiles(NameNodeDirType.IMAGE, fileName);
157    if (dstFiles.isEmpty()) {
158      throw new IOException("No targets in destination storage!");
159    }
160
161    MD5Hash advertisedDigest = parseMD5Header(request);
162    MD5Hash hash = receiveFile(fileName, dstFiles, dstStorage, true,
163        advertisedSize, advertisedDigest, fileName, stream, throttler);
164    LOG.info("Downloaded file " + dstFiles.get(0).getName() + " size "
165        + dstFiles.get(0).length() + " bytes.");
166    return hash;
167  }

169  static void downloadEditsToStorage(URL fsName, RemoteEditLog log,
170      NNStorage dstStorage) throws IOException {
171    assert log.getStartTxId() > 0 && log.getEndTxId() > 0 :
172      "bad log: " + log;
173    String fileid = ImageServlet.getParamStringForLog(
174        log, dstStorage);
175    String finalFileName = NNStorage.getFinalizedEditsFileName(
176        log.getStartTxId(), log.getEndTxId());
177
178    List<File> finalFiles = dstStorage.getFiles(NameNodeDirType.EDITS,
179        finalFileName);
180    assert !finalFiles.isEmpty() : "No checkpoint targets.";
181    
182    for (File f : finalFiles) {
183      if (f.exists() && FileUtil.canRead(f)) {
184        LOG.info("Skipping download of remote edit log " +
185            log + " since it already is stored locally at " + f);
186        return;
187      } else if (LOG.isDebugEnabled()) {
188        LOG.debug("Dest file: " + f);
189      }
190    }
191
192    final long milliTime = Time.monotonicNow();
193    String tmpFileName = NNStorage.getTemporaryEditsFileName(
194        log.getStartTxId(), log.getEndTxId(), milliTime);
195    List<File> tmpFiles = dstStorage.getFiles(NameNodeDirType.EDITS,
196        tmpFileName);
197    getFileClient(fsName, fileid, tmpFiles, dstStorage, false);
198    LOG.info("Downloaded file " + tmpFiles.get(0).getName() + " size " +
199        finalFiles.get(0).length() + " bytes.");
200
201    CheckpointFaultInjector.getInstance().beforeEditsRename();
202
203    for (StorageDirectory sd : dstStorage.dirIterable(NameNodeDirType.EDITS)) {
204      File tmpFile = NNStorage.getTemporaryEditsFile(sd,
205          log.getStartTxId(), log.getEndTxId(), milliTime);
206      File finalizedFile = NNStorage.getFinalizedEditsFile(sd,
207          log.getStartTxId(), log.getEndTxId());
208      if (LOG.isDebugEnabled()) {
209        LOG.debug("Renaming " + tmpFile + " to " + finalizedFile);
210      }
211      boolean success = tmpFile.renameTo(finalizedFile);
212      if (!success) {
213        LOG.warn("Unable to rename edits file from " + tmpFile
214            + " to " + finalizedFile);
215      }
216    }
217  }

234  /**
235   * Requests that the NameNode download an image from this node.  Allows for
236   * optional external cancelation.
237   *
238   * @param fsName the http address for the remote NN
239   * @param conf Configuration
240   * @param storage the storage directory to transfer the image from
241   * @param nnf the NameNodeFile type of the image
242   * @param txid the transaction ID of the image to be uploaded
243   * @param canceler optional canceler to check for abort of upload
244   * @throws IOException if there is an I/O error or cancellation
245   */
246  public static TransferResult uploadImageFromStorage(URL fsName, Configuration conf,
247      NNStorage storage, NameNodeFile nnf, long txid, Canceler canceler)
248      throws IOException {
249    URL url = new URL(fsName, ImageServlet.PATH_SPEC);
250    long startTime = Time.monotonicNow();
251    try {
252      uploadImage(url, conf, storage, nnf, txid, canceler);
253    } catch (HttpPutFailedException e) {
254      // translate the error code to a result, which is a bit more obvious in usage
255      TransferResult result = TransferResult.getResultForCode(e.getResponseCode());
256      if (result.shouldReThrowException) {
257        throw e;
258      }
259      return result;
260    }
261    double xferSec = Math.max(
262        ((float) (Time.monotonicNow() - startTime)) / 1000.0, 0.001);
263    LOG.info("Uploaded image with txid " + txid + " to namenode at " + fsName
264        + " in " + xferSec + " seconds");
265    return TransferResult.SUCCESS;
266  }

268  /*
269   * Uploads the imagefile using HTTP PUT method
270   */
271  private static void uploadImage(URL url, Configuration conf,
272      NNStorage storage, NameNodeFile nnf, long txId, Canceler canceler)
273      throws IOException {
274
275    File imageFile = storage.findImageFile(nnf, txId);
276    if (imageFile == null) {
277      throw new IOException("Could not find image with txid " + txId);
278    }
279
280    HttpURLConnection connection = null;
281    try {
282      URIBuilder uriBuilder = new URIBuilder(url.toURI());
283
284      // write all params for image upload request as query itself.
285      // Request body contains the image to be uploaded.
286      Map<String, String> params = ImageServlet.getParamsForPutImage(storage,
287          txId, imageFile.length(), nnf);
288      for (Entry<String, String> entry : params.entrySet()) {
289        uriBuilder.addParameter(entry.getKey(), entry.getValue());
290      }
291
292      URL urlWithParams = uriBuilder.build().toURL();
293      connection = (HttpURLConnection) connectionFactory.openConnection(
294          urlWithParams, UserGroupInformation.isSecurityEnabled());
295      // Set the request to PUT
296      connection.setRequestMethod("PUT");
297      connection.setDoOutput(true);
298
299      
300      int chunkSize = conf.getInt(
301          DFSConfigKeys.DFS_IMAGE_TRANSFER_CHUNKSIZE_KEY,
302          DFSConfigKeys.DFS_IMAGE_TRANSFER_CHUNKSIZE_DEFAULT);
303      if (imageFile.length() > chunkSize) {
304        // using chunked streaming mode to support upload of 2GB+ files and to
305        // avoid internal buffering.
306        // this mode should be used only if more than chunkSize data is present
307        // to upload. otherwise upload may not happen sometimes.
308        connection.setChunkedStreamingMode(chunkSize);
309      }
310
311      setTimeout(connection);
312
313      // set headers for verification
314      ImageServlet.setVerificationHeadersForPut(connection, imageFile);
315
316      // Write the file to output stream.
317      writeFileToPutRequest(conf, connection, imageFile, canceler);
318
319      int responseCode = connection.getResponseCode();
320      if (responseCode != HttpURLConnection.HTTP_OK) {
321        throw new HttpPutFailedException(String.format(
322            "Image uploading failed, status: %d, url: %s, message: %s",
323            responseCode, urlWithParams, connection.getResponseMessage()),
324            responseCode);
325      }
326    } catch (AuthenticationException e) {
327      throw new IOException(e);
328    } catch (URISyntaxException e) {
329      throw new IOException(e);
330    } finally {
331      if (connection != null) {
332        connection.disconnect();
333      }
334    }
335  }
336
337  private static void writeFileToPutRequest(Configuration conf,
338      HttpURLConnection connection, File imageFile, Canceler canceler)
339      throws FileNotFoundException, IOException {
340    connection.setRequestProperty(CONTENT_TYPE, "application/octet-stream");
341    connection.setRequestProperty(CONTENT_TRANSFER_ENCODING, "binary");
342    OutputStream output = connection.getOutputStream();
343    FileInputStream input = new FileInputStream(imageFile);
344    try {
345      copyFileToStream(output, imageFile, input,
346          ImageServlet.getThrottler(conf), canceler);
347    } finally {
348      IOUtils.closeStream(input);
349      IOUtils.closeStream(output);
350    }
351  }

363  private static void copyFileToStream(OutputStream out, File localfile,
364      FileInputStream infile, DataTransferThrottler throttler,
365      Canceler canceler) throws IOException {
366    byte buf[] = new byte[IO_FILE_BUFFER_SIZE];
367    try {
368      CheckpointFaultInjector.getInstance()
369          .aboutToSendFile(localfile);
370
371      if (CheckpointFaultInjector.getInstance().
372            shouldSendShortFile(localfile)) {
373          // Test sending image shorter than localfile
374          long len = localfile.length();
375          buf = new byte[(int)Math.min(len/2, IO_FILE_BUFFER_SIZE)];
376          // This will read at most half of the image
377          // and the rest of the image will be sent over the wire
378          infile.read(buf);
379      }
380      int num = 1;
381      while (num > 0) {
382        if (canceler != null && canceler.isCancelled()) {
383          throw new SaveNamespaceCancelledException(
384            canceler.getCancellationReason());
385        }
386        num = infile.read(buf);
387        if (num <= 0) {
388          break;
389        }
390        if (CheckpointFaultInjector.getInstance()
391              .shouldCorruptAByte(localfile)) {
392          // Simulate a corrupted byte on the wire
393          LOG.warn("SIMULATING A CORRUPT BYTE IN IMAGE TRANSFER!");
394          buf[0]++;
395        }
396        
397        out.write(buf, 0, num);
398        if (throttler != null) {
399          throttler.throttle(num, canceler);
400        }
401      }
402    } catch (EofException e) {
403      LOG.info("Connection closed by client");
404      out = null; // so we don't close in the finally
405    } finally {
406      if (out != null) {
407        out.close();
408      }
409    }
410  }

413   * Client-side Method to fetch file from a server
414   * Copies the response from the URL to a list of local files.
415   * @param dstStorage if an error occurs writing to one of the files,
416   *                   this storage object will be notified. 
417   * @Return a digest of the received file if getChecksum is true
418   */
419  static MD5Hash getFileClient(URL infoServer,
420      String queryString, List<File> localPaths,
421      Storage dstStorage, boolean getChecksum) throws IOException {
422    URL url = new URL(infoServer, ImageServlet.PATH_SPEC + "?" + queryString);
423    LOG.info("Opening connection to " + url);
424    return doGetUrl(url, localPaths, dstStorage, getChecksum);
425  }

427  public static MD5Hash doGetUrl(URL url, List<File> localPaths,
428      Storage dstStorage, boolean getChecksum) throws IOException {
429    HttpURLConnection connection;
430    try {
431      connection = (HttpURLConnection)
432        connectionFactory.openConnection(url, isSpnegoEnabled);
433    } catch (AuthenticationException e) {
434      throw new IOException(e);
435    }
436
437    setTimeout(connection);
438
439    if (connection.getResponseCode() != HttpURLConnection.HTTP_OK) {
440      throw new HttpGetFailedException(
441          "Image transfer servlet at " + url +
442          " failed with status code " + connection.getResponseCode() +
443          "\nResponse message:\n" + connection.getResponseMessage(),
444          connection);
445    }
446    
447    long advertisedSize;
448    String contentLength = connection.getHeaderField(CONTENT_LENGTH);
449    if (contentLength != null) {
450      advertisedSize = Long.parseLong(contentLength);
451    } else {
452      throw new IOException(CONTENT_LENGTH + " header is not provided " +
453                            "by the namenode when trying to fetch " + url);
454    }
455    MD5Hash advertisedDigest = parseMD5Header(connection);
456    String fsImageName = connection
457        .getHeaderField(ImageServlet.HADOOP_IMAGE_EDITS_HEADER);
458    InputStream stream = connection.getInputStream();
459
460    return receiveFile(url.toExternalForm(), localPaths, dstStorage,
461        getChecksum, advertisedSize, advertisedDigest, fsImageName, stream,
462        null);
463  }

465  private static void setTimeout(HttpURLConnection connection) {
466    if (timeout <= 0) {
467      Configuration conf = new HdfsConfiguration();
468      timeout = conf.getInt(DFSConfigKeys.DFS_IMAGE_TRANSFER_TIMEOUT_KEY,
469          DFSConfigKeys.DFS_IMAGE_TRANSFER_TIMEOUT_DEFAULT);
470      LOG.info("Image Transfer timeout configured to " + timeout
471          + " milliseconds");
472    }
473
474    if (timeout > 0) {
475      connection.setConnectTimeout(timeout);
476      connection.setReadTimeout(timeout);
477    }
478  }

480  private static MD5Hash receiveFile(String url, List<File> localPaths,
481      Storage dstStorage, boolean getChecksum, long advertisedSize,
482      MD5Hash advertisedDigest, String fsImageName, InputStream stream,
483      DataTransferThrottler throttler) throws IOException {
484    long startTime = Time.monotonicNow();
485    Map<FileOutputStream, File> streamPathMap = new HashMap<>();
486    StringBuilder xferStats = new StringBuilder();
487    double xferCombined = 0;
488    if (localPaths != null) {
489      // If the local paths refer to directories, use the server-provided header
490      // as the filename within that directory
491      List<File> newLocalPaths = new ArrayList<File>();
492      for (File localPath : localPaths) {
493        if (localPath.isDirectory()) {
494          if (fsImageName == null) {
495            throw new IOException("No filename header provided by server");
496          }
497          newLocalPaths.add(new File(localPath, fsImageName));
498        } else {
499          newLocalPaths.add(localPath);
500        }
501      }
502      localPaths = newLocalPaths;
503    }
504    
505
506    long received = 0;
507    MessageDigest digester = null;
508    if (getChecksum) {
509      digester = MD5Hash.getDigester();
510      stream = new DigestInputStream(stream, digester);
511    }
512    boolean finishedReceiving = false;
513
514    List<FileOutputStream> outputStreams = Lists.newArrayList();
515
516    try {
517      if (localPaths != null) {
518        for (File f : localPaths) {
519          try {
520            if (f.exists()) {
521              LOG.warn("Overwriting existing file " + f
522                  + " with file downloaded from " + url);
523            }
524            FileOutputStream fos = new FileOutputStream(f);
525            outputStreams.add(fos);
526            streamPathMap.put(fos, f);
527          } catch (IOException ioe) {
528            LOG.warn("Unable to download file " + f, ioe);
529            // This will be null if we're downloading the fsimage to a file
530            // outside of an NNStorage directory.
531            if (dstStorage != null &&
532                (dstStorage instanceof StorageErrorReporter)) {
533              ((StorageErrorReporter)dstStorage).reportErrorOnFile(f);
534            }
535          }
536        }
537        
538        if (outputStreams.isEmpty()) {
539          throw new IOException(
540              "Unable to download to any storage directory");
541        }
542      }
543      
544      int num = 1;
545      byte[] buf = new byte[IO_FILE_BUFFER_SIZE];
546      while (num > 0) {
547        num = stream.read(buf);
548        if (num > 0) {
549          received += num;
550          for (FileOutputStream fos : outputStreams) {
551            fos.write(buf, 0, num);
552          }
553          if (throttler != null) {
554            throttler.throttle(num);
555          }
556        }
557      }
558      finishedReceiving = true;
559      double xferSec = Math.max(
560                 ((float)(Time.monotonicNow() - startTime)) / 1000.0, 0.001);
561      long xferKb = received / 1024;
562      xferCombined += xferSec;
563      xferStats.append(
564          String.format(" The fsimage download took %.2fs at %.2f KB/s.",
565              xferSec, xferKb / xferSec));
566    } finally {
567      stream.close();
568      for (FileOutputStream fos : outputStreams) {
569        long flushStartTime = Time.monotonicNow();
570        fos.getChannel().force(true);
571        fos.close();
572        double writeSec = Math.max(((float)
573               (flushStartTime - Time.monotonicNow())) / 1000.0, 0.001);
574        xferCombined += writeSec;
575        xferStats.append(String
576                .format(" Synchronous (fsync) write to disk of " +
577                 streamPathMap.get(fos).getAbsolutePath() +
578                " took %.2fs.", writeSec));
579      }
580
581      // Something went wrong and did not finish reading.
582      // Remove the temporary files.
583      if (!finishedReceiving) {
584        deleteTmpFiles(localPaths);
585      }
586
587      if (finishedReceiving && received != advertisedSize) {
588        // only throw this exception if we think we read all of it on our end
589        // -- otherwise a client-side IOException would be masked by this
590        // exception that makes it look like a server-side problem!
591        deleteTmpFiles(localPaths);
592        throw new IOException("File " + url + " received length " + received +
593                              " is not of the advertised size " +
594                              advertisedSize);
595      }
596    }
597    xferStats.insert(
598        0, String.format(
599            "Combined time for fsimage download and fsync " +
600            "to all disks took %.2fs.", xferCombined));
601    LOG.info(xferStats.toString());
602
603    if (digester != null) {
604      MD5Hash computedDigest = new MD5Hash(digester.digest());
605      
606      if (advertisedDigest != null &&
607          !computedDigest.equals(advertisedDigest)) {
608        deleteTmpFiles(localPaths);
609        throw new IOException("File " + url + " computed digest " +
610            computedDigest + " does not match advertised digest " + 
611            advertisedDigest);
612      }
613      return computedDigest;
614    } else {
615      return null;
616    }    
617  }

619  private static void deleteTmpFiles(List<File> files) {
620    if (files == null) {
621      return;
622    }
623
624    LOG.info("Deleting temporary files: " + files);
625    for (File file : files) {
626      if (!file.delete()) {
627        LOG.warn("Deleting " + file + " has failed");
628      }
629    }
630  }

656  public static class HttpPutFailedException extends IOException {
