041/**
042 * DiskBalancer is a tool that can be used to ensure that data is spread evenly
043 * across volumes of same storage type.
044 * <p>
045 * For example, if you have 3 disks, with 100 GB , 600 GB and 200 GB on each
046 * disk, this tool will ensure that each disk will have 300 GB.
047 * <p>
048 * This tool can be run while data nodes are fully functional.
049 * <p>
050 * At very high level diskbalancer computes a set of moves that will make disk
051 * utilization equal and then those moves are executed by the datanode.
052 */
053public class DiskBalancer extends Configured implements Tool {

056   */

147  /**
148   * Main for the  DiskBalancer Command handling.
149   *
150   * @param argv - System Args Strings[]
151   * @throws Exception
152   */
153  public static void main(String[] argv) throws Exception {
154    DiskBalancer shell = new DiskBalancer(new HdfsConfiguration());
155    int res = 0;
156    try {
157      res = ToolRunner.run(shell, argv);
158    } catch (Exception ex) {
159      LOG.error(ex.toString());
160      res = 1;
161    }
162    System.exit(res);
163  }

177  /**
178   * Execute the command with the given arguments.
179   *
180   * @param args command specific arguments.
181   * @param out  the output stream used for printing

183   * @throws Exception

185  public int run(String[] args, final PrintStream out) throws Exception {

261  /**
262   * Adds commands for plan command.
263   *
264   * @return Options.
265   */
266  private void addPlanCommands(Options opt) {
267
268    Option plan = OptionBuilder.withLongOpt(PLAN)
269        .withDescription("Hostname, IP address or UUID of datanode " +
270            "for which a plan is created.")
271        .hasArg()
272        .create();
273    getPlanOptions().addOption(plan);
274    opt.addOption(plan);
275
276
277    Option outFile = OptionBuilder.withLongOpt(OUTFILE).hasArg()
278        .withDescription(
279            "Local path of file to write output to, if not specified "
280                + "defaults will be used.")
281        .create();
282    getPlanOptions().addOption(outFile);
283    opt.addOption(outFile);
284
285    Option bandwidth = OptionBuilder.withLongOpt(BANDWIDTH).hasArg()
286        .withDescription(
287            "Maximum disk bandwidth (MB/s) in integer to be consumed by "
288                + "diskBalancer. e.g. 10 MB/s.")
289        .create();
290    getPlanOptions().addOption(bandwidth);
291    opt.addOption(bandwidth);
292
293    Option threshold = OptionBuilder.withLongOpt(THRESHOLD)
294        .hasArg()
295        .withDescription("Percentage of data skew that is tolerated before"
296            + " disk balancer starts working. For example, if"
297            + " total data on a 2 disk node is 100 GB then disk"
298            + " balancer calculates the expected value on each disk,"
299            + " which is 50 GB. If the tolerance is 10% then data"
300            + " on a single disk needs to be more than 60 GB"
301            + " (50 GB + 10% tolerance value) for Disk balancer to"
302            + " balance the disks.")
303        .create();
304    getPlanOptions().addOption(threshold);
305    opt.addOption(threshold);
306
307
308    Option maxError = OptionBuilder.withLongOpt(MAXERROR)
309        .hasArg()
310        .withDescription("Describes how many errors " +
311            "can be tolerated while copying between a pair of disks.")
312        .create();
313    getPlanOptions().addOption(maxError);
314    opt.addOption(maxError);
315
316    Option verbose = OptionBuilder.withLongOpt(VERBOSE)
317        .withDescription("Print out the summary of the plan on console")
318        .create();
319    getPlanOptions().addOption(verbose);
320    opt.addOption(verbose);
321  }

426  /**
427   * This function parses all command line arguments and returns the appropriate

439  /**
440   * Dispatches calls to the right command Handler classes.
441   *
442   * @param cmd  - CommandLine
443   * @param opts options of command line
444   * @param out  the output stream used for printing
445   */
446  private int dispatch(CommandLine cmd, Options opts, final PrintStream out)
447      throws Exception {
448    Command currentCommand = null;
449    if (cmd.hasOption(DiskBalancer.PLAN)) {
450      currentCommand = new PlanCommand(getConf());
451    }
452
453    if (cmd.hasOption(DiskBalancer.EXECUTE)) {
454      currentCommand = new ExecuteCommand(getConf());
455    }
456
457    if (cmd.hasOption(DiskBalancer.QUERY)) {
458      currentCommand = new QueryCommand(getConf());
459    }
460
461    if (cmd.hasOption(DiskBalancer.CANCEL)) {
462      currentCommand = new CancelCommand(getConf());
463    }
464
465    if (cmd.hasOption(DiskBalancer.REPORT)) {
466      currentCommand = new ReportCommand(getConf(), out);
467    }
468
469    if (cmd.hasOption(DiskBalancer.HELP)) {
470      currentCommand = new HelpCommand(getConf());
471    }
472
473    // Invoke main help here.
474    if (currentCommand == null) {
475      new HelpCommand(getConf()).execute(null);
476      return 1;
477    }
478
479    currentCommand.execute(cmd);
480    return 0;
481  }
