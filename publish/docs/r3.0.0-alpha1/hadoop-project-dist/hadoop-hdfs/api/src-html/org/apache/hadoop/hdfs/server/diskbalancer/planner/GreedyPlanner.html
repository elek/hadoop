018package org.apache.hadoop.hdfs.server.diskbalancer.planner;
019
020import com.google.common.base.Preconditions;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;
023import org.apache.hadoop.hdfs.server.diskbalancer.datamodel
024    .DiskBalancerDataNode;
025import org.apache.hadoop.hdfs.server.diskbalancer.datamodel.DiskBalancerVolume;
026import org.apache.hadoop.hdfs.server.diskbalancer.datamodel
027    .DiskBalancerVolumeSet;
028import org.apache.hadoop.util.Time;
029
030import java.util.Iterator;
031import java.util.List;
032import java.util.TreeSet;
033
034/**
035 * Greedy Planner is a simple planner that computes the largest possible move at
036 * any point of time given a volumeSet.
037 * <p/>
038 * This is done by choosing the disks with largest  amount of data above and
039 * below the idealStorage and then a move is scheduled between them.
040 */
041public class GreedyPlanner implements Planner {
042  public static final long MB = 1024L * 1024L;
043  public static final long GB = MB * 1024L;
044  public static final long TB = GB * 1024L;
045  private static final Logger LOG =
046      LoggerFactory.getLogger(GreedyPlanner.class);
047  private final double threshold;
048
049  /**
050   * Constructs a greedy planner.
051   *
052   * @param threshold - Disk tolerance that we are ok with
053   * @param node      - node on which this planner is operating upon
054   */
055  public GreedyPlanner(double threshold, DiskBalancerDataNode node) {
056    this.threshold = threshold;
057  }
058
059  /**
060   * Computes a node plan for the given node.
061   *
062   * @return NodePlan
063   * @throws Exception
064   */
065  @Override
066  public NodePlan plan(DiskBalancerDataNode node) throws Exception {
067    long startTime = Time.monotonicNow();
068    NodePlan plan = new NodePlan(node.getDataNodeName(),
069        node.getDataNodePort());
070    LOG.info("Starting plan for Node : {}:{}",
071        node.getDataNodeName(), node.getDataNodePort());
072    while (node.isBalancingNeeded(this.threshold)) {
073      for (DiskBalancerVolumeSet vSet : node.getVolumeSets().values()) {
074        balanceVolumeSet(node, vSet, plan);
075      }
076    }
077
078    long endTime = Time.monotonicNow();
079    String message = String
080        .format("Compute Plan for Node : %s:%d took %d ms ",
081            node.getDataNodeName(), node.getDataNodePort(),
082            endTime - startTime);
083    LOG.info(message);
084    return plan;
085  }
086
087  /**
088   * Computes Steps to make a DiskBalancerVolumeSet Balanced.
089   *
090   * @param node
091   * @param vSet - DiskBalancerVolumeSet
092   * @param plan - NodePlan
093   */
094  public void balanceVolumeSet(DiskBalancerDataNode node,
095                               DiskBalancerVolumeSet vSet, NodePlan plan)
096      throws Exception {
097    Preconditions.checkNotNull(vSet);
098    Preconditions.checkNotNull(plan);
099    Preconditions.checkNotNull(node);
100    DiskBalancerVolumeSet currentSet = new DiskBalancerVolumeSet(vSet);
101
102    while (currentSet.isBalancingNeeded(this.threshold)) {
103      removeSkipVolumes(currentSet);
104
105      DiskBalancerVolume lowVolume = currentSet.getSortedQueue().first();
106      DiskBalancerVolume highVolume = currentSet.getSortedQueue().last();
107
108      Step nextStep = null;
109      // ok both volumes bytes used are in the range that we expect
110      // Then we create a move request.
111      if (!lowVolume.isSkip() && !highVolume.isSkip()) {
112        nextStep = computeMove(currentSet, lowVolume, highVolume);
113      } else {
114        LOG.debug("Skipping compute move. lowVolume: {} highVolume: {}",
115            lowVolume.getPath(), highVolume.getPath());
116      }
117
118      applyStep(nextStep, currentSet, lowVolume, highVolume);
119      if (nextStep != null) {
120        LOG.debug("Step : {} ",  nextStep.toString());
121        plan.addStep(nextStep);
122      }
123    }
124
125    String message = String
126        .format("Disk Volume set %s Type : %s plan completed.",
127            currentSet.getSetID(),
128            currentSet.getVolumes().get(0).getStorageType());
129
130    plan.setNodeName(node.getDataNodeName());
131    plan.setNodeUUID(node.getDataNodeUUID());
132    plan.setTimeStamp(Time.now());
133    plan.setPort(node.getDataNodePort());
134    LOG.info(message);
135  }
136
137  /**
138   * Apply steps applies the current step on to a volumeSet so that we can
139   * compute next steps until we reach the desired goals.
140   *
141   * @param nextStep   - nextStep or Null
142   * @param currentSet - Current Disk BalancerVolume Set we are operating upon
143   * @param lowVolume  - volume
144   * @param highVolume - volume
145   */
146  private void applyStep(Step nextStep, DiskBalancerVolumeSet currentSet,
147                         DiskBalancerVolume lowVolume,
148                         DiskBalancerVolume highVolume) throws Exception {
149
150    long used;
151    if (nextStep != null) {
152      used = lowVolume.getUsed() + nextStep.getBytesToMove();
153      lowVolume.setUsed(used);
154
155      used = highVolume.getUsed() - nextStep.getBytesToMove();
156      highVolume.setUsed(used);
157    }
158
159    // since the volume data changed , we need to recompute the DataDensity.
160    currentSet.computeVolumeDataDensity();
161    printQueue(currentSet.getSortedQueue());
162  }
163
164  /**
165   * Computes a data move from the largest disk we have to smallest disk.
166   *
167   * @param currentSet - Current Disk Set we are working with
168   * @param lowVolume  - Low Data Capacity Volume
169   * @param highVolume - High Data Capacity Volume
170   * @return Step
171   */
172  private Step computeMove(DiskBalancerVolumeSet currentSet,
173                           DiskBalancerVolume lowVolume,
174                           DiskBalancerVolume highVolume) {
175    // Compute how many bytes we can move. First Compute the maximum that
176    // low Volume Can receive, then compute maximum high volume can give
177    // Then take the minimum of those two numbers that is the bytesToMove.
178
179    long maxLowVolumeCanReceive = (long) (
180        (currentSet.getIdealUsed() * lowVolume.computeEffectiveCapacity()) -
181            lowVolume.getUsed());
182
183    // This disk cannot take any more data from any disk.
184    // Remove it from our computation matrix.
185    if (maxLowVolumeCanReceive <= 0) {
186      LOG.debug("{} Skipping disk from computation. Maximum data size " +
187          "achieved.", lowVolume.getPath());
188      skipVolume(currentSet, lowVolume);
189    }
190
191    long maxHighVolumeCanGive = highVolume.getUsed() -
192        (long) (currentSet.getIdealUsed() *
193            highVolume.computeEffectiveCapacity());
194    // This volume cannot give any more data, remove it from the
195    // computation matrix
196    if (maxHighVolumeCanGive <= 0) {
197      LOG.debug(" {} Skipping disk from computation. Minimum data size " +
198          "achieved.", highVolume.getPath());
199      skipVolume(currentSet, highVolume);
200    }
201
202
203    long bytesToMove = Math.min(maxLowVolumeCanReceive, maxHighVolumeCanGive);
204    Step nextStep = null;
205
206    if (bytesToMove > 0) {
207      // Create a new step
208      nextStep = new MoveStep(highVolume, currentSet.getIdealUsed(), lowVolume,
209          bytesToMove, currentSet.getSetID());
210      LOG.debug(nextStep.toString());
211    }
212    return nextStep;
213  }
214
215  /**
216   * Skips this volume if needed.
217   *
218   * @param currentSet - Current Disk set
219   * @param volume     - Volume
220   */
221  private void skipVolume(DiskBalancerVolumeSet currentSet,
222                          DiskBalancerVolume volume) {
223    if (LOG.isDebugEnabled()) {
224      String message =
225          String.format(
226              "Skipping volume. Volume : %s " +
227              "Type : %s Target " +
228              "Number of bytes : %f lowVolume dfsUsed : %d. Skipping this " +
229              "volume from all future balancing calls.", volume.getPath(),
230              volume.getStorageType(),
231              currentSet.getIdealUsed() * volume.getCapacity(),
232              volume.getUsed());
233      LOG.debug(message);
234    }
235    volume.setSkip(true);
236  }
237
238  // Removes all volumes which are part of the volumeSet but skip flag is set.
239  private void removeSkipVolumes(DiskBalancerVolumeSet currentSet) {
240    List<DiskBalancerVolume> volumeList = currentSet.getVolumes();
241    Iterator<DiskBalancerVolume> volumeIterator = volumeList.iterator();
242    while (volumeIterator.hasNext()) {
243      DiskBalancerVolume vol = volumeIterator.next();
244      if (vol.isSkip() || vol.isFailed()) {
245        currentSet.removeVolume(vol);
246      }
247    }
248    currentSet.computeVolumeDataDensity();
249    printQueue(currentSet.getSortedQueue());
250  }
251
252  /**
253   * This function is used only for debugging purposes to ensure queue looks
254   * correct.
255   *
256   * @param queue - Queue
257   */
258  private void printQueue(TreeSet<DiskBalancerVolume> queue) {
259    if (LOG.isDebugEnabled()) {
260      String format =
261          String.format(
262              "First Volume : %s, DataDensity : %f, " +
263              "Last Volume : %s, DataDensity : %f",
264              queue.first().getPath(), queue.first().getVolumeDataDensity(),
265              queue.last().getPath(), queue.last().getVolumeDataDensity());
266      LOG.debug(format);
267    }
268  }
