019package org.apache.hadoop.tools;
020
021import java.io.PrintStream;
022
023import org.apache.hadoop.conf.Configured;
024import org.apache.hadoop.util.Tool;
025
026/**
027 * This program is a CLI utility base class utilizing hadoop Tool class.
028 */
029public abstract class CommandShell extends Configured implements Tool {
030
031  private PrintStream out = System.out;
032  private PrintStream err = System.err;
033
034  /** The subcommand instance for this shell command, if any. */
035  private SubCommand subcommand = null;
036
037  /**
038   * Return usage string for the command including any summary of subcommands.
039   */
040  public abstract String getCommandUsage();
041
042  public void setSubCommand(SubCommand cmd) {
043    subcommand = cmd;
044  }
045
046  public void setOut(PrintStream p) {
047    out = p;
048  }
049
050  public PrintStream getOut() {
051    return out;
052  }
053
054  public void setErr(PrintStream p) {
055    err = p;
056  }
057
058  public PrintStream getErr() {
059    return err;
060  }
061
062  @Override
063  public int run(String[] args) throws Exception {
064    int exitCode = 0;
065    try {
066      exitCode = init(args);
067      if (exitCode != 0 || subcommand == null) {
068        printShellUsage();
069        return exitCode;
070      }
071      if (subcommand.validate()) {
072        subcommand.execute();
073      } else {
074        printShellUsage();
075        exitCode = 1;
076      }
077    } catch (Exception e) {
078      printShellUsage();
079      e.printStackTrace(err);
080      return 1;
081    }
082    return exitCode;
083  }
084
085  /**
086   * Parse the command line arguments and initialize subcommand instance.
087   * @param args
088   * @return 0 if the argument(s) were recognized, 1 otherwise
089   */
090  protected abstract int init(String[] args) throws Exception;
091
092  protected final void printShellUsage() {
093    if (subcommand != null) {
094      out.println(subcommand.getUsage());
095    } else {
096      out.println(getCommandUsage());
097    }
098    out.flush();
099  }
100
101  /**
102   * Base class for any subcommands of this shell command.
103   */
104  protected abstract class SubCommand {
105
106    public boolean validate() {
107      return true;
108    }
109
110    public abstract void execute() throws Exception;
111
112    public abstract String getUsage();
113  }
114}
