022import java.io.IOException;
023import java.io.PrintStream;
024import java.text.DateFormat;
025import java.util.ArrayList;
026import java.util.Date;
027import java.util.ServiceLoader;
028
029import org.apache.commons.lang.StringUtils;
030import org.apache.commons.logging.Log;
031import org.apache.commons.logging.LogFactory;

033import org.apache.hadoop.conf.Configuration;
034import org.apache.hadoop.fs.Path;
035import org.apache.hadoop.io.Text;
036import org.apache.hadoop.security.Credentials;
037import org.apache.hadoop.security.token.delegation.AbstractDelegationTokenIdentifier;

039/**
040 * DtFileOperations is a collection of delegation token file operations.
041 */
042public final class DtFileOperations {
043  private static final Log LOG = LogFactory.getLog(DtFileOperations.class);
044
045  /** No public constructor as per checkstyle. */
046  private DtFileOperations() { }
047
048  /**
049   * Use FORMAT_* as arguments to format parameters.
050   * FORMAT_PB is for protobuf output.
051   */
052  public static final String FORMAT_PB = "protobuf";
053
054  /**
055   * Use FORMAT_* as arguments to format parameters.
056   * FORMAT_JAVA is a legacy option for java serialization output.
057   */
058  public static final String FORMAT_JAVA = "java";
059
060  private static final String NA_STRING = "-NA-";
061  private static final String PREFIX_HTTP = "http://";
062  private static final String PREFIX_HTTPS = "https://";
063
064  /** Let the DtFetcher code add the appropriate prefix if HTTP/S is used. */
065  private static String stripPrefix(String u) {
066    return u.replaceFirst(PREFIX_HTTP, "").replaceFirst(PREFIX_HTTPS, "");
067  }
068
069  /** Match token service field to alias text.  True if alias is null. */
070  private static boolean matchAlias(Token<?> token, Text alias) {
071    return alias == null || token.getService().equals(alias);
072  }
073
074  /** Match fetcher's service name to the service text and/or url prefix. */
075  private static boolean matchService(
076      DtFetcher fetcher, Text service, String url) {
077    Text sName = fetcher.getServiceName();
078    return (service == null && url.startsWith(sName.toString() + "://")) ||
079           (service != null && service.equals(sName));
080  }
081
082  /** Format a long integer type into a date string. */
083  private static String formatDate(long date) {
084    DateFormat df = DateFormat.getDateTimeInstance(
085        DateFormat.SHORT, DateFormat.SHORT);
086    return df.format(new Date(date));
087  }
088
089  /** Add the service prefix for a local filesystem. */
090  private static Path fileToPath(File f) {
091    return new Path("file:" + f.getAbsolutePath());
092  }
093
094  /** Write out a Credentials object as a local file.
095   *  @param f a local File object.
096   *  @param format a string equal to FORMAT_PB or FORMAT_JAVA.
097   *  @param creds the Credentials object to be written out.
098   *  @param conf a Configuration object passed along.
099   *  @throws IOException
100   */
101  public static void doFormattedWrite(
102      File f, String format, Credentials creds, Configuration conf)
103      throws IOException {
104    if (format == null || format.equals(FORMAT_PB)) {
105      creds.writeTokenStorageFile(fileToPath(f), conf);
106    } else { // if (format != null && format.equals(FORMAT_JAVA)) {
107      creds.writeLegacyTokenStorageLocalFile(f);
108    }
109  }
110
111  /** Print out a Credentials file from the local filesystem.
112   *  @param tokenFile a local File object.
113   *  @param alias print only tokens matching alias (null matches all).
114   *  @param conf Configuration object passed along.
115   *  @param out print to this stream.
116   *  @throws IOException
117   */
118  public static void printTokenFile(
119      File tokenFile, Text alias, Configuration conf, PrintStream out)
120      throws IOException {
121    out.println("File: " + tokenFile.getPath());
122    Credentials creds = Credentials.readTokenStorageFile(tokenFile, conf);
123    printCredentials(creds, alias, out);
124  }
125
126  /** Print out a Credentials object.
127   *  @param creds the Credentials object to be printed out.
128   *  @param alias print only tokens matching alias (null matches all).
129   *  @param out print to this stream.
130   *  @throws IOException
131   */
132  public static void printCredentials(
133      Credentials creds, Text alias, PrintStream out)
134      throws IOException {
135    boolean tokenHeader = true;
136    String fmt = "%-24s %-20s %-15s %-12s %s%n";
137    for (Token<?> token : creds.getAllTokens()) {
138      if (matchAlias(token, alias)) {
139        if (tokenHeader) {
140          out.printf(fmt, "Token kind", "Service", "Renewer", "Exp date",
141                     "URL enc token");
142          out.println(StringUtils.repeat("-", 80));
143          tokenHeader = false;
144        }
145        AbstractDelegationTokenIdentifier id =
146            (AbstractDelegationTokenIdentifier) token.decodeIdentifier();
147        out.printf(fmt, token.getKind(), token.getService(),
148                   (id != null) ? id.getRenewer() : NA_STRING,
149                   (id != null) ? formatDate(id.getMaxDate()) : NA_STRING,
150                   token.encodeToUrlString());
151      }
152    }
153  }
154
155  /** Fetch a token from a service and save to file in the local filesystem.
156   *  @param tokenFile a local File object to hold the output.
157   *  @param fileFormat a string equal to FORMAT_PB or FORMAT_JAVA, for output
158   *  @param alias overwrite service field of fetched token with this text.
159   *  @param service use a DtFetcher implementation matching this service text.
160   *  @param url pass this URL to fetcher after stripping any http/s prefix.
161   *  @param renewer pass this renewer to the fetcher.
162   *  @param conf Configuration object passed along.
163   *  @throws IOException
164   */
165  public static void getTokenFile(File tokenFile, String fileFormat,
166      Text alias, Text service, String url, String renewer, Configuration conf)
167      throws Exception {
168    Token<?> token = null;
169    Credentials creds = tokenFile.exists() ?
170        Credentials.readTokenStorageFile(tokenFile, conf) : new Credentials();
171    ServiceLoader<DtFetcher> loader = ServiceLoader.load(DtFetcher.class);
172    for (DtFetcher fetcher : loader) {
173      if (matchService(fetcher, service, url)) {
174        if (!fetcher.isTokenRequired()) {
175          String message = "DtFetcher for service '" + service +
176              "' does not require a token.  Check your configuration.  " +
177              "Note: security may be disabled or there may be two DtFetcher " +
178              "providers for the same service designation.";
179          LOG.error(message);
180          throw new IllegalArgumentException(message);
181        }
182        token = fetcher.addDelegationTokens(conf, creds, renewer,
183                                            stripPrefix(url));
184      }
185    }
186    if (alias != null) {
187      if (token == null) {
188        String message = "DtFetcher for service '" + service + "'" +
189            " does not allow aliasing.  Cannot apply alias '" + alias + "'." +
190            "  Drop alias flag to get token for this service.";
191        LOG.error(message);
192        throw new IOException(message);
193      }
194      Token<?> aliasedToken = token.copyToken();
195      aliasedToken.setService(alias);
196      creds.addToken(alias, aliasedToken);
197      LOG.info("Add token with service " + alias);
198    }
199    doFormattedWrite(tokenFile, fileFormat, creds, conf);
200  }
201
202  /** Alias a token from a file and save back to file in the local filesystem.
203   *  @param tokenFile a local File object to hold the input and output.
204   *  @param fileFormat a string equal to FORMAT_PB or FORMAT_JAVA, for output
205   *  @param alias overwrite service field of fetched token with this text.
206   *  @param service only apply alias to tokens matching this service text.
207   *  @param conf Configuration object passed along.
208   *  @throws IOException
209   */
210  public static void aliasTokenFile(File tokenFile, String fileFormat,
211      Text alias, Text service, Configuration conf) throws Exception {
212    Credentials newCreds = new Credentials();
213    Credentials creds = Credentials.readTokenStorageFile(tokenFile, conf);
214    for (Token<?> token : creds.getAllTokens()) {
215      newCreds.addToken(token.getService(), token);
216      if (token.getService().equals(service)) {
217        Token<?> aliasedToken = token.copyToken();
218        aliasedToken.setService(alias);
219        newCreds.addToken(alias, aliasedToken);
220      }
221    }
222    doFormattedWrite(tokenFile, fileFormat, newCreds, conf);
223  }
224
225  /** Append tokens from list of files in local filesystem, saving to last file.
226   *  @param tokenFiles list of local File objects.  Last file holds the output.
227   *  @param fileFormat a string equal to FORMAT_PB or FORMAT_JAVA, for output
228   *  @param conf Configuration object passed along.
229   *  @throws IOException
230   */
231  public static void appendTokenFiles(
232      ArrayList<File> tokenFiles, String fileFormat, Configuration conf)
233      throws IOException {
234    Credentials newCreds = new Credentials();
235    File lastTokenFile = null;
236    for (File tokenFile : tokenFiles) {
237      lastTokenFile = tokenFile;
238      Credentials creds = Credentials.readTokenStorageFile(tokenFile, conf);
239      for (Token<?> token : creds.getAllTokens()) {
240        newCreds.addToken(token.getService(), token);
241      }
242    }
243    doFormattedWrite(lastTokenFile, fileFormat, newCreds, conf);
244  }
245
246  /** Remove a token from a file in the local filesystem, matching alias.
247   *  @param cancel cancel token as well as remove from file.
248   *  @param tokenFile a local File object.
249   *  @param fileFormat a string equal to FORMAT_PB or FORMAT_JAVA, for output
250   *  @param alias remove only tokens matching alias; null matches all.
251   *  @param conf Configuration object passed along.
252   *  @throws IOException
253   *  @throws InterruptedException
254   */
255  public static void removeTokenFromFile(boolean cancel,
256      File tokenFile, String fileFormat, Text alias, Configuration conf)
257      throws IOException, InterruptedException {
258    Credentials newCreds = new Credentials();
259    Credentials creds = Credentials.readTokenStorageFile(tokenFile, conf);
260    for (Token<?> token : creds.getAllTokens()) {
261      if (matchAlias(token, alias)) {
262        if (token.isManaged() && cancel) {
263          token.cancel(conf);
264          LOG.info("Canceled " + token.getKind() + ":" + token.getService());
265        }
266      } else {
267        newCreds.addToken(token.getService(), token);
268      }
269    }
270    doFormattedWrite(tokenFile, fileFormat, newCreds, conf);
271  }
272
273  /** Renew a token from a file in the local filesystem, matching alias.
274   *  @param tokenFile a local File object.
275   *  @param fileFormat a string equal to FORMAT_PB or FORMAT_JAVA, for output
276   *  @param alias renew only tokens matching alias; null matches all.
277   *  @param conf Configuration object passed along.
278   *  @throws IOException
279   *  @throws InterruptedException
280   */
281  public static void renewTokenFile(
282      File tokenFile, String fileFormat, Text alias, Configuration conf)
283      throws IOException, InterruptedException {
284    Credentials creds = Credentials.readTokenStorageFile(tokenFile, conf);
285    for (Token<?> token : creds.getAllTokens()) {
286      if (token.isManaged() && matchAlias(token, alias)) {
287        long result = token.renew(conf);
288        LOG.info("Renewed" + token.getKind() + ":" + token.getService() +
289                 " until " + formatDate(result));
290      }
291    }
292    doFormattedWrite(tokenFile, fileFormat, creds, conf);
293  }
