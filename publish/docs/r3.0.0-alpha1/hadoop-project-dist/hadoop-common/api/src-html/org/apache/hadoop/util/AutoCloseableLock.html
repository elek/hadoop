018package org.apache.hadoop.util;
019
020import java.util.concurrent.locks.ReentrantLock;
021
022/**
023 * This is a wrap class of a ReentrantLock. Extending AutoCloseable
024 * interface such that the users can use a try-with-resource syntax.
025 */
026public class AutoCloseableLock implements AutoCloseable {
027
028  private final ReentrantLock lock;
029
030  /**
031   * Creates an instance of {@code AutoCloseableLock}, initializes
032   * the underlying {@code ReentrantLock} object.
033   */
034  public AutoCloseableLock() {
035    this.lock = new ReentrantLock();
036  }
037
038  /**
039   * A wrapper method that makes a call to {@code lock()} of the underlying
040   * {@code ReentrantLock} object.
041   *
042   * Acquire teh lock it is not held by another thread, then sets
043   * lock held count to one, then returns immediately.
044   *
045   * If the current thread already holds the lock, increase the lock
046   * help count by one and returns immediately.
047   *
048   * If the lock is held by another thread, the current thread is
049   * suspended until the lock has been acquired by current thread.
050   *
051   * @return The {@code ReentrantLock} object itself. This is to
052   * support try-with-resource syntax.
053   */
054  public AutoCloseableLock acquire() {
055    lock.lock();
056    return this;
057  }
058
059  /**
060   * A wrapper method that makes a call to {@code unlock()} of the
061   * underlying {@code ReentrantLock} object.
062   *
063   * Attempts to release the lock.
064   *
065   * If the current thread holds the lock, decrements the hold
066   * count. If the hold count reaches zero, the lock is released.
067   *
068   * If the current thread does not hold the lock, then
069   * {@link IllegalMonitorStateException} is thrown.
070   */
071  public void release() {
072    lock.unlock();
073  }
074
075  /**
076   * Attempts to release the lock by making a call to {@code release()}.
077   *
078   * This is to implement {@code close()} method from {@code AutoCloseable}
079   * interface. This allows users to user a try-with-resource syntax, where
080   * the lock can be automatically released.
081   */
082  @Override
083  public void close() {
084    release();
085  }
086
087  /**
088   * A wrapper method that makes a call to {@code tryLock()} of
089   * the underlying {@code ReentrantLock} object.
090   *
091   * If the lock is not held by another thread, acquires the lock, set the
092   * hold count to one and returns {@code true}.
093   *
094   * If the current thread already holds the lock, the increment the hold
095   * count by one and returns {@code true}.
096   *
097   * If the lock is held by another thread then the method returns
098   * immediately with {@code false}.
099   *
100   * @return {@code true} if the lock was free and was acquired by the
101   *          current thread, or the lock was already held by the current
102   *          thread; and {@code false} otherwise.
103   */
104  public boolean tryLock() {
105    return lock.tryLock();
106  }
107
108  /**
109   * A wrapper method that makes a call to {@code isLocked()} of
110   * the underlying {@code ReentrantLock} object.
111   *
112   * Queries if this lock is held by any thread. This method is
113   * designed for use in monitoring of the system state,
114   * not for synchronization control.
115   *
116   * @return {@code true} if any thread holds this lock and
117   *         {@code false} otherwise
118   */
119  public boolean isLocked() {
120    return lock.isLocked();
121  }
122}
