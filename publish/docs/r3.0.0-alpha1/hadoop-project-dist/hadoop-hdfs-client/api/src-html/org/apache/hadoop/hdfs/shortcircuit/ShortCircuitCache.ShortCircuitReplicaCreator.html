063/**
064 * The ShortCircuitCache tracks things which the client needs to access
065 * HDFS block files via short-circuit.
066 *
067 * These things include: memory-mapped regions, file descriptors, and shared
068 * memory areas for communicating with the DataNode.
069 */
070@InterfaceAudience.Private
071public class ShortCircuitCache implements Closeable {
072  public static final Logger LOG = LoggerFactory.getLogger(
073      ShortCircuitCache.class);
074
075  /**
076   * Expiry thread which makes sure that the file descriptors get closed
077   * after a while.
078   */
079  private class CacheCleaner implements Runnable, Closeable {
080    private ScheduledFuture<?> future;
081
082    /**
083     * Run the CacheCleaner thread.
084     *
085     * Whenever a thread requests a ShortCircuitReplica object, we will make
086     * sure it gets one.  That ShortCircuitReplica object can then be re-used
087     * when another thread requests a ShortCircuitReplica object for the same
088     * block.  So in that sense, there is no maximum size to the cache.
089     *
090     * However, when a ShortCircuitReplica object is unreferenced by the
091     * thread(s) that are using it, it becomes evictable.  There are two
092     * separate eviction lists-- one for mmaped objects, and another for
093     * non-mmaped objects.  We do this in order to avoid having the regular
094     * files kick the mmaped files out of the cache too quickly.  Reusing
095     * an already-existing mmap gives a huge performance boost, since the
096     * page table entries don't have to be re-populated.  Both the mmap
097     * and non-mmap evictable lists have maximum sizes and maximum lifespans.
098     */
099    @Override
100    public void run() {
101      ShortCircuitCache.this.lock.lock();
102      try {
103        if (ShortCircuitCache.this.closed) return;
104        long curMs = Time.monotonicNow();
105
106        LOG.debug("{}: cache cleaner running at {}", this, curMs);
107
108        int numDemoted = demoteOldEvictableMmaped(curMs);
109        int numPurged = 0;
110        Long evictionTimeNs = (long) 0;
111        while (true) {
112          Entry<Long, ShortCircuitReplica> entry =
113              evictable.ceilingEntry(evictionTimeNs);
114          if (entry == null) break;
115          evictionTimeNs = entry.getKey();
116          long evictionTimeMs =
117              TimeUnit.MILLISECONDS.convert(evictionTimeNs, TimeUnit.NANOSECONDS);
118          if (evictionTimeMs + maxNonMmappedEvictableLifespanMs >= curMs) break;
119          ShortCircuitReplica replica = entry.getValue();
120          if (LOG.isTraceEnabled()) {
121            LOG.trace("CacheCleaner: purging " + replica + ": " +
122                StringUtils.getStackTrace(Thread.currentThread()));
123          }
124          purge(replica);
125          numPurged++;
126        }
127
128        LOG.debug("{}: finishing cache cleaner run started at {}. Demoted {} "
129                + "mmapped replicas; purged {} replicas.",
130            this, curMs, numDemoted, numPurged);
131      } finally {
132        ShortCircuitCache.this.lock.unlock();
133      }
134    }
135
136    @Override
137    public void close() throws IOException {
138      if (future != null) {
139        future.cancel(false);
140      }
141    }
142
143    public void setFuture(ScheduledFuture<?> future) {
144      this.future = future;
145    }
146
147    /**
148     * Get the rate at which this cleaner thread should be scheduled.
149     *
150     * We do this by taking the minimum expiration time and dividing by 4.
151     *
152     * @return the rate in milliseconds at which this thread should be
153     *         scheduled.
154     */
155    public long getRateInMs() {
156      long minLifespanMs =
157          Math.min(maxNonMmappedEvictableLifespanMs,
158              maxEvictableMmapedLifespanMs);
159      long sampleTimeMs = minLifespanMs / 4;
160      return (sampleTimeMs < 1) ? 1 : sampleTimeMs;
161    }
162  }

164  /**
165   * A task which asks the DataNode to release a short-circuit shared memory
166   * slot.  If successful, this will tell the DataNode to stop monitoring
167   * changes to the mlock status of the replica associated with the slot.
168   * It will also allow us (the client) to re-use this slot for another
169   * replica.  If we can't communicate with the DataNode for some reason,
170   * we tear down the shared memory segment to avoid being in an inconsistent
171   * state.
172   */
173  private class SlotReleaser implements Runnable {
174    /**
175     * The slot that we need to release.
176     */
177    private final Slot slot;
178
179    SlotReleaser(Slot slot) {
180      this.slot = slot;
181    }
182
183    @Override
184    public void run() {
185      LOG.trace("{}: about to release {}", ShortCircuitCache.this, slot);
186      final DfsClientShm shm = (DfsClientShm)slot.getShm();
187      final DomainSocket shmSock = shm.getPeer().getDomainSocket();
188      final String path = shmSock.getPath();
189      boolean success = false;
190      try (DomainSocket sock = DomainSocket.connect(path);
191           DataOutputStream out = new DataOutputStream(
192               new BufferedOutputStream(sock.getOutputStream()))) {
193        new Sender(out).releaseShortCircuitFds(slot.getSlotId());
194        DataInputStream in = new DataInputStream(sock.getInputStream());
195        ReleaseShortCircuitAccessResponseProto resp =
196            ReleaseShortCircuitAccessResponseProto.parseFrom(
197                PBHelperClient.vintPrefixed(in));
198        if (resp.getStatus() != Status.SUCCESS) {
199          String error = resp.hasError() ? resp.getError() : "(unknown)";
200          throw new IOException(resp.getStatus().toString() + ": " + error);
201        }
202        LOG.trace("{}: released {}", this, slot);
203        success = true;
204      } catch (IOException e) {
205        LOG.error(ShortCircuitCache.this + ": failed to release " +
206            "short-circuit shared memory slot " + slot + " by sending " +
207            "ReleaseShortCircuitAccessRequestProto to " + path +
208            ".  Closing shared memory segment.", e);
209      } finally {
210        if (success) {
211          shmManager.freeSlot(slot);
212        } else {
213          shm.getEndpointShmManager().shutdown(shm);
214        }
215      }
216    }
217  }

339  public ShortCircuitCache(int maxTotalSize, long maxNonMmappedEvictableLifespanMs,
340      int maxEvictableMmapedSize, long maxEvictableMmapedLifespanMs,
341      long mmapRetryTimeoutMs, long staleThresholdMs, int shmInterruptCheckMs) {
342    Preconditions.checkArgument(maxTotalSize >= 0);
343    this.maxTotalSize = maxTotalSize;
344    Preconditions.checkArgument(maxNonMmappedEvictableLifespanMs >= 0);
345    this.maxNonMmappedEvictableLifespanMs = maxNonMmappedEvictableLifespanMs;
346    Preconditions.checkArgument(maxEvictableMmapedSize >= 0);
347    this.maxEvictableMmapedSize = maxEvictableMmapedSize;
348    Preconditions.checkArgument(maxEvictableMmapedLifespanMs >= 0);
349    this.maxEvictableMmapedLifespanMs = maxEvictableMmapedLifespanMs;
350    this.mmapRetryTimeoutMs = mmapRetryTimeoutMs;
351    this.staleThresholdMs = staleThresholdMs;
352    DfsClientShmManager shmManager = null;
353    if ((shmInterruptCheckMs > 0) &&
354        (DomainSocketWatcher.getLoadingFailureReason() == null)) {
355      try {
356        shmManager = new DfsClientShmManager(shmInterruptCheckMs);
357      } catch (IOException e) {
358        LOG.error("failed to create ShortCircuitShmManager", e);
359      }
360    }
361    this.shmManager = shmManager;

368  /**
369   * Increment the reference count of a replica, and remove it from any free
370   * list it may be in.
371   *
372   * You must hold the cache lock while calling this function.
373   *
374   * @param replica      The replica we're removing.
375   */
376  private void ref(ShortCircuitReplica replica) {
377    lock.lock();
378    try {
379      Preconditions.checkArgument(replica.refCount > 0,
380          "can't ref %s because its refCount reached %d", replica,
381          replica.refCount);
382      Long evictableTimeNs = replica.getEvictableTimeNs();
383      replica.refCount++;
384      if (evictableTimeNs != null) {
385        String removedFrom = removeEvictable(replica);
386        if (LOG.isTraceEnabled()) {
387          LOG.trace(this + ": " + removedFrom +
388              " no longer contains " + replica + ".  refCount " +
389              (replica.refCount - 1) + " -> " + replica.refCount +
390              StringUtils.getStackTrace(Thread.currentThread()));
391
392        }
393      } else if (LOG.isTraceEnabled()) {
394        LOG.trace(this + ": replica  refCount " +
395            (replica.refCount - 1) + " -> " + replica.refCount +
396            StringUtils.getStackTrace(Thread.currentThread()));
397      }
398    } finally {
399      lock.unlock();
400    }
401  }

403  /**
404   * Unreference a replica.
405   *
406   * You must hold the cache lock while calling this function.
407   *
408   * @param replica   The replica being unreferenced.
409   */
410  void unref(ShortCircuitReplica replica) {
411    lock.lock();
412    try {
413      // If the replica is stale or unusable, but we haven't purged it yet,
414      // let's do that.  It would be a shame to evict a non-stale replica so
415      // that we could put a stale or unusable one into the cache.
416      if (!replica.purged) {
417        String purgeReason = null;
418        if (!replica.getDataStream().getChannel().isOpen()) {
419          purgeReason = "purging replica because its data channel is closed.";
420        } else if (!replica.getMetaStream().getChannel().isOpen()) {
421          purgeReason = "purging replica because its meta channel is closed.";
422        } else if (replica.isStale()) {
423          purgeReason = "purging replica because it is stale.";
424        }
425        if (purgeReason != null) {
426          LOG.debug("{}: {}", this, purgeReason);
427          purge(replica);
428        }
429      }
430      String addedString = "";
431      boolean shouldTrimEvictionMaps = false;
432      int newRefCount = --replica.refCount;
433      if (newRefCount == 0) {
434        // Close replica, since there are no remaining references to it.
435        Preconditions.checkArgument(replica.purged,
436            "Replica %s reached a refCount of 0 without being purged", replica);
437        replica.close();
438      } else if (newRefCount == 1) {
439        Preconditions.checkState(null == replica.getEvictableTimeNs(),
440            "Replica %s had a refCount higher than 1, " +
441                "but was still evictable (evictableTimeNs = %d)",
442            replica, replica.getEvictableTimeNs());
443        if (!replica.purged) {
444          // Add the replica to the end of an eviction list.
445          // Eviction lists are sorted by time.
446          if (replica.hasMmap()) {
447            insertEvictable(System.nanoTime(), replica, evictableMmapped);
448            addedString = "added to evictableMmapped, ";
449          } else {
450            insertEvictable(System.nanoTime(), replica, evictable);
451            addedString = "added to evictable, ";
452          }
453          shouldTrimEvictionMaps = true;
454        }
455      } else {
456        Preconditions.checkArgument(replica.refCount >= 0,
457            "replica's refCount went negative (refCount = %d" +
458                " for %s)", replica.refCount, replica);
459      }
460      if (LOG.isTraceEnabled()) {
461        LOG.trace(this + ": unref replica " + replica +
462            ": " + addedString + " refCount " +
463            (newRefCount + 1) + " -> " + newRefCount +
464            StringUtils.getStackTrace(Thread.currentThread()));
465      }
466      if (shouldTrimEvictionMaps) {
467        trimEvictionMaps();
468      }
469    } finally {
470      lock.unlock();
471    }
472  }

474  /**
475   * Demote old evictable mmaps into the regular eviction map.
476   *
477   * You must hold the cache lock while calling this function.
478   *
479   * @param now   Current time in monotonic milliseconds.
480   * @return      Number of replicas demoted.
481   */
482  private int demoteOldEvictableMmaped(long now) {
483    int numDemoted = 0;
484    boolean needMoreSpace = false;
485    Long evictionTimeNs = (long) 0;
486
487    while (true) {
488      Entry<Long, ShortCircuitReplica> entry =
489          evictableMmapped.ceilingEntry(evictionTimeNs);
490      if (entry == null) break;
491      evictionTimeNs = entry.getKey();
492      long evictionTimeMs =
493          TimeUnit.MILLISECONDS.convert(evictionTimeNs, TimeUnit.NANOSECONDS);
494      if (evictionTimeMs + maxEvictableMmapedLifespanMs >= now) {
495        if (evictableMmapped.size() < maxEvictableMmapedSize) {
496          break;
497        }
498        needMoreSpace = true;
499      }
500      ShortCircuitReplica replica = entry.getValue();
501      if (LOG.isTraceEnabled()) {
502        String rationale = needMoreSpace ? "because we need more space" :
503            "because it's too old";
504        LOG.trace("demoteOldEvictable: demoting " + replica + ": " +
505            rationale + ": " +
506            StringUtils.getStackTrace(Thread.currentThread()));
507      }
508      removeEvictable(replica, evictableMmapped);
509      munmap(replica);
510      insertEvictable(evictionTimeNs, replica, evictable);
511      numDemoted++;
512    }
513    return numDemoted;
514  }
515
516  /**
517   * Trim the eviction lists.
518   */
519  private void trimEvictionMaps() {
520    long now = Time.monotonicNow();
521    demoteOldEvictableMmaped(now);
522
523    while (true) {
524      long evictableSize = evictable.size();
525      long evictableMmappedSize = evictableMmapped.size();
526      if (evictableSize + evictableMmappedSize <= maxTotalSize) {
527        return;
528      }
529      ShortCircuitReplica replica;
530      if (evictableSize == 0) {
531        replica = evictableMmapped.firstEntry().getValue();
532      } else {
533        replica = evictable.firstEntry().getValue();
534      }
535      if (LOG.isTraceEnabled()) {
536        LOG.trace(this + ": trimEvictionMaps is purging " + replica +
537            StringUtils.getStackTrace(Thread.currentThread()));
538      }
539      purge(replica);
540    }
541  }

559  private String removeEvictable(ShortCircuitReplica replica) {

563    } else {

569  /**
570   * Remove a replica from an evictable map.
571   *
572   * @param replica   The replica to remove.
573   * @param map       The map to remove it from.
574   */
575  private void removeEvictable(ShortCircuitReplica replica,
576      TreeMap<Long, ShortCircuitReplica> map) {
577    Long evictableTimeNs = replica.getEvictableTimeNs();
578    Preconditions.checkNotNull(evictableTimeNs);
579    ShortCircuitReplica removed = map.remove(evictableTimeNs);
580    Preconditions.checkState(removed == replica,
581        "failed to make %s unevictable", replica);
582    replica.setEvictableTimeNs(null);
583  }

585  /**
586   * Insert a replica into an evictable map.
587   *
588   * If an element already exists with this eviction time, we add a nanosecond
589   * to it until we find an unused key.
590   *
591   * @param evictionTimeNs   The eviction time in absolute nanoseconds.
592   * @param replica          The replica to insert.
593   * @param map              The map to insert it into.
594   */
595  private void insertEvictable(Long evictionTimeNs,
596      ShortCircuitReplica replica, TreeMap<Long, ShortCircuitReplica> map) {
597    while (map.containsKey(evictionTimeNs)) {
598      evictionTimeNs++;
599    }
600    Preconditions.checkState(null == replica.getEvictableTimeNs());
601    replica.setEvictableTimeNs(evictionTimeNs);
602    map.put(evictionTimeNs, replica);
603  }

605  /**
606   * Purge a replica from the cache.
607   *
608   * This doesn't necessarily close the replica, since there may be
609   * outstanding references to it.  However, it does mean the cache won't
610   * hand it out to anyone after this.
611   *
612   * You must hold the cache lock while calling this function.
613   *
614   * @param replica   The replica being removed.
615   */
616  private void purge(ShortCircuitReplica replica) {
617    boolean removedFromInfoMap = false;
618    String evictionMapName = null;
619    Preconditions.checkArgument(!replica.purged);
620    replica.purged = true;
621    Waitable<ShortCircuitReplicaInfo> val = replicaInfoMap.get(replica.key);
622    if (val != null) {
623      ShortCircuitReplicaInfo info = val.getVal();
624      if ((info != null) && (info.getReplica() == replica)) {
625        replicaInfoMap.remove(replica.key);
626        removedFromInfoMap = true;
627      }
628    }
629    Long evictableTimeNs = replica.getEvictableTimeNs();
630    if (evictableTimeNs != null) {
631      evictionMapName = removeEvictable(replica);
632    }
633    if (LOG.isTraceEnabled()) {
634      StringBuilder builder = new StringBuilder();
635      builder.append(this).append(": ").append(": purged ").
636          append(replica).append(" from the cache.");
637      if (removedFromInfoMap) {
638        builder.append("  Removed from the replicaInfoMap.");
639      }
640      if (evictionMapName != null) {
641        builder.append("  Removed from ").append(evictionMapName);
642      }
643      LOG.trace(builder.toString());
644    }
645    unref(replica);
646  }

648  /**
649   * Fetch or create a replica.
650   *
651   * You must hold the cache lock while calling this function.
652   *
653   * @param key          Key to use for lookup.
654   * @param creator      Replica creator callback.  Will be called without
655   *                     the cache lock being held.
656   *
657   * @return             Null if no replica could be found or created.
658   *                     The replica, otherwise.
659   */
660  public ShortCircuitReplicaInfo fetchOrCreate(ExtendedBlockId key,
661      ShortCircuitReplicaCreator creator) {
662    Waitable<ShortCircuitReplicaInfo> newWaitable = null;
663    lock.lock();
664    try {
665      ShortCircuitReplicaInfo info = null;
666      do {
667        if (closed) {
668          LOG.trace("{}: can't fethchOrCreate {} because the cache is closed.",
669              this, key);
670          return null;
671        }
672        Waitable<ShortCircuitReplicaInfo> waitable = replicaInfoMap.get(key);
673        if (waitable != null) {
674          try {
675            info = fetch(key, waitable);
676          } catch (RetriableException e) {
677            LOG.debug("{}: retrying {}", this, e.getMessage());
678          }
679        }
680      } while (false);
681      if (info != null) return info;
682      // We need to load the replica ourselves.
683      newWaitable = new Waitable<>(lock.newCondition());
684      replicaInfoMap.put(key, newWaitable);
685    } finally {
686      lock.unlock();
687    }
688    return create(key, creator, newWaitable);
689  }

691  /**
692   * Fetch an existing ReplicaInfo object.
693   *
694   * @param key       The key that we're using.
695   * @param waitable  The waitable object to wait on.
696   * @return          The existing ReplicaInfo object, or null if there is
697   *                  none.
698   *
699   * @throws RetriableException   If the caller needs to retry.
700   */
701  private ShortCircuitReplicaInfo fetch(ExtendedBlockId key,
702      Waitable<ShortCircuitReplicaInfo> waitable) throws RetriableException {
703    // Another thread is already in the process of loading this
704    // ShortCircuitReplica.  So we simply wait for it to complete.
705    ShortCircuitReplicaInfo info;
706    try {
707      LOG.trace("{}: found waitable for {}", this, key);
708      info = waitable.await();
709    } catch (InterruptedException e) {
710      LOG.info(this + ": interrupted while waiting for " + key);
711      Thread.currentThread().interrupt();
712      throw new RetriableException("interrupted");
713    }
714    if (info.getInvalidTokenException() != null) {
715      LOG.info(this + ": could not get " + key + " due to InvalidToken " +
716          "exception.", info.getInvalidTokenException());
717      return info;
718    }
719    ShortCircuitReplica replica = info.getReplica();
720    if (replica == null) {
721      LOG.warn(this + ": failed to get " + key);
722      return info;
723    }
724    if (replica.purged) {
725      // Ignore replicas that have already been purged from the cache.
726      throw new RetriableException("Ignoring purged replica " +
727          replica + ".  Retrying.");
728    }
729    // Check if the replica is stale before using it.
730    // If it is, purge it and retry.
731    if (replica.isStale()) {
732      LOG.info(this + ": got stale replica " + replica + ".  Removing " +
733          "this replica from the replicaInfoMap and retrying.");
734      // Remove the cache's reference to the replica.  This may or may not
735      // trigger a close.
736      purge(replica);
737      throw new RetriableException("ignoring stale replica " + replica);
738    }
739    ref(replica);
740    return info;
741  }

743  private ShortCircuitReplicaInfo create(ExtendedBlockId key,
744      ShortCircuitReplicaCreator creator,
745      Waitable<ShortCircuitReplicaInfo> newWaitable) {
746    // Handle loading a new replica.
747    ShortCircuitReplicaInfo info = null;
748    try {
749      LOG.trace("{}: loading {}", this, key);
750      info = creator.createShortCircuitReplicaInfo();
751    } catch (RuntimeException e) {
752      LOG.warn(this + ": failed to load " + key, e);
753    }
754    if (info == null) info = new ShortCircuitReplicaInfo();
755    lock.lock();
756    try {
757      if (info.getReplica() != null) {
758        // On success, make sure the cache cleaner thread is running.
759        LOG.trace("{}: successfully loaded {}", this, info.getReplica());
760        startCacheCleanerThreadIfNeeded();
761        // Note: new ShortCircuitReplicas start with a refCount of 2,
762        // indicating that both this cache and whoever requested the
763        // creation of the replica hold a reference.  So we don't need
764        // to increment the reference count here.
765      } else {
766        // On failure, remove the waitable from the replicaInfoMap.
767        Waitable<ShortCircuitReplicaInfo> waitableInMap = replicaInfoMap.get(key);
768        if (waitableInMap == newWaitable) replicaInfoMap.remove(key);
769        if (info.getInvalidTokenException() != null) {
770          LOG.info(this + ": could not load " + key + " due to InvalidToken " +
771              "exception.", info.getInvalidTokenException());
772        } else {
773          LOG.warn(this + ": failed to load " + key);
774        }
775      }
776      newWaitable.provide(info);
777    } finally {
778      lock.unlock();
779    }
780    return info;
781  }
782
783  private void startCacheCleanerThreadIfNeeded() {
784    if (cacheCleaner == null) {
785      cacheCleaner = new CacheCleaner();
786      long rateMs = cacheCleaner.getRateInMs();
787      ScheduledFuture<?> future =
788          cleanerExecutor.scheduleAtFixedRate(cacheCleaner, rateMs, rateMs,
789              TimeUnit.MILLISECONDS);
790      cacheCleaner.setFuture(future);
791      LOG.debug("{}: starting cache cleaner thread which will run every {} ms",
792          this, rateMs);
793    }
794  }

796  ClientMmap getOrCreateClientMmap(ShortCircuitReplica replica,
797      boolean anchored) {
798    Condition newCond;
799    lock.lock();
800    try {
801      while (replica.mmapData != null) {
802        if (replica.mmapData instanceof MappedByteBuffer) {
803          ref(replica);
804          MappedByteBuffer mmap = (MappedByteBuffer)replica.mmapData;
805          return new ClientMmap(replica, mmap, anchored);
806        } else if (replica.mmapData instanceof Long) {
807          long lastAttemptTimeMs = (Long)replica.mmapData;
808          long delta = Time.monotonicNow() - lastAttemptTimeMs;
809          if (delta < mmapRetryTimeoutMs) {
810            LOG.trace("{}: can't create client mmap for {} because we failed to"
811                + " create one just {}ms ago.", this, replica, delta);
812            return null;
813          }
814          LOG.trace("{}: retrying client mmap for {}, {} ms after the previous "
815              + "failure.", this, replica, delta);
816        } else if (replica.mmapData instanceof Condition) {
817          Condition cond = (Condition)replica.mmapData;
818          cond.awaitUninterruptibly();
819        } else {
820          Preconditions.checkState(false, "invalid mmapData type %s",
821              replica.mmapData.getClass().getName());
822        }
823      }
824      newCond = lock.newCondition();
825      replica.mmapData = newCond;
826    } finally {
827      lock.unlock();
828    }
829    MappedByteBuffer map = replica.loadMmapInternal();
830    lock.lock();
831    try {
832      if (map == null) {
833        replica.mmapData = Time.monotonicNow();
834        newCond.signalAll();
835        return null;
836      } else {
837        outstandingMmapCount++;
838        replica.mmapData = map;
839        ref(replica);
840        newCond.signalAll();
841        return new ClientMmap(replica, map, anchored);
842      }
843    } finally {
844      lock.unlock();
845    }
846  }

848  /**
849   * Close the cache and free all associated resources.
850   */
851  @Override
852  public void close() {
853    try {
854      lock.lock();
855      if (closed) return;
856      closed = true;
857      LOG.info(this + ": closing");
858      maxNonMmappedEvictableLifespanMs = 0;
859      maxEvictableMmapedSize = 0;
860      // Close and join cacheCleaner thread.
861      IOUtilsClient.cleanup(LOG, cacheCleaner);
862      // Purge all replicas.
863      while (true) {
864        Entry<Long, ShortCircuitReplica> entry = evictable.firstEntry();
865        if (entry == null) break;
866        purge(entry.getValue());
867      }
868      while (true) {
869        Entry<Long, ShortCircuitReplica> entry = evictableMmapped.firstEntry();
870        if (entry == null) break;
871        purge(entry.getValue());
872      }
873    } finally {
874      lock.unlock();
875    }
876
877    releaserExecutor.shutdown();
878    cleanerExecutor.shutdown();
879    // wait for existing tasks to terminate
880    try {
881      if (!releaserExecutor.awaitTermination(30, TimeUnit.SECONDS)) {
882        LOG.error("Forcing SlotReleaserThreadPool to shutdown!");
883        releaserExecutor.shutdownNow();
884      }
885    } catch (InterruptedException e) {
886      releaserExecutor.shutdownNow();
887      Thread.currentThread().interrupt();
888      LOG.error("Interrupted while waiting for SlotReleaserThreadPool "
889          + "to terminate", e);
890    }
891
892    // wait for existing tasks to terminate
893    try {
894      if (!cleanerExecutor.awaitTermination(30, TimeUnit.SECONDS)) {
895        LOG.error("Forcing CleanerThreadPool to shutdown!");
896        cleanerExecutor.shutdownNow();
897      }
898    } catch (InterruptedException e) {
899      cleanerExecutor.shutdownNow();
900      Thread.currentThread().interrupt();
901      LOG.error("Interrupted while waiting for CleanerThreadPool "
902          + "to terminate", e);
903    }
904    IOUtilsClient.cleanup(LOG, shmManager);
905  }

908  public interface CacheVisitor {
909    void visit(int numOutstandingMmaps,
910        Map<ExtendedBlockId, ShortCircuitReplica> replicas,
911        Map<ExtendedBlockId, InvalidToken> failedLoads,
912        Map<Long, ShortCircuitReplica> evictable,
913        Map<Long, ShortCircuitReplica> evictableMmapped);
914  }
915
916  @VisibleForTesting // ONLY for testing
917  public void accept(CacheVisitor visitor) {
918    lock.lock();
919    try {
920      Map<ExtendedBlockId, ShortCircuitReplica> replicas = new HashMap<>();
921      Map<ExtendedBlockId, InvalidToken> failedLoads = new HashMap<>();
922      for (Entry<ExtendedBlockId, Waitable<ShortCircuitReplicaInfo>> entry :
923          replicaInfoMap.entrySet()) {
924        Waitable<ShortCircuitReplicaInfo> waitable = entry.getValue();
925        if (waitable.hasVal()) {
926          if (waitable.getVal().getReplica() != null) {
927            replicas.put(entry.getKey(), waitable.getVal().getReplica());
928          } else {
929            // The exception may be null here, indicating a failed load that
930            // isn't the result of an invalid block token.
931            failedLoads.put(entry.getKey(),
932                waitable.getVal().getInvalidTokenException());
933          }
934        }
935      }
936      LOG.debug("visiting {} with outstandingMmapCount={}, replicas={}, "
937              + "failedLoads={}, evictable={}, evictableMmapped={}",
938          visitor.getClass().getName(), outstandingMmapCount, replicas,
939          failedLoads, evictable, evictableMmapped);
940      visitor.visit(outstandingMmapCount, replicas, failedLoads,
941          evictable, evictableMmapped);
942    } finally {
943      lock.unlock();
944    }
945  }
