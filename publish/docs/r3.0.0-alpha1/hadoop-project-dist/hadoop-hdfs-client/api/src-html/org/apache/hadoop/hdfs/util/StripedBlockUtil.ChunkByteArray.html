049/**
050 * When accessing a file in striped layout, operations on logical byte ranges
051 * in the file need to be mapped to physical byte ranges on block files stored
052 * on DataNodes. This utility class facilities this mapping by defining and
053 * exposing a number of striping-related concepts. The most basic ones are
054 * illustrated in the following diagram. Unless otherwise specified, all
055 * range-related calculations are inclusive (the end offset of the previous
056 * range should be 1 byte lower than the start offset of the next one).
057 *
058 *  | <----  Block Group ----> |   <- Block Group: logical unit composing
059 *  |                          |        striped HDFS files.
060 *  blk_0      blk_1       blk_2   <- Internal Blocks: each internal block
061 *    |          |           |          represents a physically stored local
062 *    v          v           v          block file
063 * +------+   +------+   +------+
064 * |cell_0|   |cell_1|   |cell_2|  <- {@link StripingCell} represents the
065 * +------+   +------+   +------+       logical order that a Block Group should
066 * |cell_3|   |cell_4|   |cell_5|       be accessed: cell_0, cell_1, ...
067 * +------+   +------+   +------+
068 * |cell_6|   |cell_7|   |cell_8|
069 * +------+   +------+   +------+
070 * |cell_9|
071 * +------+  <- A cell contains cellSize bytes of data
072 */
073@InterfaceAudience.Private
074public class StripedBlockUtil {

078  /**
079   * Parses a striped block group into individual blocks.
080   * @param bg The striped block group
081   * @param ecPolicy The erasure coding policy
082   * @return An array of the blocks in the group

084  public static LocatedBlock[] parseStripedBlockGroup(LocatedStripedBlock bg,
085                                               ErasureCodingPolicy ecPolicy) {
086    return parseStripedBlockGroup(bg, ecPolicy.getCellSize(),
087        ecPolicy.getNumDataUnits(), ecPolicy.getNumParityUnits());
088  }

090  /**
091   * This method parses a striped block group into individual blocks.
092   *
093   * @param bg The striped block group
094   * @param cellSize The size of a striping cell
095   * @param dataBlkNum The number of data blocks
096   * @return An array containing the blocks in the group
097   */
098  public static LocatedBlock[] parseStripedBlockGroup(LocatedStripedBlock bg,
099      int cellSize, int dataBlkNum, int parityBlkNum) {
100    int locatedBGSize = bg.getBlockIndices().length;
101    LocatedBlock[] lbs = new LocatedBlock[dataBlkNum + parityBlkNum];
102    for (short i = 0; i < locatedBGSize; i++) {
103      final int idx = bg.getBlockIndices()[i];
104      // for now we do not use redundant replica of an internal block
105      if (idx < (dataBlkNum + parityBlkNum) && lbs[idx] == null) {
106        lbs[idx] = constructInternalBlock(bg, i, cellSize,
107            dataBlkNum, idx);
108      }
109    }
110    return lbs;
111  }

113  /**
114   * This method creates an internal block at the given index of a block group
115   *
116   * @param idxInReturnedLocs The index in the stored locations in the
117   *                          {@link LocatedStripedBlock} object
118   * @param idxInBlockGroup The logical index in the striped block group
119   * @return The constructed internal block
120   */
121  public static LocatedBlock constructInternalBlock(LocatedStripedBlock bg,
122      int idxInReturnedLocs, int cellSize, int dataBlkNum,
123      int idxInBlockGroup) {
124    final ExtendedBlock blk = constructInternalBlock(
125        bg.getBlock(), cellSize, dataBlkNum, idxInBlockGroup);
126    final LocatedBlock locatedBlock;
127    if (idxInReturnedLocs < bg.getLocations().length) {
128      locatedBlock = new LocatedBlock(blk,
129          new DatanodeInfo[]{bg.getLocations()[idxInReturnedLocs]},
130          new String[]{bg.getStorageIDs()[idxInReturnedLocs]},
131          new StorageType[]{bg.getStorageTypes()[idxInReturnedLocs]},
132          bg.getStartOffset(), bg.isCorrupt(), null);
133    } else {
134      locatedBlock = new LocatedBlock(blk, null, null, null,
135          bg.getStartOffset(), bg.isCorrupt(), null);
136    }
137    Token<BlockTokenIdentifier>[] blockTokens = bg.getBlockTokens();
138    if (idxInReturnedLocs < blockTokens.length) {
139      locatedBlock.setBlockToken(blockTokens[idxInReturnedLocs]);
140    }
141    return locatedBlock;
142  }

150  /**
151   * This method creates an internal {@link ExtendedBlock} at the given index
152   * of a block group.
153   */
154  public static ExtendedBlock constructInternalBlock(ExtendedBlock blockGroup,
155      int cellSize, int dataBlkNum, int idxInBlockGroup) {
156    ExtendedBlock block = new ExtendedBlock(blockGroup);
157    block.setBlockId(blockGroup.getBlockId() + idxInBlockGroup);
158    block.setNumBytes(getInternalBlockLength(blockGroup.getNumBytes(),
159        cellSize, dataBlkNum, idxInBlockGroup));
160    return block;
161  }

170  /**
171   * Get the size of an internal block at the given index of a block group
172   *
173   * @param dataSize Size of the block group only counting data blocks
174   * @param cellSize The size of a striping cell
175   * @param numDataBlocks The number of data blocks
176   * @param idxInBlockGroup The logical index in the striped block group
177   * @return The size of the internal block at the specified index
178   */
179  public static long getInternalBlockLength(long dataSize,
180      int cellSize, int numDataBlocks, int idxInBlockGroup) {
181    Preconditions.checkArgument(dataSize >= 0);
182    Preconditions.checkArgument(cellSize > 0);
183    Preconditions.checkArgument(numDataBlocks > 0);
184    Preconditions.checkArgument(idxInBlockGroup >= 0);
185    // Size of each stripe (only counting data blocks)
186    final int stripeSize = cellSize * numDataBlocks;
187    // If block group ends at stripe boundary, each internal block has an equal
188    // share of the group
189    final int lastStripeDataLen = (int)(dataSize % stripeSize);
190    if (lastStripeDataLen == 0) {
191      return dataSize / numDataBlocks;
192    }
193
194    final int numStripes = (int) ((dataSize - 1) / stripeSize + 1);
195    return (numStripes - 1L)*cellSize
196        + lastCellSize(lastStripeDataLen, cellSize,
197        numDataBlocks, idxInBlockGroup);
198  }

200  /**
201   * Compute the safe length given the internal block lengths.
202   *
203   * @param ecPolicy The EC policy used for the block group
204   * @param blockLens The lengths of internal blocks
205   * @return The safe length
206   */
207  public static long getSafeLength(ErasureCodingPolicy ecPolicy,
208      long[] blockLens) {
209    final int cellSize = ecPolicy.getCellSize();
210    final int dataBlkNum = ecPolicy.getNumDataUnits();
211    Preconditions.checkArgument(blockLens.length >= dataBlkNum);
212    final int stripeSize = dataBlkNum * cellSize;
213    long[] cpy = Arrays.copyOf(blockLens, blockLens.length);
214    Arrays.sort(cpy);
215    // full stripe is a stripe has at least dataBlkNum full cells.
216    // lastFullStripeIdx is the index of the last full stripe.
217    int lastFullStripeIdx =
218        (int) (cpy[cpy.length - dataBlkNum] / cellSize);
219    return lastFullStripeIdx * stripeSize; // return the safeLength
220    // TODO: Include lastFullStripeIdx+1 stripe in safeLength, if there exists
221    // such a stripe (and it must be partial).
222  }
223
224  private static int lastCellSize(int size, int cellSize, int numDataBlocks,
225      int i) {
226    if (i < numDataBlocks) {
227      // parity block size (i.e. i >= numDataBlocks) is the same as
228      // the first data block size (i.e. i = 0).
229      size -= i*cellSize;
230      if (size < 0) {
231        size = 0;
232      }
233    }
234    return size > cellSize? cellSize: size;
235  }

237  /**
238   * Given a byte's offset in an internal block, calculate the offset in
239   * the block group
240   */
241  public static long offsetInBlkToOffsetInBG(int cellSize, int dataBlkNum,
242      long offsetInBlk, int idxInBlockGroup) {
243    int cellIdxInBlk = (int) (offsetInBlk / cellSize);
244    return cellIdxInBlk * cellSize * dataBlkNum // n full stripes before offset
245        + idxInBlockGroup * cellSize // m full cells before offset
246        + offsetInBlk % cellSize; // partial cell
247  }

249  /**
250   * Get the next completed striped read task
251   *
252   * @return {@link StripingChunkReadResult} indicating the status of the read task
253   *          succeeded, and the block index of the task. If the method times
254   *          out without getting any completed read tasks, -1 is returned as
255   *          block index.
256   * @throws InterruptedException
257   */
258  public static StripingChunkReadResult getNextCompletedStripedRead(
259      CompletionService<Void> readService, Map<Future<Void>, Integer> futures,
260      final long timeoutMillis) throws InterruptedException {
261    Preconditions.checkArgument(!futures.isEmpty());
262    Future<Void> future = null;
263    try {
264      if (timeoutMillis > 0) {
265        future = readService.poll(timeoutMillis, TimeUnit.MILLISECONDS);
266      } else {
267        future = readService.take();
268      }
269      if (future != null) {
270        future.get();
271        return new StripingChunkReadResult(futures.remove(future),
272            StripingChunkReadResult.SUCCESSFUL);
273      } else {
274        return new StripingChunkReadResult(StripingChunkReadResult.TIMEOUT);
275      }
276    } catch (ExecutionException e) {
277      LOG.debug("Exception during striped read task", e);
278      return new StripingChunkReadResult(futures.remove(future),
279          StripingChunkReadResult.FAILED);
280    } catch (CancellationException e) {
281      LOG.debug("Exception during striped read task", e);
282      return new StripingChunkReadResult(futures.remove(future),
283          StripingChunkReadResult.CANCELLED);
284    }
285  }

287  /**
288   * Get the total usage of the striped blocks, which is the total of data
289   * blocks and parity blocks
290   *
291   * @param numDataBlkBytes
292   *          Size of the block group only counting data blocks
293   * @param dataBlkNum
294   *          The number of data blocks
295   * @param parityBlkNum
296   *          The number of parity blocks
297   * @param cellSize
298   *          The size of a striping cell
299   * @return The total usage of data blocks and parity blocks
300   */
301  public static long spaceConsumedByStripedBlock(long numDataBlkBytes,
302      int dataBlkNum, int parityBlkNum, int cellSize) {
303    int parityIndex = dataBlkNum + 1;
304    long numParityBlkBytes = getInternalBlockLength(numDataBlkBytes, cellSize,
305        dataBlkNum, parityIndex) * parityBlkNum;
306    return numDataBlkBytes + numParityBlkBytes;
307  }

309  /**
310   * Initialize the decoding input buffers based on the chunk states in an
311   * {@link AlignedStripe}. For each chunk that was not initially requested,
312   * schedule a new fetch request with the decoding input buffer as transfer
313   * destination.
314   */
315  public static byte[][] initDecodeInputs(AlignedStripe alignedStripe,
316      int dataBlkNum, int parityBlkNum) {
317    byte[][] decodeInputs =
318        new byte[dataBlkNum + parityBlkNum][(int) alignedStripe.getSpanInBlock()];
319    // read the full data aligned stripe
320    for (int i = 0; i < dataBlkNum; i++) {
321      if (alignedStripe.chunks[i] == null) {
322        alignedStripe.chunks[i] = new StripingChunk(decodeInputs[i]);
323        alignedStripe.chunks[i].addByteArraySlice(0,
324            (int) alignedStripe.getSpanInBlock());
325      }
326    }
327    return decodeInputs;
328  }

330  /**
331   * Some fetched {@link StripingChunk} might be stored in original application
332   * buffer instead of prepared decode input buffers. Some others are beyond
333   * the range of the internal blocks and should correspond to all zero bytes.
334   * When all pending requests have returned, this method should be called to
335   * finalize decode input buffers.
336   */
337  public static void finalizeDecodeInputs(final byte[][] decodeInputs,
338                                          AlignedStripe alignedStripe) {
339    for (int i = 0; i < alignedStripe.chunks.length; i++) {
340      final StripingChunk chunk = alignedStripe.chunks[i];
341      if (chunk != null && chunk.state == StripingChunk.FETCHED) {
342        chunk.copyTo(decodeInputs[i]);
343      } else if (chunk != null && chunk.state == StripingChunk.ALLZERO) {
344        Arrays.fill(decodeInputs[i], (byte) 0);
345      } else {
346        decodeInputs[i] = null;
347      }
348    }
349  }

351  /**
352   * Decode based on the given input buffers and erasure coding policy.
353   */
354  public static void decodeAndFillBuffer(final byte[][] decodeInputs,
355      AlignedStripe alignedStripe, int dataBlkNum, int parityBlkNum,
356      RawErasureDecoder decoder) {
357    // Step 1: prepare indices and output buffers for missing data units
358    int[] decodeIndices = new int[parityBlkNum];
359    int pos = 0;
360    for (int i = 0; i < dataBlkNum; i++) {
361      if (alignedStripe.chunks[i] != null &&
362          alignedStripe.chunks[i].state == StripingChunk.MISSING){
363        decodeIndices[pos++] = i;
364      }
365    }
366    decodeIndices = Arrays.copyOf(decodeIndices, pos);
367    byte[][] decodeOutputs =
368        new byte[decodeIndices.length][(int) alignedStripe.getSpanInBlock()];
369
370    // Step 2: decode into prepared output buffers
371    decoder.decode(decodeInputs, decodeIndices, decodeOutputs);
372
373    // Step 3: fill original application buffer with decoded data
374    for (int i = 0; i < decodeIndices.length; i++) {
375      int missingBlkIdx = decodeIndices[i];
376      StripingChunk chunk = alignedStripe.chunks[missingBlkIdx];
377      if (chunk.state == StripingChunk.MISSING) {
378        chunk.copyFrom(decodeOutputs[i]);
379      }
380    }
381  }

383  /**
384   * Similar functionality with {@link #divideByteRangeIntoStripes}, but is used
385   * by stateful read and uses ByteBuffer as reading target buffer. Besides the
386   * read range is within a single stripe thus the calculation logic is simpler.
387   */
388  public static AlignedStripe[] divideOneStripe(ErasureCodingPolicy ecPolicy,
389      int cellSize, LocatedStripedBlock blockGroup, long rangeStartInBlockGroup,
390      long rangeEndInBlockGroup, ByteBuffer buf) {
391    final int dataBlkNum = ecPolicy.getNumDataUnits();
392    // Step 1: map the byte range to StripingCells
393    StripingCell[] cells = getStripingCellsOfByteRange(ecPolicy, cellSize,
394        blockGroup, rangeStartInBlockGroup, rangeEndInBlockGroup);
395
396    // Step 2: get the unmerged ranges on each internal block
397    VerticalRange[] ranges = getRangesForInternalBlocks(ecPolicy, cellSize,
398        cells);
399
400    // Step 3: merge into stripes
401    AlignedStripe[] stripes = mergeRangesForInternalBlocks(ecPolicy, ranges);
402
403    // Step 4: calculate each chunk's position in destination buffer. Since the
404    // whole read range is within a single stripe, the logic is simpler here.
405    int bufOffset = (int) (rangeStartInBlockGroup % ((long) cellSize * dataBlkNum));
406    for (StripingCell cell : cells) {
407      long cellStart = cell.idxInInternalBlk * cellSize + cell.offset;
408      long cellEnd = cellStart + cell.size - 1;
409      for (AlignedStripe s : stripes) {
410        long stripeEnd = s.getOffsetInBlock() + s.getSpanInBlock() - 1;
411        long overlapStart = Math.max(cellStart, s.getOffsetInBlock());
412        long overlapEnd = Math.min(cellEnd, stripeEnd);
413        int overLapLen = (int) (overlapEnd - overlapStart + 1);
414        if (overLapLen > 0) {
415          Preconditions.checkState(s.chunks[cell.idxInStripe] == null);
416          final int pos = (int) (bufOffset + overlapStart - cellStart);
417          buf.position(pos);
418          buf.limit(pos + overLapLen);
419          s.chunks[cell.idxInStripe] = new StripingChunk(buf.slice());
420        }
421      }
422      bufOffset += cell.size;
423    }
424
425    // Step 5: prepare ALLZERO blocks
426    prepareAllZeroChunks(blockGroup, stripes, cellSize, dataBlkNum);
427    return stripes;
428  }

430  /**
431   * This method divides a requested byte range into an array of inclusive
432   * {@link AlignedStripe}.
433   * @param ecPolicy The codec policy for the file, which carries the numbers
434   *                 of data / parity blocks
435   * @param cellSize Cell size of stripe
436   * @param blockGroup The striped block group
437   * @param rangeStartInBlockGroup The byte range's start offset in block group
438   * @param rangeEndInBlockGroup The byte range's end offset in block group
439   * @param buf Destination buffer of the read operation for the byte range
440   * @param offsetInBuf Start offset into the destination buffer
441   *
442   * At most 5 stripes will be generated from each logical range, as
443   * demonstrated in the header of {@link AlignedStripe}.
444   */
445  public static AlignedStripe[] divideByteRangeIntoStripes(ErasureCodingPolicy ecPolicy,
446      int cellSize, LocatedStripedBlock blockGroup,
447      long rangeStartInBlockGroup, long rangeEndInBlockGroup, byte[] buf,
448      int offsetInBuf) {
449
450    // Step 0: analyze range and calculate basic parameters
451    final int dataBlkNum = ecPolicy.getNumDataUnits();
452
453    // Step 1: map the byte range to StripingCells
454    StripingCell[] cells = getStripingCellsOfByteRange(ecPolicy, cellSize,
455        blockGroup, rangeStartInBlockGroup, rangeEndInBlockGroup);
456
457    // Step 2: get the unmerged ranges on each internal block
458    VerticalRange[] ranges = getRangesForInternalBlocks(ecPolicy, cellSize,
459        cells);
460
461    // Step 3: merge into at most 5 stripes
462    AlignedStripe[] stripes = mergeRangesForInternalBlocks(ecPolicy, ranges);
463
464    // Step 4: calculate each chunk's position in destination buffer
465    calcualteChunkPositionsInBuf(cellSize, stripes, cells, buf, offsetInBuf);
466
467    // Step 5: prepare ALLZERO blocks
468    prepareAllZeroChunks(blockGroup, stripes, cellSize, dataBlkNum);
469
470    return stripes;
471  }

473  /**
474   * Map the logical byte range to a set of inclusive {@link StripingCell}
475   * instances, each representing the overlap of the byte range to a cell
476   * used by {@link DFSStripedOutputStream} in encoding
477   */
478  @VisibleForTesting
479  private static StripingCell[] getStripingCellsOfByteRange(ErasureCodingPolicy ecPolicy,
480      int cellSize, LocatedStripedBlock blockGroup,
481      long rangeStartInBlockGroup, long rangeEndInBlockGroup) {
482    Preconditions.checkArgument(
483        rangeStartInBlockGroup <= rangeEndInBlockGroup &&
484            rangeEndInBlockGroup < blockGroup.getBlockSize());
485    long len = rangeEndInBlockGroup - rangeStartInBlockGroup + 1;
486    int firstCellIdxInBG = (int) (rangeStartInBlockGroup / cellSize);
487    int lastCellIdxInBG = (int) (rangeEndInBlockGroup / cellSize);
488    int numCells = lastCellIdxInBG - firstCellIdxInBG + 1;
489    StripingCell[] cells = new StripingCell[numCells];
490
491    final int firstCellOffset = (int) (rangeStartInBlockGroup % cellSize);
492    final int firstCellSize =
493        (int) Math.min(cellSize - (rangeStartInBlockGroup % cellSize), len);
494    cells[0] = new StripingCell(ecPolicy, firstCellSize, firstCellIdxInBG,
495        firstCellOffset);
496    if (lastCellIdxInBG != firstCellIdxInBG) {
497      final int lastCellSize = (int) (rangeEndInBlockGroup % cellSize) + 1;
498      cells[numCells - 1] = new StripingCell(ecPolicy, lastCellSize,
499          lastCellIdxInBG, 0);
500    }
501
502    for (int i = 1; i < numCells - 1; i++) {
503      cells[i] = new StripingCell(ecPolicy, cellSize, i + firstCellIdxInBG, 0);
504    }
505
506    return cells;
507  }

509  /**
510   * Given a logical byte range, mapped to each {@link StripingCell}, calculate
511   * the physical byte range (inclusive) on each stored internal block.
512   */
513  @VisibleForTesting
514  private static VerticalRange[] getRangesForInternalBlocks(ErasureCodingPolicy ecPolicy,
515      int cellSize, StripingCell[] cells) {
516    int dataBlkNum = ecPolicy.getNumDataUnits();
517    int parityBlkNum = ecPolicy.getNumParityUnits();
518
519    VerticalRange ranges[] = new VerticalRange[dataBlkNum + parityBlkNum];
520
521    long earliestStart = Long.MAX_VALUE;
522    long latestEnd = -1;
523    for (StripingCell cell : cells) {
524      // iterate through all cells and update the list of StripeRanges
525      if (ranges[cell.idxInStripe] == null) {
526        ranges[cell.idxInStripe] = new VerticalRange(
527            cell.idxInInternalBlk * cellSize + cell.offset, cell.size);
528      } else {
529        ranges[cell.idxInStripe].spanInBlock += cell.size;
530      }
531      VerticalRange range = ranges[cell.idxInStripe];
532      if (range.offsetInBlock < earliestStart) {
533        earliestStart = range.offsetInBlock;
534      }
535      if (range.offsetInBlock + range.spanInBlock - 1 > latestEnd) {
536        latestEnd = range.offsetInBlock + range.spanInBlock - 1;
537      }
538    }
539
540    // Each parity block should be fetched at maximum range of all data blocks
541    for (int i = dataBlkNum; i < dataBlkNum + parityBlkNum; i++) {
542      ranges[i] = new VerticalRange(earliestStart,
543          latestEnd - earliestStart + 1);
544    }
545
546    return ranges;
547  }

549  /**
550   * Merge byte ranges on each internal block into a set of inclusive
551   * {@link AlignedStripe} instances.
552   */
553  private static AlignedStripe[] mergeRangesForInternalBlocks(
554      ErasureCodingPolicy ecPolicy, VerticalRange[] ranges) {
555    int dataBlkNum = ecPolicy.getNumDataUnits();
556    int parityBlkNum = ecPolicy.getNumParityUnits();
557    List<AlignedStripe> stripes = new ArrayList<>();
558    SortedSet<Long> stripePoints = new TreeSet<>();
559    for (VerticalRange r : ranges) {
560      if (r != null) {
561        stripePoints.add(r.offsetInBlock);
562        stripePoints.add(r.offsetInBlock + r.spanInBlock);
563      }
564    }
565
566    long prev = -1;
567    for (long point : stripePoints) {
568      if (prev >= 0) {
569        stripes.add(new AlignedStripe(prev, point - prev,
570            dataBlkNum + parityBlkNum));
571      }
572      prev = point;
573    }
574    return stripes.toArray(new AlignedStripe[stripes.size()]);
575  }

577  private static void calcualteChunkPositionsInBuf(int cellSize,
578      AlignedStripe[] stripes, StripingCell[] cells, byte[] buf,
579      int offsetInBuf) {
580    /**
581     *     | <--------------- AlignedStripe --------------->|
582     *
583     *     |<- length_0 ->|<--  length_1  -->|<- length_2 ->|
584     * +------------------+------------------+----------------+
585     * |    cell_0_0_0    |    cell_3_1_0    |   cell_6_2_0   |  <- blk_0
586     * +------------------+------------------+----------------+
587     *   _/                \_______________________
588     *  |                                          |
589     *  v offset_0                                 v offset_1
590     * +----------------------------------------------------------+
591     * |  cell_0_0_0 |  cell_1_0_1 and cell_2_0_2  |cell_3_1_0 ...|   <- buf
592     * |  (partial)  |    (from blk_1 and blk_2)   |              |
593     * +----------------------------------------------------------+
594     *
595     * Cell indexing convention defined in {@link StripingCell}
596     */
597    int done = 0;
598    for (StripingCell cell : cells) {
599      long cellStart = cell.idxInInternalBlk * cellSize + cell.offset;
600      long cellEnd = cellStart + cell.size - 1;
601      for (AlignedStripe s : stripes) {
602        long stripeEnd = s.getOffsetInBlock() + s.getSpanInBlock() - 1;
603        long overlapStart = Math.max(cellStart, s.getOffsetInBlock());
604        long overlapEnd = Math.min(cellEnd, stripeEnd);
605        int overLapLen = (int) (overlapEnd - overlapStart + 1);
606        if (overLapLen <= 0) {
607          continue;
608        }
609        if (s.chunks[cell.idxInStripe] == null) {
610          s.chunks[cell.idxInStripe] = new StripingChunk(buf);
611        }
612        s.chunks[cell.idxInStripe].addByteArraySlice(
613            (int)(offsetInBuf + done + overlapStart - cellStart), overLapLen);
614      }
615      done += cell.size;
616    }
617  }

619  /**
620   * If a {@link StripingChunk} maps to a byte range beyond an internal block's
621   * size, the chunk should be treated as zero bytes in decoding.
622   */
623  private static void prepareAllZeroChunks(LocatedStripedBlock blockGroup,
624      AlignedStripe[] stripes, int cellSize, int dataBlkNum) {
625    for (AlignedStripe s : stripes) {
626      for (int i = 0; i < dataBlkNum; i++) {
627        long internalBlkLen = getInternalBlockLength(blockGroup.getBlockSize(),
628            cellSize, dataBlkNum, i);
629        if (internalBlkLen <= s.getOffsetInBlock()) {
630          Preconditions.checkState(s.chunks[i] == null);
631          s.chunks[i] = new StripingChunk(StripingChunk.ALLZERO);
632        }
633      }
634    }
635  }

637  /**
638   * Cell is the unit of encoding used in {@link DFSStripedOutputStream}. This
639   * size impacts how a logical offset in the file or block group translates
640   * to physical byte offset in a stored internal block. The StripingCell util
641   * class facilitates this calculation. Each StripingCell is inclusive with
642   * its start and end offsets -- e.g., the end logical offset of cell_0_0_0
643   * should be 1 byte lower than the start logical offset of cell_1_0_1.
644   *
645   *  | <------- Striped Block Group -------> |
646   *    blk_0          blk_1          blk_2
647   *      |              |              |
648   *      v              v              v
649   * +----------+   +----------+   +----------+
650   * |cell_0_0_0|   |cell_1_0_1|   |cell_2_0_2|
651   * +----------+   +----------+   +----------+
652   * |cell_3_1_0|   |cell_4_1_1|   |cell_5_1_2| <- {@link #idxInBlkGroup} = 5
653   * +----------+   +----------+   +----------+    {@link #idxInInternalBlk} = 1
654   *                                               {@link #idxInStripe} = 2
655   * A StripingCell is a special instance of {@link StripingChunk} whose offset
656   * and size align with the cell used when writing data.
657   * TODO: consider parity cells
658   */
659  @VisibleForTesting
660  static class StripingCell {
661    final ErasureCodingPolicy ecPolicy;
662    /** Logical order in a block group, used when doing I/O to a block group */
663    final int idxInBlkGroup;
664    final int idxInInternalBlk;
665    final int idxInStripe;
666    /**
667     * When a logical byte range is mapped to a set of cells, it might
668     * partially overlap with the first and last cells. This field and the
669     * {@link #size} variable represent the start offset and size of the
670     * overlap.
671     */
672    final int offset;
673    final int size;
674
675    StripingCell(ErasureCodingPolicy ecPolicy, int cellSize, int idxInBlkGroup,
676        int offset) {
677      this.ecPolicy = ecPolicy;
678      this.idxInBlkGroup = idxInBlkGroup;
679      this.idxInInternalBlk = idxInBlkGroup / ecPolicy.getNumDataUnits();
680      this.idxInStripe = idxInBlkGroup -
681          this.idxInInternalBlk * ecPolicy.getNumDataUnits();
682      this.offset = offset;
683      this.size = cellSize;
684    }
685  }

687  /**
688   * Given a requested byte range on a striped block group, an AlignedStripe
689   * represents an inclusive {@link VerticalRange} that is aligned with both
690   * the byte range and boundaries of all internal blocks. As illustrated in
691   * the diagram, any given byte range on a block group leads to 1~5
692   * AlignedStripe's.
693   *
694   * |<-------- Striped Block Group -------->|
695   * blk_0   blk_1   blk_2      blk_3   blk_4
696   *                 +----+  |  +----+  +----+
697   *                 |full|  |  |    |  |    | <- AlignedStripe0:
698   *         +----+  |~~~~|  |  |~~~~|  |~~~~|      1st cell is partial
699   *         |part|  |    |  |  |    |  |    | <- AlignedStripe1: byte range
700   * +----+  +----+  +----+  |  |~~~~|  |~~~~|      doesn't start at 1st block
701   * |full|  |full|  |full|  |  |    |  |    |
702   * |cell|  |cell|  |cell|  |  |    |  |    | <- AlignedStripe2 (full stripe)
703   * |    |  |    |  |    |  |  |    |  |    |
704   * +----+  +----+  +----+  |  |~~~~|  |~~~~|
705   * |full|  |part|          |  |    |  |    | <- AlignedStripe3: byte range
706   * |~~~~|  +----+          |  |~~~~|  |~~~~|      doesn't end at last block
707   * |    |                  |  |    |  |    | <- AlignedStripe4:
708   * +----+                  |  +----+  +----+      last cell is partial
709   *                         |
710   * <---- data blocks ----> | <--- parity --->
711   *
712   * An AlignedStripe is the basic unit of reading from a striped block group,
713   * because within the AlignedStripe, all internal blocks can be processed in
714   * a uniform manner.
715   *
716   * The coverage of an AlignedStripe on an internal block is represented as a
717   * {@link StripingChunk}.
718   *
719   * To simplify the logic of reading a logical byte range from a block group,
720   * a StripingChunk is either completely in the requested byte range or
721   * completely outside the requested byte range.
722   */
723  public static class AlignedStripe {
724    public VerticalRange range;
725    /** status of each chunk in the stripe */
726    public final StripingChunk[] chunks;
727    public int fetchedChunksNum = 0;
728    public int missingChunksNum = 0;
729
730    public AlignedStripe(long offsetInBlock, long length, int width) {
731      Preconditions.checkArgument(offsetInBlock >= 0 && length >= 0);
732      this.range = new VerticalRange(offsetInBlock, length);
733      this.chunks = new StripingChunk[width];
734    }
735
736    public boolean include(long pos) {
737      return range.include(pos);
738    }
739
740    public long getOffsetInBlock() {
741      return range.offsetInBlock;
742    }
743
744    public long getSpanInBlock() {
745      return range.spanInBlock;
746    }
747
748    @Override
749    public String toString() {
750      return "Offset=" + range.offsetInBlock + ", length=" + range.spanInBlock +
751          ", fetchedChunksNum=" + fetchedChunksNum +
752          ", missingChunksNum=" + missingChunksNum;
753    }
754  }

756  /**
757   * A simple utility class representing an arbitrary vertical inclusive range
758   * starting at {@link #offsetInBlock} and lasting for {@link #spanInBlock}
759   * bytes in an internal block. Note that VerticalRange doesn't necessarily
760   * align with {@link StripingCell}.
761   *
762   * |<- Striped Block Group ->|
763   *  blk_0
764   *    |
765   *    v
766   * +-----+
767   * |~~~~~| <-- {@link #offsetInBlock}
768   * |     |  ^
769   * |     |  |
770   * |     |  | {@link #spanInBlock}
771   * |     |  v
772   * |~~~~~| ---
773   * |     |
774   * +-----+
775   */
776  public static class VerticalRange {
777    /** start offset in the block group (inclusive) */
778    public long offsetInBlock;
779    /** length of the stripe range */
780    public long spanInBlock;
781
782    public VerticalRange(long offsetInBlock, long length) {
783      Preconditions.checkArgument(offsetInBlock >= 0 && length >= 0);
784      this.offsetInBlock = offsetInBlock;
785      this.spanInBlock = length;
786    }
787
788    /** whether a position is in the range */
789    public boolean include(long pos) {
790      return pos >= offsetInBlock && pos < offsetInBlock + spanInBlock;
791    }
792  }

794  /**
795   * Indicates the coverage of an {@link AlignedStripe} on an internal block,
796   * and the state of the chunk in the context of the read request.
797   *
798   * |<---------------- Striped Block Group --------------->|
799   *   blk_0        blk_1        blk_2          blk_3   blk_4
800   *                           +---------+  |  +----+  +----+
801   *     null         null     |REQUESTED|  |  |null|  |null| <- AlignedStripe0
802   *              +---------+  |---------|  |  |----|  |----|
803   *     null     |REQUESTED|  |REQUESTED|  |  |null|  |null| <- AlignedStripe1
804   * +---------+  +---------+  +---------+  |  +----+  +----+
805   * |REQUESTED|  |REQUESTED|    ALLZERO    |  |null|  |null| <- AlignedStripe2
806   * +---------+  +---------+               |  +----+  +----+
807   * <----------- data blocks ------------> | <--- parity --->
808   */
809  public static class StripingChunk {
810    /** Chunk has been successfully fetched */
811    public static final int FETCHED = 0x01;
812    /** Chunk has encountered failed when being fetched */
813    public static final int MISSING = 0x02;
814    /** Chunk being fetched (fetching task is in-flight) */
815    public static final int PENDING = 0x04;
816    /**
817     * Chunk is requested either by application or for decoding, need to
818     * schedule read task
819     */
820    public static final int REQUESTED = 0X08;
821    /**
822     * Internal block is short and has no overlap with chunk. Chunk considered
823     * all-zero bytes in codec calculations.
824     */
825    public static final int ALLZERO = 0X0f;
826
827    /**
828     * If a chunk is completely in requested range, the state transition is:
829     * REQUESTED (when AlignedStripe created) -> PENDING -> {FETCHED | MISSING}
830     * If a chunk is completely outside requested range (including parity
831     * chunks), state transition is:
832     * null (AlignedStripe created) -> REQUESTED (upon failure) -> PENDING ...
833     */
834    public int state = REQUESTED;
835
836    public final ChunkByteArray byteArray;
837    public final ByteBuffer byteBuffer;
838
839    public StripingChunk(byte[] buf) {
840      this.byteArray = new ChunkByteArray(buf);
841      byteBuffer = null;
842    }
843
844    public StripingChunk(ByteBuffer buf) {
845      this.byteArray = null;
846      this.byteBuffer = buf;
847    }
848
849    public StripingChunk(int state) {
850      this.byteArray = null;
851      this.byteBuffer = null;
852      this.state = state;
853    }
854
855    public void addByteArraySlice(int offset, int length) {
856      assert byteArray != null;
857      byteArray.offsetsInBuf.add(offset);
858      byteArray.lengthsInBuf.add(length);
859    }
860
861    void copyTo(byte[] target) {
862      assert byteArray != null;
863      byteArray.copyTo(target);
864    }
865
866    void copyFrom(byte[] src) {
867      assert byteArray != null;
868      byteArray.copyFrom(src);
869    }
870  }
871
872  public static class ChunkByteArray {
873    private final byte[] buf;
874    private final List<Integer> offsetsInBuf;
875    private final List<Integer> lengthsInBuf;
876
877    ChunkByteArray(byte[] buf) {
878      this.buf = buf;
879      this.offsetsInBuf = new ArrayList<>();
880      this.lengthsInBuf = new ArrayList<>();
881    }
882
883    public int[] getOffsets() {
884      int[] offsets = new int[offsetsInBuf.size()];
885      for (int i = 0; i < offsets.length; i++) {
886        offsets[i] = offsetsInBuf.get(i);
887      }
888      return offsets;
889    }
890
891    public int[] getLengths() {
892      int[] lens = new int[this.lengthsInBuf.size()];
893      for (int i = 0; i < lens.length; i++) {
894        lens[i] = this.lengthsInBuf.get(i);
895      }
896      return lens;
897    }
898
899    public byte[] buf() {
900      return buf;
901    }
902
903    void copyTo(byte[] target) {
904      int posInBuf = 0;
905      for (int i = 0; i < offsetsInBuf.size(); i++) {
906        System.arraycopy(buf, offsetsInBuf.get(i),
907            target, posInBuf, lengthsInBuf.get(i));
908        posInBuf += lengthsInBuf.get(i);
909      }
910    }
911
912    void copyFrom(byte[] src) {
913      int srcPos = 0;
914      for (int j = 0; j < offsetsInBuf.size(); j++) {
915        System.arraycopy(src, srcPos, buf, offsetsInBuf.get(j),
916            lengthsInBuf.get(j));
917        srcPos += lengthsInBuf.get(j);
918      }
919    }
920  }

956  /**
957   * Check if the information such as IDs and generation stamps in block-i
958   * match the block group.
959   */
960  public static void checkBlocks(ExtendedBlock blockGroup,
961      int i, ExtendedBlock blocki) throws IOException {
962    if (!blocki.getBlockPoolId().equals(blockGroup.getBlockPoolId())) {
963      throw new IOException("Block pool IDs mismatched: block" + i + "="
964          + blocki + ", expected block group=" + blockGroup);
965    }
966    if (blocki.getBlockId() - i != blockGroup.getBlockId()) {
967      throw new IOException("Block IDs mismatched: block" + i + "="
968          + blocki + ", expected block group=" + blockGroup);
969    }
970    if (blocki.getGenerationStamp() != blockGroup.getGenerationStamp()) {
971      throw new IOException("Generation stamps mismatched: block" + i + "="
972          + blocki + ", expected block group=" + blockGroup);
973    }
974  }
