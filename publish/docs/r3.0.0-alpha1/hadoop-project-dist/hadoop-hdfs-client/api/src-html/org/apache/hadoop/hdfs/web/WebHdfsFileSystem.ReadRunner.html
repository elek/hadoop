311  // the first getAuthParams() for a non-token op will either get the
312  // internal token from the ugi or lazy fetch one
313  protected synchronized Token<?> getDelegationToken() throws IOException {
314    if (delegationToken == null) {
315      Token<?> token = tokenSelector.selectToken(
316          new Text(getCanonicalServiceName()), ugi.getTokens());
317      // ugi tokens are usually indicative of a task which can't
318      // refetch tokens.  even if ugi has credentials, don't attempt
319      // to get another token to match hdfs/rpc behavior
320      if (token != null) {
321        LOG.debug("Using UGI token: {}", token);
322        canRefreshDelegationToken = false;
323      } else {
324        if (canRefreshDelegationToken) {
325          token = getDelegationToken(null);
326          if (token != null) {
327            LOG.debug("Fetched new token: {}", token);
328          } else { // security is disabled
329            canRefreshDelegationToken = false;
330          }
331        }
332      }
333      setDelegationToken(token);
334    }
335    return delegationToken;
336  }

416  static Map<?, ?> jsonParse(final HttpURLConnection c,
417      final boolean useErrorStream) throws IOException {
418    if (c.getContentLength() == 0) {
419      return null;
420    }
421    final InputStream in = useErrorStream ?
422        c.getErrorStream() : c.getInputStream();
423    if (in == null) {
424      throw new IOException("The " + (useErrorStream? "error": "input") +
425          " stream is null.");
426    }
427    try {
428      final String contentType = c.getContentType();
429      if (contentType != null) {
430        final MediaType parsed = MediaType.valueOf(contentType);
431        if (!MediaType.APPLICATION_JSON_TYPE.isCompatible(parsed)) {
432          throw new IOException("Content-Type \"" + contentType
433              + "\" is incompatible with \"" + MediaType.APPLICATION_JSON
434              + "\" (parsed=\"" + parsed + "\")");
435        }
436      }
437      return READER.readValue(in);
438    } finally {
439      in.close();
440    }
441  }

443  private static Map<?, ?> validateResponse(final HttpOpParam.Op op,
444      final HttpURLConnection conn, boolean unwrapException)
445      throws IOException {
446    final int code = conn.getResponseCode();
447    // server is demanding an authentication we don't support
448    if (code == HttpURLConnection.HTTP_UNAUTHORIZED) {
449      // match hdfs/rpc exception
450      throw new AccessControlException(conn.getResponseMessage());
451    }
452    if (code != op.getExpectedHttpResponseCode()) {
453      final Map<?, ?> m;
454      try {
455        m = jsonParse(conn, true);
456      } catch(Exception e) {
457        throw new IOException("Unexpected HTTP response: code=" + code + " != "
458            + op.getExpectedHttpResponseCode() + ", " + op.toQueryString()
459            + ", message=" + conn.getResponseMessage(), e);
460      }
461
462      if (m == null) {
463        throw new IOException("Unexpected HTTP response: code=" + code + " != "
464            + op.getExpectedHttpResponseCode() + ", " + op.toQueryString()
465            + ", message=" + conn.getResponseMessage());
466      } else if (m.get(RemoteException.class.getSimpleName()) == null) {
467        return m;
468      }
469
470      IOException re = JsonUtilClient.toRemoteException(m);
471      // extract UGI-related exceptions and unwrap InvalidToken
472      // the NN mangles these exceptions but the DN does not and may need
473      // to re-fetch a token if either report the token is expired
474      if (re.getMessage() != null && re.getMessage().startsWith(
475          SecurityUtil.FAILED_TO_GET_UGI_MSG_HEADER)) {
476        String[] parts = re.getMessage().split(":\\s+", 3);
477        re = new RemoteException(parts[1], parts[2]);
478        re = ((RemoteException)re).unwrapRemoteException(InvalidToken.class);
479      }
480      throw unwrapException? toIOException(re): re;
481    }
482    return null;
483  }

485  /**
486   * Covert an exception to an IOException.
487   *
488   * For a non-IOException, wrap it with IOException.
489   * For a RemoteException, unwrap it.
490   * For an IOException which is not a RemoteException, return it.
491   */
492  private static IOException toIOException(Exception e) {
493    if (!(e instanceof IOException)) {
494      return new IOException(e);
495    }
496
497    final IOException ioe = (IOException)e;
498    if (!(ioe instanceof RemoteException)) {
499      return ioe;
500    }
501
502    return ((RemoteException)ioe).unwrapRemoteException();
503  }

532  Param<?,?>[] getAuthParameters(final HttpOpParam.Op op) throws IOException {
533    List<Param<?,?>> authParams = Lists.newArrayList();
534    // Skip adding delegation token for token operations because these
535    // operations require authentication.
536    Token<?> token = null;
537    if (!op.getRequireAuth()) {
538      token = getDelegationToken();
539    }
540    if (token != null) {
541      authParams.add(new DelegationParam(token.encodeToUrlString()));
542    } else {
543      UserGroupInformation userUgi = ugi;
544      UserGroupInformation realUgi = userUgi.getRealUser();
545      if (realUgi != null) { // proxy user
546        authParams.add(new DoAsParam(userUgi.getShortUserName()));
547        userUgi = realUgi;
548      }
549      authParams.add(new UserParam(userUgi.getShortUserName()));
550    }
551    return authParams.toArray(new Param<?,?>[0]);
552  }

555      final Param<?,?>... parameters) throws IOException {
556    //initialize URI path and query
557    final String path = PATH_PREFIX
558        + (fspath == null? "/": makeQualified(fspath).toUri().getRawPath());
559    final String query = op.toQueryString()
560        + Param.toSortedString("&", getAuthParameters(op))
561        + Param.toSortedString("&", parameters);
562    final URL url = getNamenodeURL(path, query);
563    LOG.trace("url={}", url);
564    return url;
565  }

568   * This class is for initialing a HTTP connection, connecting to server,
569   * obtaining a response, and also handling retry on failures.

571  abstract class AbstractRunner<T> {
572    abstract protected URL getUrl() throws IOException;
573
574    protected final HttpOpParam.Op op;
575    private final boolean redirected;
576    protected ExcludeDatanodesParam excludeDatanodes =
577        new ExcludeDatanodesParam("");
578
579    private boolean checkRetry;
580    private String redirectHost;
581
582    protected AbstractRunner(final HttpOpParam.Op op, boolean redirected) {
583      this.op = op;
584      this.redirected = redirected;
585    }
586
587    T run() throws IOException {
588      UserGroupInformation connectUgi = ugi.getRealUser();
589      if (connectUgi == null) {
590        connectUgi = ugi;
591      }
592      if (op.getRequireAuth()) {
593        connectUgi.checkTGTAndReloginFromKeytab();
594      }
595      try {
596        // the entire lifecycle of the connection must be run inside the
597        // doAs to ensure authentication is performed correctly
598        return connectUgi.doAs(
599            new PrivilegedExceptionAction<T>() {
600              @Override
601              public T run() throws IOException {
602                return runWithRetry();
603              }
604            });
605      } catch (InterruptedException e) {
606        throw new IOException(e);
607      }
608    }

610    /**
611     * Two-step requests redirected to a DN
612     *
613     * Create/Append:
614     * Step 1) Submit a Http request with neither auto-redirect nor data.
615     * Step 2) Submit another Http request with the URL from the Location header
616     * with data.
617     *
618     * The reason of having two-step create/append is for preventing clients to
619     * send out the data before the redirect. This issue is addressed by the
620     * "Expect: 100-continue" header in HTTP/1.1; see RFC 2616, Section 8.2.3.
621     * Unfortunately, there are software library bugs (e.g. Jetty 6 http server
622     * and Java 6 http client), which do not correctly implement "Expect:
623     * 100-continue". The two-step create/append is a temporary workaround for
624     * the software library bugs.
625     *
626     * Open/Checksum
627     * Also implements two-step connects for other operations redirected to
628     * a DN such as open and checksum
629     */
630    protected HttpURLConnection connect(URL url) throws IOException {
631      //redirect hostname and port
632      redirectHost = null;
633
634
635      // resolve redirects for a DN operation unless already resolved
636      if (op.getRedirect() && !redirected) {
637        final HttpOpParam.Op redirectOp =
638            HttpOpParam.TemporaryRedirectOp.valueOf(op);
639        final HttpURLConnection conn = connect(redirectOp, url);
640        // application level proxy like httpfs might not issue a redirect
641        if (conn.getResponseCode() == op.getExpectedHttpResponseCode()) {
642          return conn;
643        }
644        try {
645          validateResponse(redirectOp, conn, false);
646          url = new URL(conn.getHeaderField("Location"));
647          redirectHost = url.getHost() + ":" + url.getPort();
648        } finally {
649          conn.disconnect();
650        }
651      }
652      try {
653        return connect(op, url);
654      } catch (IOException ioe) {
655        if (redirectHost != null) {
656          if (excludeDatanodes.getValue() != null) {
657            excludeDatanodes = new ExcludeDatanodesParam(redirectHost + ","
658                + excludeDatanodes.getValue());
659          } else {
660            excludeDatanodes = new ExcludeDatanodesParam(redirectHost);
661          }
662        }
663        throw ioe;
664      }
665    }
666
667    private HttpURLConnection connect(final HttpOpParam.Op op, final URL url)
668        throws IOException {
669      final HttpURLConnection conn =
670          (HttpURLConnection)connectionFactory.openConnection(url);
671      final boolean doOutput = op.getDoOutput();
672      conn.setRequestMethod(op.getType().toString());
673      conn.setInstanceFollowRedirects(false);
674      if (restCsrfCustomHeader != null &&
675          !restCsrfMethodsToIgnore.contains(op.getType().name())) {
676        // The value of the header is unimportant.  Only its presence matters.
677        conn.setRequestProperty(restCsrfCustomHeader, "\"\"");
678      }
679      switch (op.getType()) {
680      // if not sending a message body for a POST or PUT operation, need
681      // to ensure the server/proxy knows this
682      case POST:
683      case PUT: {
684        conn.setDoOutput(true);
685        if (!doOutput) {
686          // explicitly setting content-length to 0 won't do spnego!!
687          // opening and closing the stream will send "Content-Length: 0"
688          conn.getOutputStream().close();
689        } else {
690          conn.setRequestProperty("Content-Type",
691              MediaType.APPLICATION_OCTET_STREAM);
692          conn.setChunkedStreamingMode(32 << 10); //32kB-chunk
693        }
694        break;
695      }
696      default:
697        conn.setDoOutput(doOutput);
698        break;
699      }
700      conn.connect();
701      return conn;
702    }
703
704    private T runWithRetry() throws IOException {
705      /**
706       * Do the real work.
707       *
708       * There are three cases that the code inside the loop can throw an
709       * IOException:
710       *
711       * <ul>
712       * <li>The connection has failed (e.g., ConnectException,
713       * @see FailoverOnNetworkExceptionRetry for more details)</li>
714       * <li>The namenode enters the standby state (i.e., StandbyException).</li>
715       * <li>The server returns errors for the command (i.e., RemoteException)</li>
716       * </ul>
717       *
718       * The call to shouldRetry() will conduct the retry policy. The policy
719       * examines the exception and swallows it if it decides to rerun the work.
720       */
721      for(int retry = 0; ; retry++) {
722        checkRetry = !redirected;
723        final URL url = getUrl();
724        try {
725          final HttpURLConnection conn = connect(url);
726          // output streams will validate on close
727          if (!op.getDoOutput()) {
728            validateResponse(op, conn, false);
729          }
730          return getResponse(conn);
731        } catch (AccessControlException ace) {
732          // no retries for auth failures
733          throw ace;
734        } catch (InvalidToken it) {
735          // try to replace the expired token with a new one.  the attempt
736          // to acquire a new token must be outside this operation's retry
737          // so if it fails after its own retries, this operation fails too.
738          if (op.getRequireAuth() || !replaceExpiredDelegationToken()) {
739            throw it;
740          }
741        } catch (IOException ioe) {
742          // Attempt to include the redirected node in the exception. If the
743          // attempt to recreate the exception fails, just use the original.
744          String node = redirectHost;
745          if (node == null) {
746            node = url.getAuthority();
747          }
748          try {
749            IOException newIoe = ioe.getClass().getConstructor(String.class)
750                .newInstance(node + ": " + ioe.getMessage());
751            newIoe.setStackTrace(ioe.getStackTrace());
752            ioe = newIoe;
753          } catch (NoSuchMethodException | SecurityException 
754                   | InstantiationException | IllegalAccessException
755                   | IllegalArgumentException | InvocationTargetException e) {
756          }
757          shouldRetry(ioe, retry);
758        }
759      }
760    }
761
762    private void shouldRetry(final IOException ioe, final int retry
763    ) throws IOException {
764      InetSocketAddress nnAddr = getCurrentNNAddr();
765      if (checkRetry) {
766        try {
767          final RetryPolicy.RetryAction a = retryPolicy.shouldRetry(
768              ioe, retry, 0, true);
769
770          boolean isRetry =
771              a.action == RetryPolicy.RetryAction.RetryDecision.RETRY;
772          boolean isFailoverAndRetry =
773              a.action == RetryPolicy.RetryAction.RetryDecision.FAILOVER_AND_RETRY;
774
775          if (isRetry || isFailoverAndRetry) {
776            LOG.info("Retrying connect to namenode: {}. Already tried {}"
777                    + " time(s); retry policy is {}, delay {}ms.",
778                nnAddr, retry, retryPolicy, a.delayMillis);
779
780            if (isFailoverAndRetry) {
781              resetStateToFailOver();
782            }
783
784            Thread.sleep(a.delayMillis);
785            return;
786          }
787        } catch(Exception e) {
788          LOG.warn("Original exception is ", ioe);
789          throw toIOException(e);
790        }
791      }
792      throw toIOException(ioe);
793    }

882      }

1427  private static final String OFFSET_PARAM_PREFIX = OffsetParam.NAME + "=";
1428
1429  /** Remove offset parameter, if there is any, from the url */
1430  static URL removeOffsetParam(final URL url) throws MalformedURLException {
1431    String query = url.getQuery();
1432    if (query == null) {
1433      return url;
1434    }
1435    final String lower = StringUtils.toLowerCase(query);
1436    if (!lower.startsWith(OFFSET_PARAM_PREFIX)
1437        && !lower.contains("&" + OFFSET_PARAM_PREFIX)) {
1438      return url;
1439    }
1440
1441    //rebuild query
1442    StringBuilder b = null;
1443    for(final StringTokenizer st = new StringTokenizer(query, "&");
1444        st.hasMoreTokens();) {
1445      final String token = st.nextToken();
1446      if (!StringUtils.toLowerCase(token).startsWith(OFFSET_PARAM_PREFIX)) {
1447        if (b == null) {
1448          b = new StringBuilder("?").append(token);
1449        } else {
1450          b.append('&').append(token);
1451        }
1452      }
1453    }
1454    query = b == null? "": b.toString();
1455
1456    final String urlStr = url.toString();
1457    return new URL(urlStr.substring(0, urlStr.indexOf('?')) + query);
1458  }

1749  /**
1750   * This class will allow retries to occur for both open and read operations.
1751   * The first WebHdfsFileSystem#open creates a new WebHdfsInputStream object,
1752   * which creates a new ReadRunner object that will be used to open a
1753   * connection and read or seek into the input stream.
1754   *
1755   * ReadRunner is a subclass of the AbstractRunner class, which will run the
1756   * ReadRunner#getUrl(), ReadRunner#connect(URL), and ReadRunner#getResponse
1757   * methods within a retry loop, based on the configured retry policy.
1758   * ReadRunner#connect will create a connection if one has not already been
1759   * created. Otherwise, it will return the previously created connection
1760   * object. This is necessary because a new connection should not be created
1761   * for every read.
1762   * Likewise, ReadRunner#getUrl will construct a new URL object only if the
1763   * connection has not previously been established. Otherwise, it will return
1764   * the previously created URL object.
1765   * ReadRunner#getResponse will initialize the input stream if it has not
1766   * already been initialized and read the requested data from the specified
1767   * input stream.
1768   */
1769  @VisibleForTesting
1770  protected class ReadRunner extends AbstractFsPathRunner<Integer> {
1771    private InputStream in = null;
1772    private HttpURLConnection cachedConnection = null;
1773    private byte[] readBuffer;
1774    private int readOffset;
1775    private int readLength;
1776    private RunnerState runnerState = RunnerState.DISCONNECTED;
1777    private URL originalUrl = null;
1778    private URL resolvedUrl = null;
1779
1780    private final Path path;
1781    private final int bufferSize;
1782    private long pos = 0;
1783    private long fileLength = 0;
1784
1785    /* The following methods are WebHdfsInputStream helpers. */
1786
1787    ReadRunner(Path p, int bs) throws IOException {
1788      super(GetOpParam.Op.OPEN, p, new BufferSizeParam(bs));
1789      this.path = p;
1790      this.bufferSize = bs;
1791    }
1792
1793    int read(byte[] b, int off, int len) throws IOException {
1794      if (runnerState == RunnerState.CLOSED) {
1795        throw new IOException("Stream closed");
1796      }
1797
1798      // Before the first read, pos and fileLength will be 0 and readBuffer
1799      // will all be null. They will be initialized once the first connection
1800      // is made. Only after that it makes sense to compare pos and fileLength.
1801      if (pos >= fileLength && readBuffer != null) {
1802        return -1;
1803      }
1804
1805      // If a seek is occurring, the input stream will have been closed, so it
1806      // needs to be reopened. Use the URLRunner to call AbstractRunner#connect
1807      // with the previously-cached resolved URL and with the 'redirected' flag
1808      // set to 'true'. The resolved URL contains the URL of the previously
1809      // opened DN as opposed to the NN. It is preferable to use the resolved
1810      // URL when creating a connection because it does not hit the NN or every
1811      // seek, nor does it open a connection to a new DN after every seek.
1812      // The redirect flag is needed so that AbstractRunner#connect knows the
1813      // URL is already resolved.
1814      // Note that when the redirected flag is set, retries are not attempted.
1815      // So, if the connection fails using URLRunner, clear out the connection
1816      // and fall through to establish the connection using ReadRunner.
1817      if (runnerState == RunnerState.SEEK) {
1818        try {
1819          final URL rurl = new URL(resolvedUrl + "&" + new OffsetParam(pos));
1820          cachedConnection = new URLRunner(GetOpParam.Op.OPEN, rurl, true).run();
1821        } catch (IOException ioe) {
1822          closeInputStream(RunnerState.DISCONNECTED);
1823        }
1824      }
1825
1826      readBuffer = b;
1827      readOffset = off;
1828      readLength = len;
1829
1830      int count = -1;
1831      count = this.run();
1832      if (count >= 0) {
1833        statistics.incrementBytesRead(count);
1834        pos += count;
1835      } else if (pos < fileLength) {
1836        throw new EOFException(
1837                  "Premature EOF: pos=" + pos + " < filelength=" + fileLength);
1838      }
1839      return count;
1840    }
1841
1842    void seek(long newPos) throws IOException {
1843      if (pos != newPos) {
1844        pos = newPos;
1845        closeInputStream(RunnerState.SEEK);
1846      }
1847    }
1848
1849    public void close() throws IOException {
1850      closeInputStream(RunnerState.CLOSED);
1851    }

1853    /* The following methods are overriding AbstractRunner methods,
1854     * to be called within the retry policy context by runWithRetry.
1855     */
1856
1857    @Override
1858    protected URL getUrl() throws IOException {
1859      // This method is called every time either a read is executed.
1860      // The check for connection == null is to ensure that a new URL is only
1861      // created upon a new connection and not for every read.
1862      if (cachedConnection == null) {
1863        // Update URL with current offset. BufferSize doesn't change, but it
1864        // still must be included when creating the new URL.
1865        updateURLParameters(new BufferSizeParam(bufferSize),
1866            new OffsetParam(pos));
1867        originalUrl = super.getUrl();
1868      }
1869      return originalUrl;
1870    }
1871
1872    /* Only make the connection if it is not already open. Don't cache the
1873     * connection here. After this method is called, runWithRetry will call
1874     * validateResponse, and then call the below ReadRunner#getResponse. If
1875     * the code path makes it that far, then we can cache the connection.
1876     */
1877    @Override
1878    protected HttpURLConnection connect(URL url) throws IOException {
1879      HttpURLConnection conn = cachedConnection;
1880      if (conn == null) {
1881        try {
1882          conn = super.connect(url);
1883        } catch (IOException e) {
1884          closeInputStream(RunnerState.DISCONNECTED);
1885          throw e;
1886        }
1887      }
1888      return conn;
1889    }
1890
1891    /*
1892     * This method is used to perform reads within the retry policy context.
1893     * This code is relying on runWithRetry to always call the above connect
1894     * method and the verifyResponse method prior to calling getResponse.
1895     */
1896    @Override
1897    Integer getResponse(final HttpURLConnection conn)
1898        throws IOException {
1899      try {
1900        // In the "open-then-read" use case, runWithRetry will have executed
1901        // ReadRunner#connect to make the connection and then executed
1902        // validateResponse to validate the response code. Only then do we want
1903        // to cache the connection.
1904        // In the "read-after-seek" use case, the connection is made and the
1905        // response is validated by the URLRunner. ReadRunner#read then caches
1906        // the connection and the ReadRunner#connect will pass on the cached
1907        // connection
1908        // In either case, stream initialization is done here if necessary.
1909        cachedConnection = conn;
1910        if (in == null) {
1911          in = initializeInputStream(conn);
1912        }
1913
1914        int count = in.read(readBuffer, readOffset, readLength);
1915        if (count < 0 && pos < fileLength) {
1916          throw new EOFException(
1917                  "Premature EOF: pos=" + pos + " < filelength=" + fileLength);
1918        }
1919        return Integer.valueOf(count);
1920      } catch (IOException e) {
1921        String redirectHost = resolvedUrl.getAuthority();
1922        if (excludeDatanodes.getValue() != null) {
1923          excludeDatanodes = new ExcludeDatanodesParam(redirectHost + ","
1924              + excludeDatanodes.getValue());
1925        } else {
1926          excludeDatanodes = new ExcludeDatanodesParam(redirectHost);
1927        }
1928
1929        // If an exception occurs, close the input stream and null it out so
1930        // that if the abstract runner decides to retry, it will reconnect.
1931        closeInputStream(RunnerState.DISCONNECTED);
1932        throw e;
1933      }
1934    }

1936    @VisibleForTesting
1937    InputStream initializeInputStream(HttpURLConnection conn)
1938        throws IOException {
1939      // Cache the resolved URL so that it can be used in the event of
1940      // a future seek operation.
1941      resolvedUrl = removeOffsetParam(conn.getURL());
1942      final String cl = conn.getHeaderField(HttpHeaders.CONTENT_LENGTH);
1943      InputStream inStream = conn.getInputStream();
1944      if (LOG.isDebugEnabled()) {
1945        LOG.debug("open file: " + conn.getURL());
1946      }
1947      if (cl != null) {
1948        long streamLength = Long.parseLong(cl);
1949        fileLength = pos + streamLength;
1950        // Java has a bug with >2GB request streams.  It won't bounds check
1951        // the reads so the transfer blocks until the server times out
1952        inStream = new BoundedInputStream(inStream, streamLength);
1953      } else {
1954        fileLength = getHdfsFileStatus(path).getLen();
1955      }
1956      // Wrapping in BufferedInputStream because it is more performant than
1957      // BoundedInputStream by itself.
1958      runnerState = RunnerState.OPEN;
1959      return new BufferedInputStream(inStream, bufferSize);
1960    }

1962    // Close both the InputStream and the connection.
1963    @VisibleForTesting
1964    void closeInputStream(RunnerState rs) throws IOException {
1965      if (in != null) {
1966        IOUtils.close(cachedConnection);
1967        in = null;
1968      }
1969      cachedConnection = null;
1970      runnerState = rs;
1971    }
