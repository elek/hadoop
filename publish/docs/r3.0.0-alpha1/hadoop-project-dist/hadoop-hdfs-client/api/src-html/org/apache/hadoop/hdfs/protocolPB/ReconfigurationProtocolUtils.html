019package org.apache.hadoop.hdfs.protocolPB;
020
021import java.util.Map;
022
023import org.apache.hadoop.conf.ReconfigurationTaskStatus;
024import org.apache.hadoop.conf.ReconfigurationUtil.PropertyChange;
025import org.apache.hadoop.hdfs.protocol.proto.ReconfigurationProtocolProtos.GetReconfigurationStatusConfigChangeProto;
026import org.apache.hadoop.hdfs.protocol.proto.ReconfigurationProtocolProtos.GetReconfigurationStatusResponseProto;
027
028import com.google.common.base.Optional;
029import com.google.common.collect.Maps;
030
031/**
032 * This is a client side utility class that handles
033 * common logic to to parameter reconfiguration.
034 */
035public final class ReconfigurationProtocolUtils {
036  private ReconfigurationProtocolUtils() {
037  }
038
039  public static ReconfigurationTaskStatus getReconfigurationStatus(
040      GetReconfigurationStatusResponseProto response) {
041    Map<PropertyChange, Optional<String>> statusMap = null;
042    long startTime;
043    long endTime = 0;
044
045    startTime = response.getStartTime();
046    if (response.hasEndTime()) {
047      endTime = response.getEndTime();
048    }
049    if (response.getChangesCount() > 0) {
050      statusMap = Maps.newHashMap();
051      for (GetReconfigurationStatusConfigChangeProto change : response
052          .getChangesList()) {
053        PropertyChange pc = new PropertyChange(change.getName(),
054            change.getNewValue(), change.getOldValue());
055        String errorMessage = null;
056        if (change.hasErrorMessage()) {
057          errorMessage = change.getErrorMessage();
058        }
059        statusMap.put(pc, Optional.fromNullable(errorMessage));
060      }
061    }
062    return new ReconfigurationTaskStatus(startTime, endTime, statusMap);
063  }
064}
